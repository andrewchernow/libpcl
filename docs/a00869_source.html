38 pcl_array_t *
39 pcl_proc_parsecmd(const pchar_t *shell_cmd)
40 {
41  const pchar_t *start;
42  bool in_quote = false;
43 
44  if(!shell_cmd)
45  return R_SETERR(NULL, PCL_EINVAL);
46 
47  shell_cmd = pcl_pcsskipws(shell_cmd);
48  if(*shell_cmd == '"')
49  {
50  in_quote = true;
51  shell_cmd++;
52  }
53 
54  start = shell_cmd;
55  pcl_array_t *arr = pcl_array(4, pcl_cleanup_ptr);
56 
57  while(true)
58  {
59  switch(*shell_cmd)
60  {
61  case '\\':
62  shell_cmd += 2;
63  break;
64 
65  case '\0':
66  case '"':
67  case ' ':
68  case '\t':
69  {
70  size_t n;
71 
72  if(*shell_cmd == '"')
73  {
74  if(!in_quote)
75  {
76  pcl_array_free(arr);
77  return R_SETERRMSG(NULL, PCL_ESYNTAX, "unescaped quote", 0);
78  }
79 
80  in_quote = false;
81  }
82  else if(in_quote)
83  {
84  if(*shell_cmd == '\0')
85  {
86  pcl_array_free(arr);
87  return R_SETERRMSG(NULL, PCL_ESYNTAX, "unterminated quote", 0);
88  }
89 
90  shell_cmd++;
91  break;
92  }
93 
94  n = shell_cmd - start;
95 
96  if(n > 0)
97  pcl_array_append(arr, pcl_pcsndup(start, shell_cmd - start));
98 
99  if(*shell_cmd == '\0')
100  goto finish;
101 
102  shell_cmd = pcl_pcsskipws(++shell_cmd);
103 
104  if(*shell_cmd == '"')
105  {
106  in_quote = true;
107  shell_cmd++;
108  }
109 
110  start = shell_cmd;
111  break;
112  }
113 
114  default:
115  shell_cmd++;
116  break;
117  }
118  }
119 
120 finish:
121 
122  /* make sure to NULL terminate array */
123  pcl_array_append(arr, NULL);
124 
125  /* don't include NULL in element count */
126  arr->count--;
127 
128  return arr;
129 }
