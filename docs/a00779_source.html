36 pcl_json_t *
37 pcl_json_str(char *str, uint32_t flags)
38 {
39  if(!str)
40  {
41  if(flags & PCL_JSON_ALLOWNULL)
42  return pcl_json_null();
43 
44  return R_SETERR(NULL, PCL_EINVAL);
45  }
46 
47  size_t len = strlen(str);
48 
49  if(!(flags & PCL_JSON_SKIPUTF8CHK) && ipcl_json_utf8check(str, len) < 0)
50  return R_TRC(NULL);
51 
52  if((flags & PCL_JSON_EMPTYASNULL) && len == 0)
53  return pcl_json_null();
54 
55  pcl_json_t *val = pcl_malloc(sizeof(pcl_json_t));
56 
57  val->type = 's';
58  val->nrefs = 1;
59  val->string = (flags & PCL_JSON_SHALLOW) ? str : pcl_strndup(str, len);
60 
61  return val;
62 }
