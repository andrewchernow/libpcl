45 static void
46 convert_times(pcl_stat_t *buf, sys_stat_t *os_buf)
47 {
48  /* Linux stat uses a macro to provide backward compatability with the old st_[a|m|c]time naming.
49  * If that macro is defined, we are being compiled to use st_atim vs. st_atimensec.
50  *
51  * Note: Darwin has both but the way we compile it uses the xtimespec struct member.
52  */
53 
54  /* convert to nanoseconds */
55  buf->atime.sec = os_buf->st_atime;
56 #ifdef PCL_DARWIN
57  buf->atime.nsec = (int) os_buf->st_atimespec.tv_nsec;
58 #elif defined(st_atime)
59  buf->atime.nsec = (int) os_buf->st_atim.tv_nsec;
60 #else
61  buf->atime.nsec = (int) os_buf->st_atimensec;
62 #endif
63 
64  /* convert to nanoseconds */
65  buf->mtime.sec = os_buf->st_mtime;
66 #ifdef PCL_DARWIN
67  buf->mtime.nsec = (int) os_buf->st_mtimespec.tv_nsec;
68 #elif defined(st_mtime)
69  buf->mtime.nsec = (int) os_buf->st_mtim.tv_nsec;
70 #else
71  buf->mtime.nsec = (int) os_buf->st_mtimensec;
72 #endif
73 
74  /* convert to nanoseconds */
75  buf->ctime.sec = os_buf->st_ctime;
76 #ifdef PCL_DARWIN
77  buf->ctime.nsec = (int) os_buf->st_ctimespec.tv_nsec;
78 #elif defined(st_ctime)
79  buf->ctime.nsec = (int) os_buf->st_ctim.tv_nsec;
80 #else
81  buf->ctime.nsec = (int) os_buf->st_ctimensec;
82 #endif
83 
84 #ifdef PCL_DARWIN
85  buf->btime.sec = os_buf->st_birthtimespec.tv_sec;
86  buf->btime.nsec = (int) os_buf->st_birthtimespec.tv_nsec;
87 #endif
88 }

91 int
92 ipcl_statent(const pchar_t *path, int fd, pcl_stat_t *buf, int flags)
93 {
94  int n;
95 
96  if(strempty(path) && fd < 0)
97  return BADARG();
98 
99 #ifdef HAVE_STATX // linux only call
100  struct statx osx;
101 
102  if(fd > -1)
103  n = statx(fd, "", AT_EMPTY_PATH, STATX_ALL, &osx);
104  else if(flags & PCL_STAT_LINK)
105  n = statx(AT_FDCWD, path, AT_SYMLINK_NOFOLLOW, STATX_ALL, &osx);
106  else
107  n = statx(AT_FDCWD, path, 0, STATX_ALL, &osx);
108 
109  if(n)
110  return SETLASTERR();
111 
112  if(!buf)
113  return 0; // exists check only
114 
115  buf->size = (int64_t) osx.stx_size;
116  buf->rdev = makedev(osx.stx_rdev_major, osx.stx_rdev_minor);
117  buf->dev = makedev(osx.stx_dev_major, osx.stx_dev_minor);
118  buf->ino = osx.stx_ino;
119  buf->uid = osx.stx_uid;
120  buf->gid = osx.stx_gid;
121  buf->nlink = osx.stx_nlink;
122  buf->mode = osx.stx_mode;
123  buf->blksize = osx.stx_blksize;
124  buf->blocks = osx.stx_blocks;
125  buf->atime.sec = osx.stx_atime.tv_sec;
126  buf->atime.nsec = osx.stx_atime.tv_nsec;
127  buf->mtime.sec = osx.stx_mtime.tv_sec;
128  buf->mtime.nsec = osx.stx_mtime.tv_nsec;
129  buf->ctime.sec = osx.stx_ctime.tv_sec;
130  buf->ctime.nsec = osx.stx_ctime.tv_nsec;
131  buf->btime.sec = osx.stx_btime.tv_sec;
132  buf->btime.nsec = osx.stx_btime.tv_nsec;
133 
134  uint64_t attrs = osx.stx_attributes & osx.stx_attributes_mask;
135 
136  if(attrs & STATX_ATTR_COMPRESSED)
137  buf->flags |= UF_COMPRESSED;
138  if(attrs & STATX_ATTR_IMMUTABLE)
139  buf->flags |= UF_IMMUTABLE;
140  if(attrs & STATX_ATTR_APPEND)
141  buf->flags |= UF_APPEND;
142  if(attrs & STATX_ATTR_NODUMP)
143  buf->flags |= UF_NODUMP;
144  if(attrs & STATX_ATTR_ENCRYPTED)
145  buf->flags |= UF_ENCRYPTED;
146 
147 #else
148  sys_stat_t os;
149 
150  if(fd > -1)
151  n = sys_fstat(fd, &os);
152  else if(flags & PCL_STAT_LINK)
153  n = sys_lstat(path, &os);
154  else
155  n = sys_stat(path, &os);
156 
157  if(n)
158  return SETLASTERR();
159 
160  if(!buf)
161  return 0; // exists check only
162 
163  buf->size = (int64_t) os.st_size;
164  buf->rdev = os.st_rdev;
165  buf->dev = os.st_dev;
166  buf->ino = os.st_ino;
167  buf->uid = os.st_uid;
168  buf->gid = os.st_gid;
169  buf->nlink = os.st_nlink;
170  buf->mode = os.st_mode;
171  buf->blksize = os.st_blksize;
172  buf->blocks = os.st_blocks;
173 
174  convert_times(buf, &os);
175 
176 # ifdef PCL_DARWIN
177  buf->flags |= os.st_flags;
178 # endif
179 #endif
180 
181  buf->dsize = roundup(buf->size, buf->blksize);
182 
183  if(path)
184  {
185  const pchar_t *name = pcl_pcsrchr(path, PCL_PPATHSEPCHAR);
186 
187  if(name)
188  name++;
189  else
190  name = path;
191 
192  if(*name == '.')
193  buf->flags |= UF_HIDDEN;
194  }
195 
196  return 0;
197 }
