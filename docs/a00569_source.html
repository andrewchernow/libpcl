40 FILE *
41 pcl_fdopen(pcl_file_t *file)
42 {
43  FILE *stream;
44  const char *mode;
45 
46  if(!file)
47  {
48  SETERR(PCL_EINVAL);
49  return NULL;
50  }
51 
52  if((file->flags & PCL_FF_RDWR) == PCL_FF_RDWR)
53  mode = "w+";
54  else if(file->flags & PCL_FF_WRITE)
55  mode = "w";
56  else
57  mode = "r";
58 
59 #ifdef PCL_WINDOWS
60  /* convert HANDLE to an 'fd' */
61  int fd = _open_osfhandle((intptr_t)file->fd, 0);
62 
63  if(fd == -1)
64  return R_SETOSERR(NULL, pcl_err_crt2os(errno));
65 
66  /* convert an fd to a 'FILE*', stream */
67  if(!(stream = _fdopen(fd, mode)))
68  {
69  HANDLE h = file->fd;
70 
71  FREEZE_ERR
72  (
73  /* CRT _close() will pass our handle to CloseHandle(). Need to
74  * increase its ref count so we can safely cleanup the CRT fd
75  * w/o closing file->fd (which is the file's win32 HANDLE).
76  */
77  DuplicateHandle(GetCurrentProcess(), h, GetCurrentProcess(), &file->fd,
78  0, false, DUPLICATE_SAME_ACCESS);
79 
80  _close(fd);
81  );
82 
83  return R_SETOSERR(NULL, pcl_err_crt2os(errno));
84  }
85 
86 #else
87  if(!(stream = fdopen(file->fd, mode)))
88  {
89  SETLASTERR();
90  return NULL;
91  }
92 #endif
93 
94  /* stole the handle, nuke the file */
95  file->fd = PCL_BADFD;
96  pcl_file_close(file);
97 
98  return stream;
99 }
