41 pcl_dir_t *
42 pcl_opendir(const pchar_t *path)
43 {
44  if(strempty(path))
45  return R_SETERR(NULL, PCL_EINVAL);
46 
47  int len;
48  pchar_t *fullpath = pcl_realpath(path, &len);
49 
50  if(!fullpath)
51  return R_TRC(NULL);
52 
53  pcl_dir_t *dir = pcl_malloc(sizeof(pcl_dir_t));
54  dir->path = fullpath;
55  dir->pathlen = len;
56 
57 #ifdef PCL_WINDOWS
58  dir->path = pcl_realloc(dir->path, (dir->pathlen + 3) * sizeof(pchar_t));
59 
60  /* windows requires a "\*" at the end of path */
61  if(dir->path[len - 1] != PCL_PPATHSEPCHAR)
62  dir->path[len++] = PCL_PPATHSEPCHAR;
63 
64  dir->path[len++] = _P('*');
65  dir->path[len] = 0;
66 
67  /* FindExInfoBasic excludes short file names increasing performance. Also,
68  * FIND_FIRST_EX_LARGE_FETCH uses larger buffers increasing performance.
69  */
70  dir->handle = FindFirstFileEx(dir->path, FindExInfoBasic, &dir->wfd,
71  FindExSearchNameMatch, NULL, FIND_FIRST_EX_LARGE_FETCH);
72 
73  /* remove the trailing "\*" appended above */
74  dir->path[dir->pathlen] = 0;
75 
76  if(dir->handle == INVALID_HANDLE_VALUE)
77  {
78  SETLASTERRMSG("Cannot open directory: %Ps", path);
79  pcl_closedir(dir);
80  return NULL;
81  }
82 
83  dir->findnextfile = false;
84 
85 #else
86  dir->handle = opendir(dir->path);
87 
88  if(!dir->handle)
89  {
90  SETLASTERRMSG("Cannot open directory: %Ps", path);
91  pcl_closedir(dir);
92  return NULL;
93  }
94 #endif
95 
96  return dir;
97 }
