37 ipcl_err_serialize(int indent, const char *format, va_list ap)
38 {
39  pcl_err_t *err = pcl_err_get();
40  pcl_err_trace_t *t = err->strace;
41 
42  if(!t || err->err == PCL_EOKAY)
43  return NULL;
44 
45  char line[24];
46  bool hasfile, hasfunc, hasmsg;
47  pcl_buf_t *b = err->buffer;
48 
49  if(!b)
50  b = err->buffer = pcl_buf_init(NULL, 512, PclBufText);
51  else
52  pcl_buf_reset(b); // resue it (again, per-thread)
53 
54  bool panic = indent == -1;
55 
56  indent = max(0, indent);
57 
58  if(!strempty(format))
59  {
60  const char *prefix = panic ? "PANIC: " : "ERROR: ";
61 
62  pcl_buf_putf(b, "%*s%s", indent, indent ? " " : "", prefix);
63  pcl_buf_vputf(b, format, ap);
64 
65  /* ensure message is terminated with LF */
66  if(b->data[b->len - 1] != '\n')
67  pcl_buf_putchar(b, '\n');
68  }
69 
70  for(; t; t = t->next, indent += 2)
71  {
72  /* don't handle tail here. It's formatted differently being the actual error site */
73  if(!t->next)
74  break;
75 
76  hasmsg = !strempty(t->msg);
77  hasfile = !strempty(t->file);
78  hasfunc = !strempty(t->func);
79 
80  *line = 0;
81  if(t->line > 0)
82  pcl_sprintf(line, sizeof(line), "(%d)", t->line);
83 
84  pcl_buf_putf(b, "%*s%s%s%s%s%s%s\n",
85  indent, indent ? " " : "",
86  hasfile ? t->file : "",
87  hasfile && hasfunc ? ":" : "", /* if we have both, include separator */
88  hasfunc ? t->func : "",
89  line,
90  hasmsg ? ": " : "",
91  hasmsg ? t->msg : "");
92  }
93 
94  /* 't' is stack trace tail, where the actual error occured */
95  hasfile = !strempty(t->file);
96  hasfunc = !strempty(t->func);
97 
98  *line = 0;
99  if(t->line > 0)
100  pcl_sprintf(line, sizeof(line), "(%d)", t->line);
101 
102  pcl_buf_putf(b, "%*s%s%s%s%s%s%s - %s\n",
103  indent, indent ? " " : "",
104  hasfile ? t->file : "",
105  hasfile && hasfunc ? ":" : "", /* if we have both, include separator */
106  hasfunc ? t->func : "",
107  line,
108  (hasfile || hasfunc || *line) ? ": " : "",
109  pcl_err_name(err->err),
110  pcl_err_msg(err->err));
111 
112  /* print actual error message, this could be empty if a context msg was not provided. */
113  if(!strempty(t->msg))
114  pcl_buf_putf(b, "%*s>%s\n", indent + 2, " ", t->msg);
115 
116  return b;
117 }
