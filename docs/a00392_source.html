39 pcl_err_setjson(pcl_json_t *error)
40 {
41  pcl_err_t *err = pcl_err_get();
42 
43  if(err->frozen)
44  return 0;
45 
46  if(!pcl_json_isobj(error))
47  return SETERR(PCL_ETYPE);
48 
49  long long errcode = pcl_json_objgetint(error, "err");
50  long long oscode = pcl_json_objgetint(error, "oserr");
51  pcl_json_t *strace = pcl_json_objget(error, "strace");
52 
53  if(errcode == PCL_JSON_INVINT || oscode == PCL_JSON_INVINT || !pcl_json_isarr(strace))
54  return SETERR(PCL_EFORMAT);
55 
56  pcl_err_clear();
57 
58  /* ignore errorrs while building pcl_err_t */
59  err->frozen = true;
60  err->err = (int) errcode;
61  err->oserr = (uint32_t) oscode;
62 
63  for(int i = pcl_json_count(strace) - 1; i >= 0; i--)
64  {
65  pcl_json_t *trc = pcl_json_arrget(strace, i);
66 
67  if(pcl_json_isobj(trc))
68  {
69  const char *file = pcl_json_objgetstr(trc, "file");
70  const char *func = pcl_json_objgetstr(trc, "func");
71  long long line = pcl_json_objgetint(trc, "line");
72  const char *msg = pcl_json_objgetstr(trc, "msg");
73 
74  if(line == PCL_JSON_INVINT)
75  line = 0;
76 
77  pcl_err_trace(file, func, (int) line, msg);
78  }
79  }
80 
81  err->frozen = false;
82  return 0;
83 }
