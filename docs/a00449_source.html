37 {
38  if(!file || file->fd == PCL_BADFD)
39  return BADARG();
40 
41  switch(operation)
42  {
43  case PCL_RDLOCK:
44  case PCL_WRLOCK:
45  case PCL_UNLOCK:
46  break;
47 
48  default:
49  return SETERRMSG(PCL_EINVAL, "Unknown lock operation: %d", operation);
50  }
51 
52 #ifdef PCL_UNIX
53  struct flock fl = {
54  .l_whence = SEEK_SET,
55  .l_start = 0,
56  .l_len = 0,
57  .l_type = (short) operation,
58  .l_pid = getpid()
59  };
60 
61  int type = file->flags & PCL_FF_NONBLOCK ? F_SETLK : F_SETLKW;
62 
63  if(fcntl(file->fd, type, &fl) == -1)
64  return SETLASTERR();
65 
66  if(fl.l_type == PCL_UNLOCK)
67  file->flags &= ~PCL_FF_FLOCK;
68  else
69  file->flags |= PCL_FF_FLOCK;
70 #else
71  OVERLAPPED ol = {0};
72  DWORD flags = 0;
73 
74  /* unlock */
75  if(operation == PCL_UNLOCK)
76  {
77  if(!UnlockFileEx(file->fd, 0, MAXDWORD, MAXDWORD, &ol))
78  return SETLASTERR();
79 
80  file->flags &= ~PCL_FF_FLOCK;
81  return 0;
82  }
83 
84  if(operation == PCL_WRLOCK)
85  flags |= LOCKFILE_EXCLUSIVE_LOCK;
86 
87  if(file->flags & PCL_FF_NONBLOCK)
88  flags |= LOCKFILE_FAIL_IMMEDIATELY;
89 
90  if(!LockFileEx(file->fd, flags, 0, MAXDWORD, MAXDWORD, &ol))
91  {
92  if((file->flags & PCL_FF_NONBLOCK) && pcl_oserrno == ERROR_IO_PENDING)
93  return SETERR(PCL_EAGAIN);
94 
95  return SETLASTERR();
96  }
97 
98  file->flags |= PCL_FF_FLOCK;
99 #endif
100 
101  return 0;
102 }
