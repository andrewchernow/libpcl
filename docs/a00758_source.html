37 static int
38 create_sslctx(pcl_ssl_t *ssl)
39 {
40  ERR_clear_error();
41 
42  const SSL_METHOD *method;
43 
44  if(ssl->flags & (PCL_SSL_PASSIVE | PCL_SSL_SERVER))
45  method = TLS_server_method();
46  else
47  method = TLS_client_method();
48 
49  ssl->ctx = SSL_CTX_new(method);
50 
51  if(!ssl->ctx)
52  return SETERRMSG(PCL_ETYPE, "Invalid TLS method: %s", sslerror(ssl));
53 
54  SSL_CTX_set_min_proto_version(ssl->ctx, TLS1_2_VERSION);
55  SSL_CTX_set_max_proto_version(ssl->ctx, TLS_MAX_VERSION);
56 
57  /* We don't need session caching. Since we have no cache, we don't need
58  * to be concerned with SSL_ctx_set_timeout's default of 300 seconds.
59  */
60  SSL_CTX_set_session_cache_mode(ssl->ctx, SSL_SESS_CACHE_OFF);
61 
62  /* Force a new session to be created for renegotiation. */
63  if(ssl->flags & PCL_SSL_PASSIVE)
64  SSL_CTX_set_options(ssl->ctx, SSL_OP_NO_SESSION_RESUMPTION_ON_RENEGOTIATION);
65 
66  SSL_CTX_set_verify(ssl->ctx, SSL_VERIFY_NONE, NULL);
67  (void)pcl_ssl_setciphlist(ssl, PCL_SSL_DEFCIPHLIST);
68 
69  return 0;
70 }

72 pcl_ssl_t *
73 pcl_ssl_create(int flags, ...)
74 {
75  pcl_ssl_t *ssl = pcl_zalloc(sizeof(pcl_ssl_t));
76 
77  ssl->flags = flags & (PCL_SSL_PASSIVE | PCL_SOCK_NONBLOCK);
78 
79  if(ssl->flags & PCL_SSL_PASSIVE)
80  {
81  va_list ap;
82  const char *host;
83  int port;
84 
85  va_start(ap, flags);
86  host = va_arg(ap, const char *);
87  port = va_arg(ap, int);
88  va_end(ap);
89 
90  if(ipcl_ssl_configure_socket(ssl, host, port))
91  {
92  pcl_ssl_free(ssl);
93  return R_TRC(NULL);
94  }
95 
96  if(pcl_bind(ssl->sock))
97  {
98  pcl_ssl_free(ssl);
99  return R_TRCMSG(NULL, "Cannot bind to port %d", port);
100  }
101 
102  if(pcl_listen(ssl->sock, 50))
103  {
104  pcl_ssl_free(ssl);
105  return R_TRCMSG(NULL, "Cannot listen on host=%s port=%d", host, port);
106  }
107  }
108 
109  if(create_sslctx(ssl))
110  {
111  pcl_ssl_free(ssl);
112  return R_TRC(NULL);
113  }
114 
115  return ssl;
116 }
