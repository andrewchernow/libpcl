35 SECURITY_DESCRIPTOR *
36 sd_get(const pchar_t *path, DWORD *len, int sec_info)
37 {
38  DWORD size;
39  SECURITY_DESCRIPTOR *sd;
40  BOOL success;
41  HANDLE file = CreateFile(path, READ_CONTROL | FILE_READ_ATTRIBUTES | GENERIC_READ | GENERIC_WRITE,
42  0, NULL, OPEN_EXISTING, FILE_FLAG_BACKUP_SEMANTICS | FILE_ATTRIBUTE_NORMAL, NULL);
43 
44  if(file == INVALID_HANDLE_VALUE)
45  return R_SETLASTERR(NULL);
46 
47  /* query for size */
48  GetKernelObjectSecurity(file, sec_info, NULL, 0, &size);
49  if(GetLastError() != ERROR_INSUFFICIENT_BUFFER)
50  {
51  CloseHandle(file);
52  return R_SETLASTERR(NULL);
53  }
54 
55  /* Allocate SECURITY_DESCRIPTOR and query for data */
56  sd = (SECURITY_DESCRIPTOR *) pcl_malloc(size);
57  success = GetKernelObjectSecurity(file, sec_info, sd, size, &size); // self-relative format
58 
59  CloseHandle(file);
60 
61  if(!success)
62  {
63  pcl_free(sd);
64  return R_SETLASTERR(NULL);
65  }
66 
67  if(len)
68  *len = (DWORD) size;
69 
70  return sd;
71 }

73 int
74 ipcl_win32_sd_get(const pchar_t *path, SECURITY_DESCRIPTOR **sd, DWORD *sdlen,
75  PSID *world_sid, PSID *owner_sid, PSID *group_sid, BOOL *have_dacl, PACL *dacl,
76  SECURITY_INFORMATION secinfo)
77 {
78  BOOL u;
79 
80  *have_dacl = false;
81 
82  if(world_sid)
83  *world_sid = ipcl_win32_world_sid();
84 
85  if(!(*sd = sd_get(path, sdlen, secinfo)))
86  {
87  FreeSid(*world_sid);
88  return TRC();
89  }
90 
91  if((secinfo & OWNER_SECURITY_INFORMATION) && owner_sid &&
92  !GetSecurityDescriptorOwner(*sd, owner_sid, &u))
93  {
94  pcl_free(*sd);
95  FreeSid(*world_sid);
96  return SETLASTERR();
97  }
98 
99  if((secinfo & GROUP_SECURITY_INFORMATION) && group_sid &&
100  !GetSecurityDescriptorGroup(*sd, group_sid, &u))
101  {
102  pcl_free(*sd);
103  FreeSid(*world_sid);
104  return SETLASTERR();
105  }
106 
107  if(!dacl)
108  return 0;
109 
110  *dacl = NULL;
111 
112  if(!GetSecurityDescriptorDacl(*sd, have_dacl, dacl, &u))
113  {
114  pcl_free(*sd);
115  FreeSid(*world_sid);
116  return SETLASTERR();
117  }
118  else if(!*dacl)
119  {
120  *have_dacl = false;
121  }
122 
123  return 0;
124 }
