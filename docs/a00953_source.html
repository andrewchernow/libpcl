35 XFUNC(ncpy)(xchar *dest, size_t size, const xchar *src, size_t len)
36 {
37  if(!dest || !size)
38  return R_SETERR(NULL, PCL_EBUF);
39 
40  *dest = 0;
41  if(!src)
42  return NULL;
43 
44  if(!len && !(len = xstrlen(src)))
45  return dest;
46 
47  if(len >= size)
48  return R_SETERR(NULL, PCL_EBUF);
49 
50  memcpy(dest, src, len * sizeof(xchar));
51  dest[len] = 0;
52 
53  return dest;
54 }
