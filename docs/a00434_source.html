51 pcl_fdpath(int fd, pchar_t **out)
52 {
53  int out_len = 0;
54 
55  if(!out || fd < 0)
56  return BADARG();
57 
58 #ifdef PCL_WINDOWS
59  HANDLE hfile = (HANDLE)_get_osfhandle(fd);
60  GetFinalPathNameByHandle_t getpath = (GetFinalPathNameByHandle_t)GetProcAddress(
61  GetModuleHandleA("kernel.dll"), "GetFinalPathNameByHandle");
62 
63  if(getpath)
64  {
65  pchar_t tmp[1], *result = NULL;
66  DWORD dw = getpath(hfile, tmp, countof(tmp), FILE_NAME_NORMALIZED);
67 
68  /* function failed */
69  if(dw == 0)
70  return SETLASTERR();
71 
72  result = pcl_pmalloc(dw);
73  dw = getpath(hfile, result, dw-1, FILE_NAME_NORMALIZED);
74 
75  if(dw == 0)
76  {
77  pcl_free(result);
78  return SETLASTERR();
79  }
80 
81  /* Remove the longpath "\\?\" prefix */
82  if(!pcl_pcsnicmp(result, _P("\\\\?\\"), 4))
83  {
84  memmove(result, result + 4, ((dw - 4) + 1) * sizeof(pchar_t));
85  dw -= 4;
86  }
87 
88  *out = result;
89  out_len = (int)(dw - 1); /* includes NUL, remove it */
90  }
91  /* GetFinalPathNameByHandle() function appeared in Vista. */
92  else
93  {
94  HANDLE hmap;
95  void *addr;
96  DWORD szhi = 0;
97  pchar_t devpath[512 + PCL_MAXPATH];
98  bool found = false;
99  pchar_t drivebuf[1024];
100  pchar_t *drive = drivebuf;
101  DWORD szlow = GetFileSize(hfile, &szhi);
102 
103  /* Can't map zero byte file */
104  if(szhi == 0 && szlow == 0)
105  return SETERR(PCL_ENOTSUP);
106 
107  if(!(hmap = CreateFileMapping(hfile, NULL, PAGE_READONLY, 0, 1, NULL)))
108  return SETLASTERR();
109 
110  if(!(addr = MapViewOfFile(hmap, FILE_MAP_READ, 0, 0, 1)))
111  {
112  FREEZE_ERR(CloseHandle(hmap));
113  return SETLASTERR();
114  }
115 
116  /* this gives us a device path, that's okay */
117  if(!GetMappedFileName(GetCurrentProcess(), addr, devpath, countof(devpath)))
118  {
119  FREEZE_ERR(
120  UnmapViewOfFile(addr);
121  CloseHandle(hmap);
122  );
123 
124  return SETLASTERR();
125  }
126 
127  UnmapViewOfFile(addr);
128  CloseHandle(hmap);
129 
130  /* GetMappedFilename returns device paths:
131  * C:\WINNT would be \Device\Harddisk0\Partition1\WINNT
132  * Convert to drive letter path.
133  */
134  if(!GetLogicalDriveStrings(countof(drivebuf)-1, drivebuf))
135  return SETLASTERR();
136 
137  do
138  {
139  pchar_t name[1024];
140 
141  if(QueryDosDevice(drive, name, countof(name)))
142  {
143  size_t nlen = pcl_pcslen(name);
144 
145  if(nlen < countof(name) && !pcl_pcsnicmp(devpath, name, nlen) &&
146  devpath[nlen] == _P('\\'))
147  {
148  out_len = pcl_aspprintf(out, _P("%Ps%Ps"), drive, devpath + nlen);
149  if(out_len < 0)
150  return TRC();
151 
152  found = true;
153  break;
154  }
155  }
156 
157  drive += (pcl_pcslen(drive) + 1);
158  }
159  while(*drive);
160 
161  /* use device path if no drive letter mapping was found */
162  if(!found)
163  {
164  *out = pcl_pcsdup(devpath);
165  out_len = (int)pcl_pcslen(*out);
166  }
167  }
168 
169 #elif defined(PCL_DARWIN)
170  char pbuf[MAXPATHLEN + 1];
171 
172  if(fcntl(fd, F_GETPATH, pbuf) == -1)
173  return SETLASTERR();
174 
175  *out = pcl_pcsdup(pbuf);
176  out_len = (int) pcl_pcslen(*out);
177 
178 #else
179  char fdpath[128];
180  char pbuf[PATH_MAX + 1];
181 
182  pcl_sprintf(fdpath, sizeof(fdpath), "/proc/%d/fd/%d", getpid(), fd);
183  out_len = (int) readlink(fdpath, pbuf, sizeof(pbuf));
184  if(out_len == -1)
185  return SETLASTERR();
186 
187  *out = pcl_strndup(pbuf, out_len); /* NUL terminates for us */
188 #endif
189 
190  return out_len;
191 }
