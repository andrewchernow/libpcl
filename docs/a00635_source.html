40 pcl_getenv(const pchar_t *name)
41 {
42  pchar_t *value = NULL;
43 
44  if(!name)
45  return NULL;
46 
47 #ifdef PCL_WINDOWS
48  DWORD len = GetEnvironmentVariable(name, NULL, 0);
49 
50  if(len)
51  {
52  value = pcl_malloc(len * sizeof(pchar_t));
53 
54  if(GetEnvironmentVariable(name, value, len) == 0)
55  value = pcl_free(value);
56  }
57 
58 #else
59  value = getenv(name);
60 
61  if(value)
62  value = pcl_pcsdup(value);
63 #endif
64 
65  return value;
66 }
