81 int
82 ipcl_statent(const pchar_t *_path, int fd, pcl_stat_t *buf, int flags)
83 {
84  HANDLE hfind;
85  WIN32_FIND_DATA wfd;
86 
87  if(strempty(_path) && fd < 0)
88  return BADARG();
89 
90  memset(&wfd, 0, sizeof(wfd));
91 
92  /* this is a windows fstat */
93  if(fd > -1)
94  {
95  BY_HANDLE_FILE_INFORMATION info = {0};
96 
97  if(!GetFileInformationByHandle((HANDLE) _get_osfhandle(fd), &info))
98  return SETLASTERR();
99 
100  /* no stat buf, caller is just doing an exists check */
101  if(!buf)
102  return 0;
103 
104  wfd.dwFileAttributes = info.dwFileAttributes;
105  wfd.ftCreationTime = info.ftCreationTime;
106  wfd.ftLastAccessTime = info.ftLastAccessTime;
107  wfd.ftLastWriteTime = info.ftLastWriteTime;
108  wfd.nFileSizeHigh = info.nFileSizeHigh;
109  wfd.nFileSizeLow = info.nFileSizeLow;
110 
111  convstat(NULL, buf, &wfd, flags, false);
112  return 0;
113  }
114 
115  pchar_t *path = pcl_realpath(_path, NULL);
116 
117  if(!path)
118  return TRC();
119 
120  /* UNC Server: \\server */
121  if(PathIsUNCServer(path))
122  {
123  wfd.dwFileAttributes = GetFileAttributes(path);
124 
125  if(wfd.dwFileAttributes == INVALID_FILE_ATTRIBUTES)
126  {
127  SETLASTERRMSG("cannot get information about '%Ps'", path);
128  pcl_free(path);
129  return -1;
130  }
131 
132  if(buf)
133  convstat(path, buf, &wfd, flags, false);
134 
135  pcl_free(path);
136  return 0;
137  }
138 
139  /* Roots: C:\, \, \\server\share (PathIsUNCServerShare) */
140  if(PathIsRoot(path))
141  {
142  WIN32_FILE_ATTRIBUTE_DATA attr;
143 
144  if(!GetFileAttributesEx(path, GetFileExInfoStandard, &attr))
145  {
146  SETLASTERRMSG("cannot get information about '%Ps'", path);
147  pcl_free(path);
148  return -1;
149  }
150 
151  if(buf)
152  {
153  wfd.dwFileAttributes = attr.dwFileAttributes;
154  wfd.ftCreationTime = attr.ftCreationTime;
155  wfd.ftLastAccessTime = attr.ftLastAccessTime;
156  wfd.ftLastWriteTime = attr.ftLastWriteTime;
157  wfd.nFileSizeHigh = attr.nFileSizeHigh;
158  wfd.nFileSizeLow = attr.nFileSizeLow;
159  convstat(path, buf, &wfd, flags, true);
160  }
161 
162  pcl_free(path);
163  return 0;
164  }
165 
166  /* Regular file system object */
167  hfind = FindFirstFile(path, &wfd);
168 
169  if(hfind == INVALID_HANDLE_VALUE)
170  {
171  SETLASTERRMSG("cannot get information about '%Ps'", path);
172  pcl_free(path);
173  return -1;
174  }
175 
176  FindClose(hfind);
177 
178  if(buf)
179  convstat(path, buf, &wfd, flags, false);
180 
181  pcl_free(path);
182  return 0;
183 }

185 static void
186 convstat(pchar_t *path, pcl_stat_t *st, WIN32_FIND_DATA *wfd, int flags, bool is_root)
187 {
188  st->nlink = 1;
189 
190  /* default is SECURITY_NULL_SID_AUTHORITY: ie Unknown */
191  pcl_strcpy(st->uid, countof(st->gid), "S-1-0-0");
192  pcl_strcpy(st->gid, countof(st->gid), "S-1-0-0");
193 
194  /* get unix permission bits from the DACL. If that fails, use boiler
195  * plate hack from Microsoft's CRT library.
196  */
197  if((flags & PCL_STAT_SKIP_WIN32OWNER) || !path ||
198  ipcl_win32_file_security(path, &st->mode, &st->uid, &st->gid))
199  {
200  if(wfd->dwFileAttributes & FILE_ATTRIBUTE_READONLY)
201  st->mode |= (_S_IREAD + (_S_IREAD >> 3) + (_S_IREAD >> 6));
202  else
203  st->mode |= ((_S_IREAD | _S_IWRITE) + ((_S_IREAD | _S_IWRITE) >> 3) +
204  ((_S_IREAD | _S_IWRITE) >> 6));
205  }
206 
207  /* Use drive letter as device id */
208  if(!strempty(path) && path[1] == _P(':'))
209  {
210  int drive = pcl_topupper(path[0]);
211 
212  st->dev = makedev(FILE_TYPE_DISK, drive);
213 
214  if(is_root)
215  st->mode |= S_IFBLK;
216  }
217 
218  if((flags & PCL_STAT_LINK) && (wfd->dwFileAttributes & FILE_ATTRIBUTE_REPARSE_POINT))
219  st->mode |= S_IFLNK;
220 
221  /* only set dir or reg if link is not already set */
222  if(!S_ISLNK(st->mode))
223  {
224  if(wfd->dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY)
225  st->mode |= S_IFDIR;
226  else
227  st->mode |= S_IFREG;
228  }
229 
230  /* set st_size to the number of used bytes */
231  if(is_root)
232  {
233  ULARGE_INTEGER btot, bfree;
234  if(path && GetDiskFreeSpaceEx(path, NULL, &btot, &bfree))
235  st->size = btot.QuadPart - bfree.QuadPart;
236  }
237  else
238  {
239  st->size = ((int64_t) wfd->nFileSizeHigh * ((int64_t) MAXDWORD + 1)) +
240  (int64_t) wfd->nFileSizeLow;
241  }
242 
243  /* File Times */
244  st->atime = ipcl_win32_ftime_to_pcl(&wfd->ftLastAccessTime);
245  st->mtime = ipcl_win32_ftime_to_pcl(&wfd->ftLastWriteTime);
246  st->crtime = ipcl_win32_ftime_to_pcl(&wfd->ftCreationTime);
247  st->ctime = st->mtime; /* changed time resembles mtime the most */
248 
249  if(wfd->dwFileAttributes & FILE_ATTRIBUTE_ARCHIVE)
250  st->flags |= SF_ARCHIVED;
251  if(wfd->dwFileAttributes & FILE_ATTRIBUTE_HIDDEN)
252  st->flags |= UF_HIDDEN;
253  if(wfd->dwFileAttributes & FILE_ATTRIBUTE_COMPRESSED)
254  st->flags |= UF_COMPRESSED;
255  if(wfd->dwFileAttributes & FILE_ATTRIBUTE_ENCRYPTED)
256  st->flags |= UF_ENCRYPTED;
257  if(wfd->dwFileAttributes & FILE_ATTRIBUTE_SPARSE_FILE)
258  st->flags |= UF_SPARSE;
259  if(wfd->dwFileAttributes & FILE_ATTRIBUTE_SYSTEM)
260  st->flags |= UF_SYSTEM;
261 
262  if(!path)
263  return;
264 
265  /* GetDiskFreeSpace() is expensive which is why we cache each root path's block size as we
266  * see them. We do this per thread to avoid requiring locks.
267  */
268 
269  st->blksize = 4096;
270 
271  if(!PathIsRelative(path) && PathStripToRoot(path))
272  {
273  /* GetDiskFreeSpace needs a trailing slash */
274  size_t path_len = pcl_pcslen(path);
275 
276  if(path[path_len - 1] != PCL_PPATHSEPCHAR)
277  {
278  path[path_len++] = L'\\';
279  path[path_len] = 0;
280  }
281 
282  /* check cache */
283  int i;
284  pcl_vector_t *bsize_cache = pcl_tls_get(bsize_key);
285 
286  for(i = 0; i < bsize_cache->count; i++)
287  {
288  bsize_cache_t *bs = pcl_vector_get(bsize_cache, i);
289 
290  if(!pcl_pcsicmp(bs->root, path))
291  {
292  st->blksize = bs->blksize;
293  break;
294  }
295  }
296 
297  /* not in cache yet */
298  if(i == bsize_cache->count)
299  {
300  DWORD a, b, sectors_per_cluster, bytes_per_sec;
301 
302  if(GetDiskFreeSpace(path, &sectors_per_cluster, &bytes_per_sec, &a, &b))
303  {
304  bsize_cache_t elem;
305 
306  elem.root = pcl_pcsdup(path);
307  st->blksize = elem.blksize = (int) (sectors_per_cluster * bytes_per_sec);
308  pcl_vector_append(bsize_cache, &elem);
309  }
310  }
311  }
312 
313  st->dsize = roundup(st->size, st->blksize);
314 
315  if(st->dsize == st->size)
316  st->dsize += st->blksize;
317 
318  st->blocks = st->dsize / 512;
319  if(st->blocks <= 0)
320  st->blocks = 1;
321 }
