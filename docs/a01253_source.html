45 
46  if(!path || !mode)
47  return SETERR(PCL_EINVAL);

51  if(ipcl_win32_sd_get(path, &sd, &sdlen, &world_sid, &owner_sid, &group_sid,
52  &have_dacl, &dacl, secinfo))
53  return TRC();
54 
55  if(uidp)
56  {
57  char *s;
58  if(ConvertSidToStringSidA(owner_sid, &s))
59  {

63  }
64 
65  if(gidp)

67  char *s;

76  {
77  *mode = S_IRWXU | S_IRWXG | S_IRWXO;
78  pcl_free(sd);
79  FreeSid(world_sid);
80  return 0;

82 
83  for(DWORD i = 0; i < dacl->AceCount; i++)
84  {
85  ACCESS_ALLOWED_ACE *ace;
86  PSID ace_sid;
87 
88  if(!GetAce(dacl, i, (PVOID * ) & ace))
89  continue;
90 
91  if(ace->Header.AceFlags & INHERIT_ONLY_ACE)
92  continue;
93 
94  ace_sid = (PSID) & ace->SidStart;
95 
96  if(ace->Header.AceType == ACCESS_ALLOWED_ACE_TYPE)
97  {
98  if(EqualSid(owner_sid, ace_sid))
99  *mode |= ipcl_win32_rights_to_mode(ace->Mask, S_IRUSR);
100  else if(EqualSid(group_sid, ace_sid))
101  *mode |= ipcl_win32_rights_to_mode(ace->Mask, S_IRGRP);
102  else if(EqualSid(world_sid, ace_sid))
103  *mode |= ipcl_win32_rights_to_mode(ace->Mask, S_IROTH);
104  }
105  }
106 
107  pcl_free(sd);
108  FreeSid(world_sid);
109  return 0;
110 }
