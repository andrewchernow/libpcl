55 /** Create a cipher object.
56  * @param algo cipher algorithm: supports "aes-128-gcm", "aes-256-gcm", "aes-128-cbc",
57  * "aes-256-cbc", "aes-128-ocb", "aes-256-ocb", "chacha20-poly1305".
58  * @param key cipher key. length expected to match algorithm
59  * @param iv cipher iv. The length depends on the cipher and mode. It is recommended to always
60  * provide 16 bytes to keep things simple, which is the maximum IV length across ciphers.
61  * @param aad additional authenticated data. This is only used by ciphers that support
62  * authenticated encryption with associated data (AEAD). This can be \c NULL in either case.
63  * @param aad_len number of \a aad bytes. If zero, \a aad parameter is ignored.
64  * @param enc when non-zero, create an encryption cipher, otherwise decryption
65  * @return pointer to a cipher object or \c NULL on error
66  * @see PCL_MAXIVLEN, enum pcl_cipher_algo
67  */
68 PCL_EXPORT pcl_cipher_t *pcl_cipher_create(const char *algo, const void *key,
69  const void *iv, const void *aad, uint32_t aad_len, bool enc);

71 /** Reset a cipher so it can be reused.
72  * This should be called after calling ::pcl_cipher_final.
73  *
74  * This requires providing a new IV since using the same IV across cycles is a very bad idea.
75  * The key cannot be changed. If the key needs to be changed, free this cipher and create a new
76  * one.
77  *
78  * The goal of reset is to avoid tearing everything down and creating a whole new context when it
79  * is desired to reuse the same KEY: set of files, network packets, etc. Instead, it can be reset
80  * with a new IV and optional AAD data. After this call, use ::pcl_cipher_update and
81  * ::pcl_cipher_final as usual.
82  *
83  * @param ciph pointer to a cipher object
84  * @param iv
85  * @param aad additional authenticated data. This is only used by ciphers that support
86  * authenticated encryption with associated data (AEAD). This can be \c NULL in either case.
87  * @param aad_len number of \a aad bytes. If zero, \a aad parameter is ignored.
88  * @return 0 for success and -1 on error.
89  */
90 PCL_EXPORT int pcl_cipher_reset(pcl_cipher_t *ciph, const void *iv,
91  const void *aad, uint32_t aad_len);

114 /** Finalize encrypt or decrypt process.
115  * @param ciph pointer to a cipher object
116  * @param out pointer to output buffer which must be at least the size of a single block.
117  * @param tag Only used for AEAD ciphers. When encrypting, this must be a buffer of at least
118  * ::PCL_AEAD_TAGLEN bytes. When decrypting, this is the tag value which must be
119  * ::PCL_AEAD_TAGLEN bytes. This can be \c NULL.
120  * @return number of bytes written to out (zero is valid) or -1 on error. When using AEAD during
121  * decryption, this will fail if the "expected" \a tag value is incorrect. In this case, the
122  * plaintext is not trustworthy and should be discarded.
123  * @see PCL_AEAD_TAGLEN
124  */
125 PCL_EXPORT int pcl_cipher_final(pcl_cipher_t *ciph, char *out, char *tag);
