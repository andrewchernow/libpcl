35 /** @defgroup array Array Management
36  * An array is a pointer table of allocated elements. This differs from a PCL vector in that
37  * there is no concept of element size. The elements are user allocated pointers that can
38  * all be the same type of object, different objects or even \c NULL.
39  *
40  * It is common to interact (read) with an array's \a elements directly, while this is cumbersome
41  * with a vector. Although a vector can be used like a pointer table, it is a bit awkward because
42  * elements must be added as pointers to pointers: element size would be \c sizeof(void*). Thus,
43  * when getting an element from a vector, one must dereference the element value:
44  * `*(void **) pcl_vector_get(v, 0)`.
45  *
46  * ### Create argv array
47  * @code
48  * // create an argv array of 4 arguments with a trailing NULL,
49  * // including the file to be executed and trailing NULL.
50  * pcl_array_t *arr = pcl_array(6, pcl_cleanup_ptr);
51  *
52  * char *file = strdup("some_program");
53  *
54  * pcl_array_append(arr, file);
55  * pcl_array_append(arr, strdup("-a"));
56  * pcl_array_append(arr, strdup("--quiet"));
57  * pcl_array_append(arr, strdup("--file=\"example/file.txt\""));
58  * pcl_array_append(arr, strdup("-t"));
59  *
60  * // element 5 already NULL
61  *
62  * // replace element 4 with -b, cleanup handler called for element 4
63  * if(something_is_true)
64  * pcl_array_set(arr, strdup("-b"), 4);
65  *
66  * // remove quiet if noisey is true, internally does a cleanup and memmove
67  * if(noisey)
68  * pcl_array_remove(arr, 2);
69  *
70  * // note, any unset element defaults to NULL.
71  * execv(file, arr->elements);
72  *
73  * pcl_array_free(arr);
74  * @endcode
75  *
76  * @{
77  */

123 /** Set an element of an array, replacing a possible existing element. When setting elements,
124  * the array's \a count is not consulted. Instead, this allows a set operation as long as
125  * \a index is less than \a capacity. This will set \c count to `index + 1` if \a count is
126  * less than `index + 1`. This is very different than ::pcl_array_insert or ::pcl_array_append,
127  * which always inserts an element, bound to array's current \c count, and growing the array if required.
128  * When performing a set operation, the array is never grown. Thus, trying to set an element at
129  * an \a index greater than or equal to \a capacity will throw a ::PCL_EINDEX error.
130  *
131  * If an element exists at \a index, it is removed and passed to the cleanup handler if both the
132  * cleanup handler and element are not \c NULL.
133  *
134  * If setting an element past \a count, all elements between \c count and \c index will be \c NULL.
135  *
136  * @param arr arr pointer to an array object
137  * @param elem pointer to an element
138  * @param index element index
139  * @return 0 for success and -1 on error.
140  * @see pcl_array_append
141  */
142 PCL_PUBLIC int pcl_array_set(pcl_array_t *arr, void *elem, int index);
