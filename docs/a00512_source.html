35 int
36 ipcl_htable_capacity(int capacity)
37 {
38  /* ~50 million max capacity for 32-bit machines and ~1.6 billion max capacity on
39  * 64-bit machines. This is plenty for both architectures and the result of using the
40  * max capacity on either architecture is a significant amount of memory. 64-bit could
41  * address more, but a table capacity of 1.6 billion (~1.2 billion entries considering
42  * load factor) is an insanely sized hash table.
43  */
44  static int prime_table[] = {
45  MIN_TABLE_SIZE, 53, 97, 193, 389, 769, 1543, 3079, 6151,
46  12289, 24593, 49157, 98317, 196613, 393241, 786433, 1572869,
47  3145739, 6291469, 12582917, 25165843, 50331653
48 #ifdef PCL_64BIT
49  , 100663319, 201326611 ,402653189, 805306457, 1610612741
50 #endif
51  };
52 
53  for(int i=0; i < countof(prime_table); i++)
54  if(capacity < prime_table[i])
55  return prime_table[i];
56 
57  return SETERR(PCL_ERANGE);
58 }
