61 /** Pipe a stream to or from a process. This differs from the POSIX popen in that there is
62  * no type argument. Instead, this uses redirection symbols like a Unix shell.
63  *
64  * @note The Windows CRT popen only works when used within a console application. PCL's popen
65  * has no such restriction. This function internally calls ::pcl_proc_exec.
66  *
67  * @code
68  * Mode Description Same as...
69  * < read from stdout popen("cmd", "r")
70  * > write to stdin popen("cmd", "w")
71  * ^ read from stderr [no equivalent]
72  *
73  * The below documents the other "flags" that can be specified in mode:
74  *
75  * ! set O_NONBLOCK [no equivalent]
76  *
77  *
78  * // read from the process's stdout without blocking. Process created with
79  * // the following command: `ls -l /home/username`
80  * pchar_t *homedir = pcl_getenv("HOME");
81  * FILE *stream = pcl_popen(_P("<! ls -l %Ps"), homedir);
82  * pcl_free(homedir);
83  * @endcode
84  *
85  * @param command pointer to the command and optional flags
86  * @param ... variable arguments
87  * @return stream pointer or \c NULL on error
88  */
89 PCL_PUBLIC FILE *pcl_popen(const pchar_t *command, ...);

97 /** Open a file handle as a stream.
98  * On success, a file stream is returned that must be closed via \c fclose. The provided file
99  * handle is released and cannot be used again. However, if this function fails, the caller
100  * is still repsonsible for closing the given file handle.
101  * @code
102  * pcl_file_t *file = open_a_file();
103  * FILE *fp = pcl_fdopen(file);
104  *
105  * if(fp) // success
106  * {
107  * fprintf(fp, "'fp' is valid here! 'file' is not.\n");
108  * fclose(fp);
109  * }
110  * else // failure
111  * {
112  * printf("'file' is valid here. 'fp' is not.\n");
113  * pcl_file_close(file); // callee's responsibility
114  * }
115  * @endcode
116  * @param file pointer to a file handle
117  * @return a file stream that must be closed with \c fclose
118  */
119 PCL_PUBLIC FILE *pcl_fdopen(pcl_file_t *file);

121 /** Create and open a unique temporary file.
122  * This function will make several attempts to create the unique temporary file. The below is
123  * a list (in order) of what directories are attempted:
124  * 1. try the \a dir argument
125  * 2. Windows \c TMP or Unix \c TMPDIR
126  * 3. Windows \c TEMP or Unix \c P_tmpdir
127  * 4. "."
128  * @param dir Optional directory path to create temp file within, can be \c NULL.
129  * @param prefix Optional prefix for the generated pathname. Can be \c NULL
130  * @param suffix Optional suffix for the generated pathname. Can be \c NULL
131  * @param pathp pointer to a ::pchar_t string that will receive the temp pathname. This
132  * can be \c NULL. When not \c NULL, this must be freed by the caller.
133  * @param keep If false, returned temp file will be automatically deleted once closed.
134  * When true, the file will persist after file close.
135  * @return
136  */
