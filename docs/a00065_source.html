88 /* Opens a pcl_file_t as a FILE stream. On success, a FILE stream is returned that must be closed
89  * via fclose. The provided pcl_file_t is released and cannot be used again. On error, NULL is
90  * returned and the provided pcl_file_t is left unchanged and completely usable.
91  *
92  * FILE *fp = pcl_fdopen(file);
93  *
94  * // success, 'file' is no longer valid ... do not use
95  *
96  * if(fp) // success
97  * {
98  * fprintf(fp, "'fp' is valid here! 'file' is not.\n");
99  * fclose(fp);
100  * }
101  * else // failure
102  * {
103  * pcl_file_printf(file, "'file' is valid here. 'fp' is not.\n");
104  * pcl_close(file); // callee still responsible for this...
105  * }
106  */
