171 /** A JSON value object. This represents any valid JSON value: boolean, number (integer or real),
172  * string, null, array or object.
173  */
174 struct tag_pcl_json
175 {
176  /** JSON value type. This can be: \c 0 for null, \c 'r' for real, \c 'i' for integer,
177  * \c 'b' for bool, \c 's' for string, \c 'a' for array, \c 'o' for object. Both real and
178  * integer are actually a JSON number but are separated for simplicity. There are macros
179  * for detecting value type that all begin with \c pcl_json_is.
180  */
181  char type;
182 
183  /** A JSON object's reference count.
184  * @see pcl_json_ref
185  */
186  int nrefs;
187 
188  union
189  {
190  /** JSON boolean value when \a type is \c 'b'. */
191  bool boolean;
192 
193  /** JSON real (number) value when \a type is \c 'r'. THis cannot be NaN, Infinity or
194  * ::PCL_JSON_INVREAL.
195  */
196  double real;
197 
198  /** JSON integer (number) value when \a type is \c 'i'. This cannot be ::PCL_JSON_INVINT. */
199  long long integer;
200 
201  /** JSON string value when \a type is \c 's'. This is always NUL-terminated. */
202  char *string;
203 
204  /** JSON array value when \a type is \c 'a'. */
205  pcl_array_t *array;
206 
207  /** JSON object value when \a type is \c 'o'. */
208  pcl_htable_t *object;
209  };
210 };

212 /** Decode a JSON string. This parses a JSON string that represents a single JSON value. A JSON
213  * value can be an object \c {}, array \c [], string \c "", number \c -102.84, boolean \c true or
214  * \c false or a null \c null. Parsing stops when the parsing of the JSON value at \a json
215  * is complete, regardless of the \a len argument. However, if a complete JSON value is not found
216  * before reaching \a json \c + \a len, an error occurs. The \a json string can contain bogus data
217  * after the value being parsed, since the parser stops at the end of the value. For example:
218  * @code
219  * {
220  * "name": "Henry Ford"
221  * }#invalid json here
222  * @endcode
223  * The above is completely valid, since a valid JSON object exists and ends before any invalid
224  * JSON text is encountered. When the \a end argument is supplied, it would be pointing at
225  * the '#' in the above example. This is very useful if a string contains multiple JSON values.
226  * pcl_json_decode can be called multiple times by passing the value of \a end as \a json,
227  * allowing one to iterate through a series of json values.
228  *
229  * @param json pointer to a json string. This must be NUL terminated if \a len is 0.
230  * @param len number of bytes within \a json argument. If 0, \a json must be NUL terminated.
231  * @param end pointer to the first character not parsed. This can be \c NULL.
232  * @return pointer to a json value or \c NULL on error
233  */
234 PCL_PUBLIC pcl_json_t *pcl_json_decode(const char *json, size_t len, const char **end);
