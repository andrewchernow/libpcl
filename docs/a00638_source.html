38 pcl_proc_parsecmd(const pchar_t *shell_cmd, pchar_t ***out)
39 {
40  const pchar_t *start;
41  bool in_quote = false;
42  pchar_t **targv;
43  size_t size = 4;
44  int count = 0;
45 
46  if(!shell_cmd)
47  return BADARG();
48 
49  shell_cmd = pcl_pcsskipws(shell_cmd);
50  if(*shell_cmd == '"')
51  {
52  in_quote = true;
53  shell_cmd++;
54  }
55 
56  start = shell_cmd;
57  targv = (pchar_t **)pcl_malloc((size + 1) * sizeof(pchar_t *));
58 
59  while(true)
60  {
61  switch(*shell_cmd)
62  {
63  case '\\':
64  shell_cmd += 2;
65  break;
66 
67  case '\0':
68  case '"':
69  case ' ':
70  case '\t':
71  {
72  size_t n;
73 
74  if(*shell_cmd == '"')
75  {
76  if(!in_quote)
77  {
78  pcl_proc_freeargv(count, targv);
79  return SETERRMSG(PCL_ESYNTAX, "unescaped quote", 0);
80  }
81 
82  in_quote = false;
83  }
84  else if(in_quote)
85  {
86  if(*shell_cmd == '\0')
87  {
88  pcl_proc_freeargv(count, targv);
89  return SETERRMSG(PCL_ESYNTAX, "unterminated quote", 0);
90  }
91 
92  shell_cmd++;
93  break;
94  }
95 
96  n = shell_cmd - start;
97 
98  if(n > 0)
99  {
100  if(count == (int)size)
101  {
102  size = (size * 3) / 2;
103  targv = (pchar_t **)pcl_realloc(targv, (size + 1) * sizeof(pchar_t *));
104  }
105 
106  targv[count++] = pcl_pcsndup(start, shell_cmd - start);
107  }
108 
109  if(*shell_cmd == '\0')
110  goto finish;
111 
112  shell_cmd = pcl_pcsskipws(++shell_cmd);
113 
114  if(*shell_cmd == '"')
115  {
116  in_quote = true;
117  shell_cmd++;
118  }
119 
120  start = shell_cmd;
121  break;
122  }
123 
124  default:
125  shell_cmd++;
126  break;
127  }
128  }
129 
130 finish:
131 
132  targv[count] = NULL;
133  if(out)
134  *out = targv;
135 
136  return count;
137 }
