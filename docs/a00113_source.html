36 /** @defgroup vector Vector
37  * A PCL Vector is a dynamic and contiguous block of memory for an array of same-sized elements.
38  * The vector automatically expands this block of memory if an insertion would exceed
39  * the vector's current capacity. The vector never automatically shrinks itself during an element
40  * removal, but a vector can be shrunk using ::pcl_vector_compact.
41  * #### Basic Usage
42  * @code
43  * #include <pcl/init.h>
44  * #include <pcl/vector.h>
45  * #include <stdio.h>
46  *
47  * int main(int argc, char **argv)
48  * {
49  * // always first
50  * pcl_init();
51  *
52  * // cleanup handler can be NULL, integer doesn't need it
53  * pcl_vector_t *vec = pcl_vector_create(8, sizeof(int), NULL);
54  *
55  * for(int i=0; i < vec->capacity * 2; i++)
56  * pcl_vector_append(vec, &i); // vector will have to expand after 8 appends
57  *
58  * for(int i=0; i < vec->count; i++)
59  * {
60  * int *elem = pcl_vector_get(vec, i);
61  * printf("%d\n", *elem); // 0\n1\n...15\n
62  * }
63  *
64  * pcl_vector_free(vec);
65  * }
66  * @endcode
67  * @{
68  */

76 /** A callback cleanup function for vector elements. This is needed if the vector elements
77  * contain any dynamic memory or the elements are pointers to pointers: such as a vector of
78  * strings. When a structure contains allocated members, only free the allocated members -- not
79  * the structure/element itself. The vector owns the structure. However, if elements are pointers
80  * to pointers, you will need to derefence \a elem and free that pointer.
81  * @note optionally, the pcl_vector_t.cleanup_ptr can be set and utilized within the callback.
82  * @param v pointer to a vector
83  * @param elem pointer to the element being removed. If elements are pointers to pointers,
84  * like strings, remember to dereference: `pcl_free(*(char **) elem)`.
85  * @see pcl_vector_create, pcl_vector_cleanup_dblptr
86  */
87 typedef void (*pcl_vector_cleanup_t)(pcl_vector_t *v, void *elem);
88 
89 /** Vector object. Used to track elements and grow the internal elements array when required. */
90 struct tag_pcl_vector
91 {
92  /** number of elements in vector */
93  int count;
94 
95  /** number of elements allocated */
96  int capacity;
97 
98  /** byte size of an element */
99  size_t size;
100 
101  /** Element array. This is always pcl_vector_t.capacity \c * pcl_vector_t.size bytes */
102  char *elems;
103 
104  /** Optional user-defined pointer available within the cleanup callback. */
105  void *cleanup_ptr;
106 
107  /** Element cleanup callback. Optionally, one can set cleanup_ptr and access it
108  * within the cleanup callback.
109  * #### Example
110  * @code
111  * void mycleanup(pcl_vector_t *v, void *elem)
112  * {
113  * myobj *o = (myobj *) v->cleanup_ptr;
114  *
115  * if(o && o->allocated_buf)
116  * {
117  * elemobj *e = (elemobj *) elem;
118  * pcl_free(e->buf);
119  * }
120  * }
121  * @endcode
122  */
123  pcl_vector_cleanup_t cleanup;
124 };

126 /** Get an element.
127  * @warning this provides no bounds checking on \a pos and is designed to be used in cases
128  * where \a pos is known to be valid: such as a loop constrained by the vector's count. This
129  * is an optimization for the most common usage pattern. Expect a core dump or undefined
130  * behavior if \a v or \a pos is invalid.
131  * @param v pointer to a vector
132  * @param pos zero-based position of element to retreive
133  * @return pointer to element at \a pos
134  * @see pcl_vector_get_safe
135  */
136 #define pcl_vector_get(v, pos) \
137  ((void *)(((pcl_vector_t *)(v))->elems + ((pos) * ((pcl_vector_t *)(v))->size)))
138 
139 /** Dereference an element and return it.
140  * @warning this provides no bounds checking on \a pos and is designed to be used in cases
141  * where \a pos is known to be valid: such as a loop constrained by the vector's count. This
142  * is an optimization for the most common usage pattern. Expect a core dump or undefined
143  * behavior if \a v or \a pos is invalid.
144  * @param v pointer to a vector
145  * @param pos zero-based position of element to retreive
146  * @return pointer to element at \a pos
147  * @see pcl_vector_getptr_safe
148  */
149 #define pcl_vector_getptr(v, pos) *((void **)pcl_vector_get(v, pos))

193 /** Allocate and append a string. This function should only be used with string vectors, where
