36 int
37 pcl_htable_put(pcl_htable_t *ht, const void *key, void *value, bool unique)
38 {
39  /* NULL values allowed */
40  if(!(ht && key))
41  return BADARG();
42 
43  int index;
44  uintptr_t code;
45  pcl_htable_entry_t *e = ipcl_htable_lookup(ht, key, &code, &index);
46 
47  if(e)
48  {
49  if(unique)
50  return SETERR(PCL_EEXIST);
51 
52  const void *old_key = e->key;
53  void *old_value = e->value;
54 
55  e->key = key;
56  e->value = value;
57 
58  /* can't use ipcl_htable_remove_entry since that will free our entry object. In this case,
59  * we can reuse the entry object, just allow the user to free the old key & value.
60  */
61  if(ht->remove_entry)
62  ht->remove_entry(old_key, old_value, ht->userp);
63  }
64  /* doesn't exist, add it */
65  else
66  {
67  /* check if a rehash is needed. The rehash function only returns an error if the
68  * capacity has exceeded the maximum size for the architecture. Sets PCL_ERANGE.
69  */
70  if(ht->count > (int) (ht->max_loadfac * (float) ht->capacity))
71  {
72  if(ipcl_htable_rehash(ht, true))
73  return TRC();
74 
75  /* regenerate index, current index is based off old capacity */
76  index = (int) (code % ht->capacity);
77  }
78 
79  pcl_htable_entry_t *ent = pcl_malloc(sizeof(pcl_htable_entry_t));
80 
81  ent->key = key;
82  ent->value = value;
83  ent->code = code;
84 
85  /* add new entry to collision list */
86  pcl_htable_entry_t *head = ht->entries[index];
87  ht->entries[index] = ent;
88  ent->next = head;
89  ht->count++;
90  }
91 
92  return 0;
93 }
