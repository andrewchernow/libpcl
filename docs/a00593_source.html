36 vxprintf(xchar *buf, size_t size, const xchar *format, va_list ap)
37 {
38  if(!format)
39  return -PCL_EINVAL;
40 
41  xchar fmtbuf[4096];
42  xchar *fmt = xfmtconv(fmtbuf, countof(fmtbuf), format);
43  int r = 0, err = 0;
44 
45  errno = 0;
46 
47 #ifdef PCL_WINDOWS
48  #ifdef XWIDE
49  r = !buf ? _vscwprintf(fmt, ap) : _vsnwprintf_s(buf, size, size-1, fmt, ap);
50  #else
51  r = !buf ? _vscprintf(fmt, ap) : _vsnprintf_s(buf, size, size-1, fmt, ap);
52  #endif
53 
54  if(buf && r == -1 && errno == ERANGE) /* special case, buffer too small */
55  err = PCL_EBUF;
56 
57 /* Unix wide */
58 #elif defined(XWIDE)
59  if(!buf)
60  {
61  /* vswprintf always returns -1, write to /dev/null to get length */
62  FILE *fp = fopen("/dev/null", "wb");
63  r = vfwprintf(fp, fmt, ap);
64  fclose(fp);
65  }
66  else
67  {
68  /* error reporting on darwin/linux is terrible. */
69  r = vswprintf(buf, size, fmt, ap);
70  if(r < 0 && size != 0)
71  err = PCL_EBUF; /* best guess */
72  }
73 
74  /* Unix non-wide */
75 #else
76  if(!buf)
77  {
78  r = vsnprintf(NULL, 0, fmt, ap);
79  }
80  else
81  {
82  r = vsnprintf(buf, size, fmt, ap);
83  if(r >= (int) size) /* buffer to small */
84  err = PCL_EBUF;
85  }
86 #endif
87 
88  /* set pcl error if not already set, returned as -(err) */
89  if(err == 0 && r < 0)
90  {
91  err = pcl_err_crt2pcl(errno);
92  if(err == 0 || err == PCL_EUNDEF)
93  err = PCL_EINVAL;
94  }
95 
96  if(fmt != fmtbuf)
97  pcl_free(fmt);
98 
99  return err ? -err : r;
100 }
