37 static int
38 rmdir_r(pchar_t *path, size_t size)
39 {
40  int path_len;
41  pcl_dirent_t ent;
42  pcl_dir_t *dir = pcl_opendir(path);
43 
44  if(!dir)
45  return TRC();
46 
47  path_len = (int) pcl_pcslen(path);
48  path[path_len] = PCL_PPATHSEPCHAR;
49 
50  while(pcl_readdir(dir, &ent, NULL))
51  {
52  /* append entry name */
53  pcl_pcscpy(path + path_len + 1, size - path_len - 1, ent.name);
54 
55  /* recurse */
56  if(ent.type == PclDirentDir)
57  {
58  if(rmdir_r(path, size))
59  {
60  pcl_closedir(dir);
61  path[path_len] = 0;
62  return TRC();
63  }
64  }
65  else if(pcl_unlink(path))
66  {
67  pcl_closedir(dir);
68  path[path_len] = 0;
69  return TRC();
70  }
71  }
72 
73  pcl_closedir(dir);
74  path[path_len] = 0;
75 
76  /* readdir failed */
77  if(pcl_errno != PCL_ENOMORE)
78  return TRC();
79 
80  if(pcl_rmdir(path))
81  return TRC();
82 
83  return 0;
84 }

86 int
87 pcl_rmdir_r(const pchar_t *path)
88 {
89  pchar_t pbuf[32 * 1024];
90 
91  pcl_pcscpy(pbuf, countof(pbuf), path);
92 
93  if(rmdir_r(pbuf, countof(pbuf)))
94  return TRC();
95 
96  return 0;
97 }
