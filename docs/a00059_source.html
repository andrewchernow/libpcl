36 /** @defgroup htable Hash Table
37  * This defines a hash table object and associated management functions. This implementation
38  * will preserve insertion order: like Java's LinkedHashMap but without the doubly-linked list.
39  *
40  * ### Implementation
41  * Whenever an entry is put, it is added sequentially to an array of entry
42  * objects. To handle collisions, the entry objects are a linked list using an integer value
43  * that represents the index of the next collision within the entry array. The key is hashed
44  * using Google's FarmHash and bitwise AND'd with the current table size minus 1 to get the
45  * hashed index value. The hashed index value is an index to a second array (entry_lookup) of
46  * entry array indexes. Thus, looking up a hashed index within the \a entry_lookup array, will
47  * give you the entry array index, which is the beginning of the entry collision list. This is how
48  * the PCL hash table can preserve insertion order: one array using hash index to find the index
49  * into the sequential array of entries.
50  *
51  * @note The entry_lookup and entry array idea came directly from PHP's new hash table
52  * implementation released in 7.0. Just like PHP, PCL allocates the entry and entry_lookup arrays
53  * as one allocation: both with \a capacity elements. When an entry is removed, the entry array
54  * marks it as deleted but does not adjust the array to close the gap. An entry is marked as
55  * deleted by setting its \a key to \c NULL, which PCL treats as an invalid key. On rehash, the
56  * entry array is defragmented.
57  *
58  * ### Default Hash Function
59  * Unlike PHP's implementation, PCL does not use the DJBX33A hash as its default, which stands
60  * for “Daniel J. Bernstein, Times 33 with Addition”. Google's farmhash proved to have far less
61  * collisions (on string data since DJBX33A is for string keys). In one test of 470,000
62  * english words, Google's FarmHash had no collisions while DJBX33A had 320. Performance wise,
63  * they are about the same. Another benifit of Google's FarmHash is that it is not just designed
64  * for string keys, which PCL requires to allow binary and string keys. It is also prefered
65  * to provide a single generalized hash function for both binary and string keys out of the box.
66  *
67  * ### Customizing
68  * This implementation has the ability to set a minimum and maximum load factor. You can also
69  * set the \a key_len if not using string keys. If key_len is 0 (strings), then the default
70  * key_equals function uses strcmp. If key_len is not 0, memcmp is used. There are three callbacks
71  * to customize a hash table: pcl_htable_t.remove_entry, pcl_htable_t.hashcode and
72  * pcl_htable_t.key_equals. All of these customizations can be set after table creation.
73  *
74  * ### Rehashing
75  * Rehashing is implemented by allocating a new entry and entry_lookup array (single allocation)
76  * that is either double the current size or half the current size (uses powers of 2). When the
77  * copy is complete, the old entry and entry_lookup arrays are freed. No hash codes are recomputed,
78  * since each entry caches its own hash code. The entry array is defragmented, meaning all deleted
79  * entries are removed; \a count_used will always be the same as \a count.
80  *
81  * ### Table Size Limitations
82  * The table can grow to 33,554,432 on 32-bit machines and 1,073,741,824 on 64-bit machines. This
83  * table expands and contracts based on a max and min load factor -- 0.75 and 0.20 respectively.
84  * Table capacity is always powers of two.
85  *
86  * ### String and Binary Keys
87  * The PCL hash table supports binary keys, which are fixed-length and normally do not change for
88  * the life of the hash table. After creating a hash table, one can specify the key length in bytes
89  * or provide zero (the default) for variable-length keys (strings). For example, when using a UUID
90  * string as a key, each key would require 36 bytes. However, only 16 bytes per key if binary. In
91  * this case, you would set the key_len to 16. The default \c key_equals will use \c memcmp when
92  * \a key_len is not zero and \c strcmp otherwise.
93  *
94  * #### UUID Key Example
95  * @code
96  * #include <pcl/htable.h>
97  * #include <pcl/init.h>
98  * #include <pcl/string.h>
99  * #include "app/user.h"
100  *
101  * static unsigned char *uuidtokey(unsigned char *out, const char *uuid)
102  * {
103  * char *hex = pcl_strreplace(uuid, "-", "");
104  * pcl_hex_decode(out, 16, hex);
105  * pcl_free(hex);
106  * return out;
107  * }
108  *
109  * static void remove_entry(const void *key, void *value)
110  * {
111  * pcl_free(key);
112  * app_user_free((app_user_t *) value);
113  * }
114  *
115  * int main(void)
116  * {
117  * pcl_init();
118  *
119  * int num;
120  * app_user_t **users = app_user_fetchall(&num);
121  *
122  * pcl_htable_t *ht = pcl_htable(num);
123  *
124  * // let htable know key length, only used by default key_equals and hashcode callbacks
125  * ht->key_len = 16;
126  * ht->remove_entry = remove_entry;
127  *
128  * unsigned char uuid[16];
129  *
130  * for(int i = 0; i < num; i++)
131  * {
132  * void *key = pcl_malloc(16);
133  * memcpy(key, uuidtokey(uuid, users[i]->uuid), 16);
134  * pcl_htable_put(ht, key, users[i], true);
135  * }
136  *
137  * // elements will be freed by hash table's remove_entry
138  * pcl_free(users);
139  *
140  * const char *john = "37462d98-371b-11eb-adc1-0242ac120002";
141  * pcl_user_t *user = pcl_htable_get(ht, uuidtokey(uuid, john));
142  *
143  * printf("%s\n", user->surname);
144  *
145  * pcl_htable_free(ht);
146  *
147  * return 0;
148  * }
149  *
150  * @endcode
151  *
152  * One step further, is taking a 32 or 64-bit sequence value, like a \c user_id, and storing its
153  * value as the key pointer address: `key = (const void *)(uintptr_t) user_id`. This would require
154  * a custom key_equals (where key1 == key2 could be the implementation) and a custom
155  * hashcode callback (where return (uintptr_t) key; could be the implementation). This method
156  * is very efficient both in memory and performance, but only works if the numeric bit length is
157  * \c <= the number of bits used by pointers. Also, user_id of 0 is invalid, since keys cannot
158  * be \c NULL, but it is uncommon to start sequence values like user_ids with 0. An example of
159  * this is below.
160  *
161  * #### UserId Key Example
162  * @code
163  * #include <pcl/htable.h>
164  * #include <pcl/init.h>
165  * #include "app/user.h"
166  *
167  * #define IDTOKEY(id) (const void *) ((uintptr_t) (id))
168  *
169  * static bool key_equals(const void *a, const void *b, size_t key_len)
170  * {
171  * UNUSED(key_len);
172  * return a == b; // key addresses are user_ids, compare them directly.
173  * }
174  *
175  * static uintptr_t hashcode(const void *key, size_t key_len)
176  * {
177  * UNUSED(key_len);
178  * return (uintptr_t) key; // no need to hash keys, user_id is unique
179  * }
180  *
181  * static void remove_entry(const void *key, void *value)
182  * {
183  * UNUSED(key);
184  * app_user_free((app_user_t *) value); // key doesn't need to be freed
185  * }
186  *
187  * int main(void)
188  * {
189  * pcl_init();
190  *
191  * int num_users;
192  * app_user_t **users = app_user_fetchall(&num_users);
193  *
194  * // a table mapping a user_id => user.
195  * pcl_htable_t *ht = pcl_htable(num_users);
196  *
197  * // configuring a pcl_htable_t
198  * ht->remove_entry = remove_entry;
199  * ht->hashcode = hashcode;
200  * ht->key_equals = key_equals;
201  *
202  * // ignored since example sets hashcode and key_equals. Set if your callbacks need it.
203  * //ht->key_len = sizeof(int64_t);
204  *
205  * // set min and max load factor
206  * //ht->min_loadfac = 0.13f;
207  * //ht->max_loadfac = 0.85f;
208  *
209  * // set key address to user_id
210  * for(int i = 0; i < num_users; i++)
211  * pcl_htable_put(ht, IDTOKEY(users[i]->id), users[i], true);
212  *
213  * pcl_free(users); // elements freed by remove_entry
214  *
215  * app_user_t *user = pcl_htable_get(ht, IDTOKEY(1982734));
216  *
217  * printf("%s\n", user->name);
218  *
219  * pcl_htable_free(ht); // calls remove_entry foreach entry in table
220  *
221  * return 0;
222  * }
223  * @endcode
224  * @{
225  */

232 /** Hash table entry object. The overhead of each entry is 16 bytes on 32-bit machines and
233  * 32 bytes on 64-bit machines. On 64-bit machines, this is really 28 bytes but will be
234  * padded to 32.

238  /** leave me alone (collision list) */
239  int next;

241  /** hash code of entry key, 4 bytes on 32-bit and 8 bytes on 64-bit */
242  uintptr_t code;
243 
244  /** entry key */
245  const void *key;
246 
247  /** entry value */
248  void *value;

255  /** count of entries. differs from \a count_used because this doesn't include deleted entries.
256  * @warning treat this as immutable
257  */
258  int count;
259 
260  /** count of used entries (active + deleted). A deleted entry has a \c NULL key and
261  * should always be skipped.
262  * @warning treat this as immutable
263  */
264  int count_used;
265 
266  /** current table capacity
267  * @warning treat this as immutable
268  */
269  int capacity;
270 
271  /** mask used to find buckets, table `capacity - 1`
272  * @warning treat this as immutable
273  */
274  int table_mask;
275 
276  /** A sequential array of entries as they are inserted. This not allows for preserving
277  * insertion order, by avoids allocating each individual entry. This always contains
278  * \a capacity available entries. The next insert index is bound to \a count_used, not
279  * \a count. \a count_used is always incremented, unlike \a count. It is reset to 0
280  * by ::pcl_htable_clear and reset to \a count during an internal rehash.
281  * @note each entry element is part of a collision list by using its \a next
282  * member to indicate the index of the next entry in the list. A \a next value of -1 indicates
283  * the end of the list.
284  * @note allocated in same block of memeory as \a entry_lookup
285  */
286  pcl_htable_entry_t *entries;
287 
288  /** An entry lookup array where each element is an \a entries index value. When given a key,
289  * the key's hashcode is bitwise AND'd with the table's \a capacity minus 1, which produces the
290  * index to this array; termed the hashed index or \c hashidx. Elements in this array are set
291  * to -1 to indicate they are not in use. This always contains \a capacity indexes.
292  * @note allocated in same block of memeory as \a entries
293  */
294  int *entry_lookup;
295 
296  /* READ/WRITE SECTION */
297 
298  /** Key length in bytes. The default is zero, which means variable-length (strings). This value
299  * is only used by the default \a key_equals and \a hashcode callbacks. If a custom version of
300  * both callbacks are set, this can remain unset unless the custom implementation requires
301  * this value.
302  *
303  * The table should be empty when setting this value: after ::pcl_htable
304  * or ::pcl_htable_clear.
305  */
306  size_t key_len;
307 
308  /** The minimun load factor of the table. When the number of entries in the table
309  * falls below the product of this value and current capacity, the table is shrunk to
310  * half its capacity (always a power of 2). The default value is \c 0.20f.
311  *
312  * The table should be empty when setting this value: after ::pcl_htable
313  * or ::pcl_htable_clear.
314  */
315  float min_loadfac;
316 
317  /** The maximum load factor of the table. When the number of entries in the table
318  * exceeds the product of this value and current capacity, the table is grown to
319  * double its capacity (always a power of 2). The default is \c 0.75f.
320  *
321  * The table should be empty when setting this value: after ::pcl_htable
322  * or ::pcl_htable_clear.
323  */
324  float max_loadfac;
325 
326  /** Called when an entry is being removed from the table. There is no default version, this
327  * is set to \c NULL on the hash table. When \c NULL, the \a key and \a value are not freed
328  * which might be the desired behavior. In most cases, an application will want to set this to
329  * free keys and/or values. In either case, the entry itself is marked as deleted (\c NULL key).
330  * @param key pointer to the key beign removed
331  * @param value pointer to the value
332  */
333  void (*remove_entry)(const void *key, void *value);
334 
335  /** Determine if two keys are equal. The default comparator uses \c strcmp when \a key_len
336  * is zero and \c memcmp otherwise. The \a key_len passed is the pcl_htable_t.key_len member.
337  * @param key1 pointer to a key
338  * @param key1 pointer to a key to compare against \a key1
339  * @param key_len key length in bytes, expected to be the length for both \a key1 and \a key2
340  * @return true if \a key1 and \a key2 are equal
341  */
342  bool (*key_equals)(const void *key1, const void *key2, size_t key_len);
343 
344  /** Compute a hash code for the given key. Google's farmhash is used as the default hashfunc
345  * for both 32 and 64-bit architectures. On 32-bit machines, a 32-bit code is computed and
346  * on 64-bit machines, a 64-bit hash code.
347  *
348  * For the default hashcode implementation, if \a key_len is zero, it is assumed that \a key is
349  * a string and \c strlen is used to get the \a key_len. The \a key_len passed is the
350  * pcl_htable_t.key_len member.
351  *
352  * This is only executed once for each entry put into the table. The computed hash code is
353  * cached within the entry object. This is also executed, on the \a key argument, each time
354  * either ::pcl_htable_get, ::pcl_htable_lookup or ::pcl_htable_remove are called.
355  *
356  * Hash code values use the \c uintptr_t data type. This means the number of bits allowed
357  * depends on the number of bits within a \c uintptr_t.
358  * @param key pointer to the key to hash
359  * @param key_len key length in bytes
360  * @return hash code no more than \c unitptr_t bytes
361  */
362  uintptr_t (*hashcode)(const void *key, size_t key_len);
363 };
