45 /* Default hashfunc is Google's farmhash: https://github.com/google/farmhash */
46 static uintptr_t
47 default_hashcode(const void *key, size_t key_len, void *userp)
48 {
49  UNUSED(userp);
50 
51  if(!key_len)
52  key_len = strlen((const char *) key);
53 
54 #ifdef PCL_64BIT
55  return (uintptr_t) pcl_farmhash64_seed(key, key_len, 0);
56 #else
57  return (uintptr_t) pcl_farmhash32_seed(key, key_len, 0);
58 #endif
59 }

61 pcl_htable_t *
62 pcl_htable(int capacity)
63 {
64  capacity = ipcl_htable_capacity(max(capacity, 0));
65 
66  if(capacity < 0)
67  return R_TRC(NULL);
68 
69  pcl_htable_t *ht = pcl_zalloc(sizeof(pcl_htable_t));
70 
71  ht->key_len = 0;
72  ht->capacity = capacity;
73  ht->min_loadfac = MIN_LOADFAC;
74  ht->max_loadfac = MAX_LOADFAC;
75  ht->key_equals = default_key_equals;
76  ht->hashcode = default_hashcode;
77  ipcl_htable_init(ht->capacity, &ht->entries, &ht->hashidx);
78 
79  return ht;
80 }
