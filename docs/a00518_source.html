46 pcl_htable_entry_t *
47 ipcl_htable_lookup(pcl_htable_t *ht, const void *key, uintptr_t *code, int *index)
48 {
49  uintptr_t c = ht->hashcode(key, ht->key_len, ht->userp);
50 
51  if(code)
52  *code = c;
53 
54  int i = (int) (c % ht->capacity);
55 
56  if(index)
57  *index = i;
58 
59  /* search collisions */
60  for(pcl_htable_entry_t *e = ht->entries[i]; e; e = e->next)
61  if(ht->key_equals(e->key, key, ht->key_len, ht->userp))
62  return e;
63 
64  return R_SETERR(NULL, PCL_ENOTFOUND);
65 }
