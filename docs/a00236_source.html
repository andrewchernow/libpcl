35 pcl_digest(const char *algo, const void *data, size_t *len)
36 {
37  if(!data || !len)
38  return R_SETERR(NULL, PCL_EINVAL);
39 
40  pcl_digest_t *d = pcl_digest_create(algo);
41 
42  if(!d)
43  return R_TRC(NULL);
44 
45  if(pcl_digest_update(d, data, *len))
46  return R_TRC(NULL);
47 
48  int mdlen;
49  void *digest = pcl_digest_final(d, &mdlen);
50 
51  pcl_digest_free(d);
52 
53  if(!digest)
54  return R_TRC(NULL);
55 
56  *len = mdlen;
57  return digest;
58 }
