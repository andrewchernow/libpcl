37 pcl_buf_vputf(pcl_buf_t *b, const void *format, va_list ap)
38 {
39  int len;
40 
41  /* query for formatted length */
42  va_list ap2;
43  va_copy(ap2, ap);
44  len = b->chrsize == 1 ? pcl_vsprintf(NULL, 0, format, ap2) : pcl_vswprintf(NULL, 0, format, ap2);
45  va_end(ap2);
46 
47  if(len < 0)
48  return SETERR(-len);
49 
50  /* binary adds a NUL to every string */
51  pcl_buf_grow(b, b->mode == PclBufBinary ? len + 1 : len);
52 
53  /* do a vsprintf buffer put */
54  if(b->chrsize == 1)
55  len = pcl_vsprintf(b->data + b->pos, b->size - b->pos, format, ap);
56  else
57  len = pcl_vswprintf(((wchar_t *) b->data) + b->pos, b->size - b->pos, format, ap);
58 
59  /* no NUL included yet */
60  b->pos += len;
61 
62  if(b->mode == PclBufBinary)
63  {
64  /* count NUL already written: "Abc\0" */
65  b->pos++;
66  len++;
67  }
68 
69  /* add always present NUL: binary buffers "Abc\0\0", text buffers "Abc\0" */
70  memset(b->data + b->pos * b->chrsize, 0, b->chrsize);
71 
72  if(b->len < b->pos)
73  b->len = b->pos;
74 
75  return len;
76 }
