37 hex2num(ipcl_json_state_t *s, uint32_t *codepoint)
38 {
39  char *end, num[5];
40 
41  memcpy(num, s->next, 4);
42  num[4] = 0;
43 
44  if(pcl_strtoi(num, &end, 16, (int *) codepoint) < 0)
45  return R_TRCMSG(NULL, "parsing \\u sequence '%s'", num);
46 
47  /* not all characters were decoded, bogus \u0000 input */
48  if(*end)
49  JSON_THROW("invalid unicode character: '%s'", num);
50 
51  s->next += 4;

55 /** Convert a UTF-16 codepoint (including possible surrogate pairs) to UTF-8. This first converts
56  * a \u0000 codepoint. If it is a surrogate, this will parse the second point in the pair.
57  * Thus, this function can consume an additional \u0000 if a surrogate pair is detected. After
58  * getting a valid unicode codepoint, it is converted to UTF-8 and stored within provided buffer.
59  * @param s pointer to a json parser state obejct
60  * @param b pointer to a buffer
61  * @return pointer to state object or NULL on error
62  */
63 static ipcl_json_state_t *
64 utf16_to_utf8(ipcl_json_state_t *s, pcl_buf_t *b)
65 {
66  uint32_t pair[2], code;
67 
68  if(s->end - s->next < 4)
69  JSON_THROW("unexpected end of input: \\u0000 encoding missing digits", 0);
70 
71  if(!hex2num(s, &pair[0]))
72  return NULL;
73 
74  /* UTF-16 surrogate pair */
75  if(pair[0] >= 0xd800 && pair[0] <= 0xdbff)
76  {
77  if(s->end - s->next < 6)
78  JSON_THROW("unexpected end of input: \\u0000 encoding missing digits", 0);
79 
80  if(*s->next != '\\' || s->next[1] != 'u')
81  JSON_THROW("invalid unicode character: missing UTF-16 surrogate '%.12s'", s->next - 6);
82 
83  s->next += 2;
84 
85  if(!hex2num(s, &pair[1]))
86  return NULL;
87 
88  /* invalid UTF-16 surrogate */
89  if(pair[1] < 0xdc00 || pair[1] > 0xdfff)
90  JSON_THROW("invalid unicode character: surrogate '%.6s'", s->next - 6);
91 
92  code = 0x10000;
93  code += (pair[0] & 0x03ff) << 10;
94  code += (pair[1] & 0x03ff);
95  }
96  else
97  {
98  code = pair[0];
99  }
100 
101  /* convert to utf-8 */
102  if(code <= 0x7F) /* ascii */
103  {
104  pcl_buf_putchar(b, (char) code);
105  }
106  else if(code <= 0x07FF) /* 2-byte sequence */
107  {
108  pcl_buf_putchar(b, (char) (((code >> 6) & 0x1F) | 0xC0));
109  pcl_buf_putchar(b, (char) (((code >> 0) & 0x3F) | 0x80));
110  }
111  else if(code <= 0xFFFF) /* 3-byte sequence */
112  {
113  pcl_buf_putchar(b, (char) (((code >> 12) & 0x0F) | 0xE0));
114  pcl_buf_putchar(b, (((code >> 6) & 0x3F) | 0x80));
115  pcl_buf_putchar(b, (char) (((code >> 0) & 0x3F) | 0x80));
116  }
117  else if(code <= 0x10FFFF) /* 4-byte sequence */
118  {
119  pcl_buf_putchar(b, (char) (((code >> 18) & 0x07) | 0xF0));
120  pcl_buf_putchar(b, (char) (((code >> 12) & 0x3F) | 0x80));
121  pcl_buf_putchar(b, (char) (((code >> 6) & 0x3F) | 0x80));
122  pcl_buf_putchar(b, (char) (((code >> 0) & 0x3F) | 0x80));
123  }
124  else
125  {
126  JSON_THROW("invalid unicode character: 0x%08x", code);
127  }
128 
129  return s;
130 }

132 char *
133 ipcl_json_parse_string(ipcl_json_state_t *s)
134 {
135  if(*s->next != '"')
136  JSON_THROW("expected opening quote", 0);
137 
138  pcl_buf_t buf;
139  pcl_buf_t *b = pcl_buf_init(&buf, 32, PclBufText);
140 
141  s->ctx = s->next++;
142 
143  /* first encode any escapes */
144  for(; *s->next; s->next++)
145  {
146  if(*s->next != '\\')
147  {
148  if(*s->next == '"')
149  break;
150 
151  pcl_buf_putchar(b, *s->next);
152  continue;
153  }
154 
155  /* we have an escape sequence */
156  switch(*++s->next)
157  {
158  case '\\':
159  case '/':
160  case '"':
161  pcl_buf_putchar(b, *s->next);
162  break;
163 
164  case 'n':
165  pcl_buf_putchar(b, '\n');
166  break;
167 
168  case 'r':
169  pcl_buf_putchar(b, '\r');
170  break;
171 
172  case 'b':
173  pcl_buf_putchar(b, '\b');
174  break;
175 
176  case 'f':
177  pcl_buf_putchar(b, '\f');
178  break;
179 
180  case 't':
181  pcl_buf_putchar(b, '\t');
182  break;
183 
184  case 'u':
185  {
186  s->next++;
187 
188  if(!utf16_to_utf8(s, b))
189  {
190  pcl_buf_clear(b);
191  return NULL;
192  }
193 
194  s->next--;
195  break;
196  }
197 
198  /* json is very specific about what constitutes a valid escape sequence */
199  default:
200  JSON_THROW("invalid escape sequence within string \\%c", *s->next);
201  }
202  }
203 
204  if(*s->next != '"')
205  {
206  pcl_buf_clear(b);
207  JSON_THROW("expected closing double quote", 0);
208  }
209 
210  s->next++;
211 
212  /* now check that the string is valid utf8, which the above unescaping did not do */
213  if(ipcl_json_utf8check(b->data, b->len) < 0)
214  return R_TRC(NULL);
215 
216  return b->data;
217 }
