37 pcl_ssl_connect(pcl_ssl_t *ssl, const char *host, int port, int *want_io)
38 {
39  int x;
40 
41  if(!want_io)
42  want_io = &x;
43  *want_io = 0;
44 
45  if(ssl->flags & (PCL_SSL_PASSIVE | PCL_SSL_SERVER))
46  return SETERRMSG(PCL_EINVAL, "Not an SSL client", 0);
47 
48  if(!ssl->sock)
49  {
50  if(strempty(host) || port < 0 || port > 65535)
51  return SETERR(PCL_EINVAL);
52 
53  if(ipcl_ssl_configure_socket(ssl, host, port))
54  {
55  pcl_ssl_close(ssl);
56  return TRC();
57  }
58  }
59 
60  if(!pcl_socket_isalive(ssl->sock) && pcl_connect(ssl->sock))
61  {
62  if(pcl_errno != PCL_EAGAIN)
63  pcl_ssl_close(ssl);
64  return TRC();
65  }
66 
67  if(!ssl->ssl)
68  {
69  ssl->ssl = SSL_new(ssl->ctx);
70  SSL_set_fd(ssl->ssl, (int)pcl_socket_fd(ssl->sock));
71  (void) SSL_set_tlsext_host_name(ssl->ssl, host); // SNI host
72  }
73 
74  /* Negotiate */
75  ERR_clear_error();
76  switch(SSL_get_error(ssl->ssl, SSL_connect(ssl->ssl)))
77  {
78  case SSL_ERROR_NONE:
79  break;
80 
81  case SSL_ERROR_ZERO_RETURN:
82  pcl_ssl_close(ssl);
83  return SETERR(PCL_ECONNRESET);
84 
85  case SSL_ERROR_WANT_READ:
86  *want_io = PCL_SSL_WANTREAD;
87  return SETERR(PCL_EAGAIN);
88 
89  case SSL_ERROR_WANT_WRITE:
90  *want_io = PCL_SSL_WANTWRITE;
91  return SETERR(PCL_EAGAIN);
92 
93  case SSL_ERROR_SYSCALL:
94  {
95  const char *e = sslerror(ssl);
96  if(e)
97  SETLASTERRMSG("%s", e);
98  else
99  SETLASTERR();
100  pcl_ssl_close(ssl);
101  return -1;
102  }
103 
104  default:
105  {
106  const char *e = sslerror(ssl);
107  if(e)
108  SETERRMSG(PCL_ESSL, "%s", e);
109  else
110  SETERR(PCL_ESSL);
111  pcl_ssl_close(ssl);
112  return -1;
113  }
114  }
115 
116  if(ssl->flags & PCL_SSL_VERIFY_PEER)
117  {
118  long v_err;
119  X509 *peer = SSL_get_peer_certificate(ssl->ssl);
120 
121  if(!peer)
122  {
123  pcl_ssl_close(ssl);
124  return SETERRMSG(PCL_ENOTFOUND, "No certificate presented by peer", 0);
125  }
126 
127  if((v_err = SSL_get_verify_result(ssl->ssl)) != X509_V_OK)
128  {
129  pcl_ssl_close(ssl);
130  return SETERRMSG(PCL_ESSL, "[%ld] Failed to verify peer certificate", v_err);
131  }
132  }
133 
134  return 0;
135 }
