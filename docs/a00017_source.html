39 /** Freeze the current thread's error while executing code.
40  * This is useful when performing cleanup code and it is desired to preserve error.
41  * In addition to freezing the per-thread error, this also preserves the current OS error code.
42  *
43  * @code
44  * FREEZE_ERR
45  * (
46  * cleanup_files(files_arr);
47  * cleanup_memory(table);
48  * etc...
49  * );
50  * @endcode
51  * @param code_fragment one or more statments to execute
52  */
53 #define FREEZE_ERR(code_fragment) do{ \
54  pcl_err_freeze(true); \
55  uint32_t __save_oserr__ = pcl_oserrno; \
56  { code_fragment; }; \
57  pcl_setoserrno(__save_oserr__); \
58  pcl_err_freeze(false); \
59 }while(0)

77  * @return -1 if PCL error code is not PCL_OKAY(0), 0 otherwise

253 /** Adds a stack trace to the current thread's error.
254  * When a nested function fails, known to have already set an error/trace, tracing macros can
255  * be used to track the call stack.
256  * @code
257  * if(func_sets_pcl_error() < 0)
258  * return TRC();
259  * @endcode
260  * @return -1 if PCL error code is not PCL_OKAY(0), 0 otherwise
261  */
262 #define TRC() TRCMSG(NULL, 0)

264 /** Adds a stack trace to the current thread's error with an error message.
265  * When a nested function fails, known to have already set an error/trace, tracing macros can
266  * be used to track the call stack.
267  * @code
268  * if(func_sets_pcl_error(value) < 0)
269  * return TRCMSG("bad value: %d", value);
270  * @endcode
271  * @param _msgfmt error message as a printf-style format string
272  * @param ... optional variable arguments for the printf-style _msgfmt
273  * @return -1 if PCL error code is not PCL_OKAY(0), 0 otherwise
274  */
275 #define TRCMSG(_msgfmt, ...) pcl_err_trace(PCL_LOCATION_ARGS, _msgfmt, __VA_ARGS__)
