41 pcl_proc_pipe(pcl_file_t *pipes[2], int rd_oflags, int wr_oflags)
42 {
43  pcl_fd_t fds[2];
44 
45 #ifdef PCL_WINDOWS
46  char name[64];
47  uint32_t id[2];
48  DWORD busize = 65536;
49  DWORD open_mode = PIPE_ACCESS_INBOUND;
50  DWORD pipe_mode = PIPE_TYPE_BYTE | PIPE_WAIT;
51  DWORD file_flags = FILE_ATTRIBUTE_NORMAL;
52  SECURITY_ATTRIBUTES sa = {0};
53 
54  pcl_rand(id, (int)sizeof(id));
55  sprintf(name, "\\\\.\\Pipe\\%08x%08x%08x", pcl_proc_self(), id[0], id[1]);
56 
57  /* create the read end of the pipe (server) */
58  sa.nLength = (DWORD)sizeof(sa);
59  sa.bInheritHandle = (rd_oflags & PCL_O_CLOEXEC) == 0;
60  if(rd_oflags & PCL_O_NONBLOCK)
61  open_mode |= FILE_FLAG_OVERLAPPED;
62 
63  fds[0] = CreateNamedPipeA(name, open_mode, pipe_mode, 1, busize, busize, 0, &sa);
64  if(fds[0] == INVALID_HANDLE_VALUE)
65  return SETLASTERR();
66 
67  /* create the write end of the pipe (client) */
68  sa.bInheritHandle = (wr_oflags & PCL_O_CLOEXEC) == 0;
69  if(wr_oflags & PCL_O_NONBLOCK)
70  file_flags |= FILE_FLAG_OVERLAPPED;
71 
72  fds[1] = CreateFileA(name, GENERIC_WRITE, 0, &sa, OPEN_EXISTING, file_flags, NULL);
73  if(fds[1] == INVALID_HANDLE_VALUE)
74  {
75  SETLASTERR();
76  CloseHandle(fds[0]);
77  return -1;
78  }
79 
80 #else
81 
82  /* pipe2() exists but only on newer linux kernels, osx doesn't appear to
83  * have it at all. pipe2() allows specifying nonblock and cloexec flags.
84  */
85  if(pipe(fds))
86  return SETLASTERR();
87 
88  for(int i = 0; i < 2; i++)
89  {
90  int oflags = i == 0 ? rd_oflags : wr_oflags;
91 
92  if(oflags & PCL_O_NONBLOCK)
93  {
94  int n = fcntl(fds[i], F_GETFL);
95 
96  if(n < 0)
97  SETLASTERR();
98  else if((n = fcntl(fds[i], F_SETFL, n | O_NONBLOCK)) < 0)
99  SETLASTERR();
100 
101  if(n < 0)
102  {
103  close(fds[0]);
104  close(fds[1]);
105  return -1;
106  }
107  }
108 
109  if(oflags & PCL_O_CLOEXEC)
110  {
111  int n = fcntl(fds[i], F_GETFD);
112 
113  if(n < 0)
114  SETLASTERR();
115  else if((n = fcntl(fds[i], F_SETFD, n | FD_CLOEXEC)) < 0)
116  SETLASTERR();
117 
118  if(n < 0)
119  {
120  close(fds[0]);
121  close(fds[1]);
122  return -1;
123  }
124  }
125  }
126 #endif
127 
128  rd_oflags &= ~(PCL_O_WRONLY | PCL_O_RDWR);
129  pipes[0] = ipcl_file_alloc(fds[0], 0);
130  ipcl_file_init(pipes[0], rd_oflags | PCL_O_RDONLY);
131 
132  wr_oflags &= ~(PCL_O_RDONLY | PCL_O_RDWR);
133  pipes[1] = ipcl_file_alloc(fds[1], 0);
134  ipcl_file_init(pipes[0], wr_oflags | PCL_O_WRONLY);
135 
136  return 0;
137 }
