43 pcl_net_host2addr(const char *host, struct sockaddr *sa)
44 {
45  if(strempty(host))
46  return BADARG();
47 
48  if(*host == '/')
49  {
50  if(!sa)
51  {
52  if(strlen(host) >= PCL_MAXUNIXSOCKPATH)
53  return SETERR(PCL_EOVERFLOW);
54 
55  return 0;
56  }
57 
58  struct sockaddr_un *u = (struct sockaddr_un *)sa;
59 
60  u->sun_family = AF_UNIX;
61 
62  if(!pcl_strcpy(u->sun_path, sizeof(u->sun_path), host))
63  return SETERR(PCL_EOVERFLOW);
64 
65  return 0;
66  }
67 
68  struct addrinfo *info = NULL;
69  int r = getaddrinfo(host, NULL, NULL, &info);
70 
71  if(r)
72  return SETERR(pcl_net_ai2pcl(r));
73 
74  if(sa)
75  memcpy(sa, info->ai_addr, info->ai_addrlen);
76 
77  freeaddrinfo(info);
78  return 0;
79 }
