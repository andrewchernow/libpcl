37 {
38  int flags = 0;
39 
40  if(oflags & PCL_O_WRONLY)
41  flags |= PCL_FF_WRITE;
42  else if(oflags & PCL_O_RDWR)
43  flags |= PCL_FF_RDWR;
44  else
45  flags |= PCL_FF_READ;
46 
47  if(oflags & PCL_O_APPEND)
48  flags |= PCL_FF_APPEND;
49 
50  if(oflags & PCL_O_NONBLOCK)
51  flags |= PCL_FF_NONBLOCK;
