41 static void
42 linux_distro(char *out, size_t len)
43 {
44  int r;
45  char buf[512];
46  FILE *fp = fopen("/etc/os-release", "r");
47 
48  *out = 0;
49 
50  /* note: all "modern" distros should have this: /etc/os-release */
51  if(fp)
52  {
53  char name[64];
54  char pretty_name[100];
55  char version[64];
56 
57  *name = *pretty_name = *version = 0;
58 
59  while(fgets(buf, sizeof(buf), fp))
60  {
61  if(!*name && !pcl_strnicmp(buf, "NAME=", 5))
62  {
63  char *p = pcl_strtrimset(buf + 5, "\"\n", true, true);
64  pcl_strcpy(name, sizeof(name), p);
65  }
66  else if(!*version && !pcl_strnicmp(buf, "VERSION=", 8))
67  {
68  char *p = pcl_strtrimset(buf + 8, "\"\n", true, true);
69  pcl_strcpy(version, sizeof(version), p);
70  }
71  else if(!*pretty_name && !pcl_strnicmp(buf, "PRETTY_NAME=", 12))
72  {
73  char *p = pcl_strtrimset(buf + 12, "\"\n", true, true);
74  pcl_strcpy(pretty_name, sizeof(pretty_name), p);
75  }
76  }
77 
78  fclose(fp);
79 
80  /* Quite a few distros have PRETTY_NAMEs that exclude the codename.
81  * So prefer NAME + VERSION and fallback to PRETTY_NAME or just NAME.
82  * See: https://gist.github.com/natefoo/814c5bf936922dad97ff
83  */
84  if(*name && *version)
85  pcl_sprintf(out, len, "%s %s", name, version);
86  else if(*pretty_name)
87  pcl_strcpy(out, len, pretty_name);
88  else if(*name)
89  pcl_strcpy(out, len, name);
90 
91  if(*out)
92  return;
93  }
94 
95  /* first fallback */
96  if((fp = popen("lsb_release -d 2>/dev/null", "r")))
97  {
98  *buf = 0;
99  (void) fgets(buf, sizeof(buf), fp);
100  r = pclose(fp);
101 
102  /* indicates success, check buffer as well */
103  if(WIFEXITED(r) && WEXITSTATUS(r) == 0 && pcl_strnicmp(buf, "Description:", 12) == 0)
104  {
105  /* skip 'Description:' and trim spaces */
106  pcl_strcpy(out, len, pcl_strtrim(buf + 12));
107  if(*out)
108  return;
109  }
110  }
111 
112  /* last ditch effort: older Linux machines: check most common */
113  static const char *release_files[] = {
114  "/etc/redhat-release",
115  "/etc/SuSE-release",
116  "/etc/fedora-release",
117  "/etc/gentoo-release",
118  "/etc/slackware-release",
119  "/etc/centos-release",
120  "/etc/slackware-version",
121  };
122 
123  for(*buf = 0, r=0; r < countof(release_files); r++)
124  {
125  const char *file = release_files[r];
126 
127  if((fp = fopen(file, "r")) && fgets(buf, sizeof(buf), fp) && *buf)
128  pcl_strcpy(out, len, buf);
129 
130  if(fp)
131  fclose(fp);
132 
133  if(*out)
134  return;
135  }
136 
137  /* upgrade your system */
138  pcl_strcpy(out, len, "Linux");
139 }

141 static uint64_t
142 linux_meminfo(void)
143 {
144  uint64_t mem=0;
145  char buf[256];
146  FILE *fp = fopen("/proc/meminfo", "r");
147 
148  if(!fp)
149  return mem;
150 
151  *buf = 0;
152  while(fgets(buf, sizeof(buf), fp))
153  {
154  if(!pcl_strnicmp(buf, "MemTotal:", 9))
155  {
156  char *unit;
157 
158  /* Example format: '12 kB' */
159  mem = strtoll(buf + 9, &unit, 10);
160  pcl_strltrim(unit);
161 
162  /* appears to always be in kB but play it safe */
163  switch((unsigned char) tolower((int) *unit))
164  {
165  case 'k': mem *= 1024; break;
166  case 'm': mem *= 1048576; break;
167  case 'g': mem *= 1073741824; break;
168  case 't': mem *= 1099511627776LL; break;
169  }
170 
171  break;
172  }
173  }
174 
175  fclose(fp);
176 
177  return mem;
178 
179 }

181 static void
182 linux_cpuinfo(pcl_sysinfo_t *info)
183 {
184  char buf[1024]; // account for "flags" row
185  uint16_t phy_ids[1024]; // limit to 1024 physical CPUs
186  FILE *fp = fopen("/proc/cpuinfo", "r");
187 
188  if(!fp)
189  return;
190 
191  while(fgets(buf, sizeof(buf), fp) && info->cpu_sockets < countof(phy_ids))
192  {
193  if(!pcl_strnicmp(buf, "model name", 10))
194  {
195  /* note: model could be different across physical CPUs. We don't
196  * handle that unusual case. We assume all are identical.
197  */
198  if(!*info->cpu_model)
199  {
200  char *p = pcl_strtrim(strchr(buf + 10, ':') + 1);
201  pcl_strcpy(info->cpu_model, sizeof(info->cpu_model), p);
202  }
203  }
204  else if(!pcl_strnicmp(buf, "physical id", 11))
205  {
206  int i, id = (int) strtol(strchr(buf + 11, ':') + 1, NULL, 10);
207 
208  /* have we seen this CPU (socket) yet? */
209  for(i=0; i < info->cpu_sockets; i++)
210  if(phy_ids[i] == id)
211  break;
212 
213  /* new physical CPU socket */
214  if(i == info->cpu_sockets)
215  phy_ids[info->cpu_sockets++] = id;
216 
217  /* hyper-threading or not, add to core count */
218  info->cpu_cores++;
219  }
220  }
221 
222  fclose(fp);
223 }
