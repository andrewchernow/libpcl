44 int
45 ipcl_win32_sd_create(SECURITY_DESCRIPTOR *sd, const tchar_t *owner,
46  const tchar_t *group, mode_t mode)
47 {
48  DWORD acl_len;
49  PACL dacl = NULL;
50  PSID owner_sid = NULL;
51  PSID group_sid = NULL;
52  PSID world_sid = NULL;
53 
54  ipcl_win32_sd_init(sd);
55 
56  if(mode & S_IRWXU)
57  {
58  if(!strempty(owner))
59  {
60  int type;
61 
62  if((owner_sid = ipcl_win32_name_to_sid(owner, &type)) && type != SidTypeUser)
63  {
64  pcl_free(owner_sid);
65  owner_sid = NULL;
66  }
67  }
68 
69  /* no owner provided or it was not found, use process owner */
70  if(!owner_sid)
71  {
72  HANDLE htok;
73 
74  if(ipcl_win32_process_token(&htok) == 0)
75  {
76  TOKEN_USER *tokUser;
77 
78  if(ipcl_win32_token_into(htok, TokenUser, &tokUser) == 0)
79  {
80  owner_sid = ipcl_win32_sid_dup(tokUser->User.Sid);
81  pcl_free(tokUser);
82  }
83 
84  CloseHandle(htok);
85  }
86  }
87 
88  if(owner_sid && !SetSecurityDescriptorOwner(sd, owner_sid, false))
89  {
90  pcl_free(owner_sid);
91  return SETLASTERR();
92  }
93  }
94 
95  if(mode & S_IRWXG)
96  {
97  if(!strempty(group))
98  {
99  int type;
100 
101  if((group_sid = ipcl_win32_name_to_sid(group, &type)) && !SID_ISGROUP(type))
102  {
103  pcl_free(group_sid);
104  group_sid = NULL;
105  }
106  }
107 
108  /* no group provided or it was not found, use process group */
109  if(!group_sid)
110  {
111  HANDLE htok;
112 
113  if(ipcl_win32_process_token(&htok) == 0)
114  {
115  TOKEN_PRIMARY_GROUP *tokGroup;
116 
117  if(ipcl_win32_token_into(htok, TokenPrimaryGroup, &tokGroup) == 0)
118  {
119  group_sid = ipcl_win32_sid_dup(tokGroup->PrimaryGroup);
120  pcl_free(tokGroup);
121  }
122 
123  CloseHandle(htok);
124  }
125  }
126 
127  if(group_sid && !SetSecurityDescriptorGroup(sd, group_sid, false))
128  {
129  SetSecurityDescriptorOwner(sd, NULL, false);
130  pcl_free(owner_sid);
131  pcl_free(group_sid);
132  return SETLASTERR();
133  }
134  }
135 
136  if(mode & S_IRWXO)
137  world_sid = ipcl_win32_world_sid();
138 
139  acl_len = sizeof(ACL);
140 
141  if(owner_sid)
142  acl_len += (sizeof(ACCESS_ALLOWED_ACE) - sizeof(DWORD) + GetLengthSid(owner_sid));
143 
144  if(group_sid)
145  acl_len += (sizeof(ACCESS_ALLOWED_ACE) - sizeof(DWORD) + GetLengthSid(group_sid));
146 
147  if(world_sid)
148  acl_len += (sizeof(ACCESS_ALLOWED_ACE) - sizeof(DWORD) + GetLengthSid(world_sid));
149 
150  dacl = (PACL) pcl_malloc(acl_len);
151 
152  if(!InitializeAcl(dacl, acl_len, ACL_REVISION))
153  {
154  CLEANUP;
155  return SETLASTERR();
156  }
157 
158  if(owner_sid)
159  {
160  if(!AddAccessAllowedAceEx(dacl, ACL_REVISION, OBJECT_INHERIT_ACE | CONTAINER_INHERIT_ACE,
161  ipcl_win32_mode_to_rights(mode, S_IRUSR), owner_sid))
162  {
163  CLEANUP;
164  return SETLASTERR();
165  }
166  }
167 
168  if(group_sid)
169  {
170  if(!AddAccessAllowedAceEx(dacl, ACL_REVISION, OBJECT_INHERIT_ACE | CONTAINER_INHERIT_ACE,
171  ipcl_win32_mode_to_rights(mode, S_IRGRP), group_sid))
172  {
173  CLEANUP;
174  return SETLASTERR();
175  }
176  }
177 
178  if(world_sid)
179  {
180  if(!AddAccessAllowedAceEx(dacl, ACL_REVISION, OBJECT_INHERIT_ACE | CONTAINER_INHERIT_ACE,
181  ipcl_win32_mode_to_rights(mode, S_IROTH), world_sid))
182  {
183  CLEANUP;
184  return SETLASTERR();
185  }
186  }
187 
188  if(!SetSecurityDescriptorDacl(sd, true, dacl, false))
189  {
190  CLEANUP;
191  return SETLASTERR();
192  }
193 
194  /* NO CLEANUP HERE! Note that SetSecurityDescriptorDacl is specifically documented to require
195  * an "absolute" descriptor, meaning its members are pointers to structures rather than
196  * being self-referencing (offsets to contiguous data). This is why each SID is allocated.
197  * For the security descriptor to be usable after this call, we can't free any *_sid or the
198  * DACL. Again SetSecurityDescriptorDacl docs state "The DACL is referenced by, not copied
199  * into, the security descriptor". One must call ipcl_win32_sd_clear to cleanup properly.
200  *
201  * The exception: world_sid is only used by AddAccessAllowedAceEx and is not referenced by the
202  * created SECURITY_DESCRIPTOR.
203  */
204 
205  if(world_sid)
206  FreeSid(world_sid);
207 
208  return 0;
209 }
