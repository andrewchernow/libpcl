47 /* Converts an IP string address to a binary internet address. af can
48  * be AF_INET or AF_INET6. This function is equivilant to inet_pton,
49  * which is not always available. Returns 0 on success and
50  * -1 on error. 'inaddr' is expected to be sizeof(struct in_addr) if
51  * AF_INET is specified and sizeof(struct in6_addr) if AF_INET6 is specified.
52  * Supports AF_INET, AF_INET6.
53  */
54 PCL_EXPORT int pcl_net_ip2inet(int af, const char *ipstr, void *inaddr);
55 
56 /* Converts a binary inet addr (in_addr or in6_addr) to a string. Same as
57  * inet_ntop, which is not always available. If af is AF_INET,
58  * 'inaddr' is a struct in_addr, if AF_INET6 its a struct in6_addr.
59  * Returns NULL on error. NOTE: this wraps pcl_net_addr2ip().
60  * Supports AF_INET, AF_INET6 and AF_UNIX.
61  */
62 PCL_EXPORT char *pcl_net_inet2ip(int af, const void *inaddr, char *ipbuf, size_t len);
63 
64 /* Converts an IP or hostname to a binary socket address. It is
65  * recommended that if 'sa' is not NULL, is should be wide enough for an
66  * IPv6 sockaddr_in6 or a AF_UNIX path. NOTE: sa can be NULL, which will just
67  * verify the host. Returns zero on success and -1 on error.
68  * Supports AF_INET, AF_INET6 and AF_UNIX.
69  */
70 PCL_EXPORT int pcl_net_host2addr(const char *host, struct sockaddr *sa);
71 
72 /* Converts a sockaddr to its string representation and stores the
73  * result at the address pointed to by ipbuf, which is len bytes
74  * wide. Returns a pointer to ipbuf on success and NULL on error.
75  * Supports AF_INET, AF_INET6 and AF_UNIX.
76  */
77 PCL_EXPORT char *pcl_net_addr2ip(struct sockaddr *sa, char *ipbuf, size_t len);
