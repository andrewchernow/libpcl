36 /** @defgroup dir Directory Management
37  * The directory API provides a suite of function for managing and traversing directories.
38  * #### Basic Usage
39  * @code
40  * #include <pcl/init.h>
41  * #include <pcl/error.h>
42  * #include <pcl/dir.h>
43  * #include <pcl/io.h>
44  *
45  * int pcl_main(int argc, pchar_t *argv)
46  * {
47  * // always first
48  * pcl_init();
49  *
50  * if(argc < 2)
51  * PANIC("Missing directory argument", 0);
52  *
53  * pcl_dir_t *dir = pcl_opendir(argv[1]);
54  *
55  * if(!dir)
56  * PANIC("Failed to open directory: '%Ps'", argv[1]);
57  *
58  * pcl_dirent_t ent;
59  * const pchar_t *dirpath = pcl_dir_path(dir, NULL);
60  *
61  * while(pcl_readdir(dir, &ent, NULL))
62  * {
63  * char *type;
64  *
65  * switch(ent.type)
66  * {
67  * case PclDirentFile: type = "file"; break;
68  * case PclDirentDir: type = "directory"; break;
69  * case PclDirentFifo: type = "named pipe"; break;
70  * case PclDirentLink: type = "symbolic link"; break;
71  * case PclDirentChar: type = "character device"; break;
72  * case PclDirentBlock: type = "block device"; break;
73  * case PclDirentSocket: type = "unix domain socket"; break;
74  * default: type = "unknown";
75  * }
76  *
77  * // %/ is replaced with the platform-specific path separator
78  * pcl_printf("%s] %Ps%/%Ps\n", type, dirpath, ent.name);
79  * }
80  *
81  * pcl_closedir(dir);
82  *
83  * // when readdir is done, it returns false and pcl_errno is PCL_ENOMORE
84  * // note that 'dirpath' was freed with above pcl_closedir
85  * if(pcl_errno != PCL_ENOMORE)
86  * PANIC("readdir failed: '%Ps'", argv[1]);
87  *
88  * return 0;
89  * }
90  * @endcode
91  * @{
92  */
93 #include <pcl/types.h>

147 /** Read an entry from a directory handle.

149  * @param ent pointer to an entry buffer to receive the results
