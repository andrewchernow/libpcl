59 /** Represents a PCL error context, with optional stack trace. This can be thought of
60  * as Unix `errno` on steroids.
61  */
62 struct tag_pcl_err_ctx
63 {
64  /** Freezes context, making it immutable. It should be noted that a frozen context object
65  * cannot be cleared or freed. The only exception is the per-thread context, which can and
66  * will be freed if the thread exits.
67  *
68  * Typically, freezing an object is only useful when used with the PCL error API. For example:
69  * freezing the per-thread context before performing cleanup after an error, where the
70  * cleanup could overwrite the "real" error: ie. `freeze=true; cleanup_code; freeze=false;`.
71  *
72  * Internally, all printf-style context functions freeze the object before performing output
73  * and thaw before returning.
74  *
75  * @see FREEZE_ERR
76  */
77  bool frozen;
78 
79  /** PCL error code: PCL_EXXX */
80  int err;
81 
82  /** Platform-specific error code: Unix errno, Windows Error (ie: GetLastError, _dos_errno),
83  * etc. This can be zero even when 'err' is not. However, if this is non-zero 'err' will
84  * also be non-zero.
85  */
86  uint32_t oserr;
87 
88  /** Stack trace implemented as a linked list. This is NULL if there is no stack trace. */
89  pcl_err_trace_t *strace;
90 
91  /** An output buffer used by output formatting functions like pcl_err_ctx_fprintf. This is
92  * lazily-created so can be NULL. This is really just for internal use and provides for
93  * a cached/allocated buffer to use for multiple output calls. This provides no context
94  * functionality, implementation detail.
95  */
96  pcl_buf_t *buffer;
97 };
