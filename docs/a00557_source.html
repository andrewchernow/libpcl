39 FILE *
40 pcl_popen(const tchar_t *command, ...)
41 {
42  va_list ap;
43  tchar_t *cmd;
44  int flags = 0;
45  FILE *stream = NULL;
46 
47  if(strempty(command))
48  return R_SETERRMSG(NULL, PCL_EINVAL, "command must be provided", 0);
49 
50  while(isspace(*command))
51  command++;
52 
53  /* get the mode */
54  if(*command == '<')
55  flags |= PCL_EXEC_STDOUT;
56  else if(*command == '>')
57  flags |= PCL_EXEC_STDIN;
58  else if(*command == '^') /* like the FISH shell */
59  flags |= PCL_EXEC_STDERR;
60  else
61  return R_SETERRMSG(NULL, PCL_EINVAL, "invalid command mode '%tc'", *command);
62 
63  /* optional flags, no spaces between mode and flags! */
64  command++;
65 
66  if(*command == '!')
67  {
68  command++;
69 
70  if(flags & PCL_EXEC_STDIN)
71  flags |= PCL_EXEC_STDIN_NB;
72  else if(flags & PCL_EXEC_STDOUT)
73  flags |= PCL_EXEC_STDOUT_NB;
74  else if(flags & PCL_EXEC_STDERR)
75  flags |= PCL_EXEC_STDERR_NB;
76  }
77 
78  while(isspace(*command))
79  command++;
80 
81  /* format command */
82  va_start(ap, command);
83  int r = pcl_vastprintf(&cmd, command, ap);
84  va_end(ap);
85 
86  if(r < 0)
87  return R_SETERR(NULL, -r);
88 
89  /* perform an pcl_proc_exec() */
90  pcl_proc_exec_t x = {.command = cmd};
91 
92  if(pcl_proc_exec(&x, flags))
93  {
94  pcl_free(cmd);
95  return R_TRC(NULL);
96  }
97 
98  pcl_file_t *f = x.f_stdin;
99 
100  if(!f)
101  f = x.f_stdout;
102  if(!f)
103  f = x.f_stderr;
104 
105  if(!(stream = pcl_fdopen(f)))
106  {
107  pcl_file_close(f);
108  TRC();
109  }
110 
111  pcl_free(cmd);
112 
113  return stream;
114 }
