45 // combo of tmpfile and tempnam
46 FILE *
47 pcl_tmpfile(const pchar_t *dir, const pchar_t *prefix, const pchar_t *suffix,
48  pchar_t **pathp, bool keep)
49 {
50  pchar_t *tmpdirs[4];
51  size_t prefix_len = prefix ? pcl_pcslen(prefix) : 0;
52  size_t suffix_len = suffix ? pcl_pcslen(suffix) : 0;
53 
54  if(prefix_len + suffix_len > PCL_MAXNAME)
55  return R_SETERRMSG(NULL, PCL_EINVAL, "prefix and suffix cannot exceed maximum "
56  "file name length '%d'", PCL_MAXNAME);
57 
58  if(!prefix)
59  prefix = _P("");
60  if(!suffix)
61  suffix = _P("");
62 
63  tmpdirs[0] = (pchar_t *) dir;
64 
65 #ifdef PCL_WINDOWS
66  tmpdirs[1] = pcl_getenv(_P("TMP"));
67  tmpdirs[2] = pcl_getenv(_P("TEMP"));
68 #else
69  tmpdirs[1] = pcl_getenv(_P("TMPDIR"));
70  tmpdirs[2] = P_tmpdir;
71 #endif
72 
73  tmpdirs[3] = _P(".");
74 
75  pchar_t *path = NULL;
76  pcl_file_t *file = NULL;
77  size_t rand_len = PCL_MAXNAME - (prefix_len + suffix_len);
78  int oflags = PCL_O_CREAT | PCL_O_EXCL | PCL_O_RDWR;
79 
80  rand_len = min(RANDLEN, rand_len);
81  if(!keep)
82  oflags |= PCL_O_TEMP;
83 
84  for(int i = 0; i < countof(tmpdirs); i++)
85  {
86  pchar_t *tmpdir = tmpdirs[i];
87 
88  if(!tmpdir)
89  continue;
90 
91  int attempts = 0;
92  pcl_stat_t st;
93  uint8_t rand[RANDLEN];
94  char randhex[(RANDLEN * 2) + 1];
95 
96  if(pcl_stat(tmpdir, &st) || !S_ISDIR(st.mode))
97  continue;
98 
99  /* first +1 is path separator and 2nd +1 is NUL */
100  size_t path_len = pcl_pcslen(tmpdir) + 1 + prefix_len + rand_len + suffix_len + 1;
101 
102  path = pcl_malloc(path_len * sizeof(pchar_t));
103 
104  do
105  {
106  /* generate random part of temp name */
107  pcl_rand(rand, (int) rand_len);
108  pcl_hex_encode(randhex, sizeof(randhex), rand, rand_len);
109 
110  /* create tmp path */
111  pcl_spprintf(path, path_len, _P("%Ps%/%Ps%s%Ps"), tmpdir, prefix, randhex, suffix);
112 
113  /* create tmp file */
114  file = pcl_file_open(path, oflags, S_IRUSR | S_IWUSR);
115 
116  if(file)
117  goto END_LOOPS;
118  }
119  while(++attempts < 10); // try 10 times then move to next tmpdir
120 
121  path = pcl_free(path);
122  } // for tmpdirs
123 
124 END_LOOPS:
125 
126  pcl_free_safe(tmpdirs[1]);
127 #ifdef PCL_WINDOWS
128  pcl_free_safe(tmpdirs[2]);
129 #endif
130 
131  if(!file)
132  return R_SETERR(NULL, PCL_ENOTFOUND);
133 
134  if(pathp)
135  *pathp = path;
136  else
137  pcl_free(path);
138 
139  /* convert to a FILE stream */
140  FILE *fp = pcl_fdopen(file);
141 
142  if(!fp)
143  {
144  pcl_file_close(file);
145  return R_TRC(NULL);
146  }
147 
148  return fp;
149 }
