42 pcl_vector_t *
43 ipcl_json_parse_array(ipcl_json_state_t *s)
44 {
45  if(*s->next++ != '[')
46  JSON_THROW("expected opening array '['", 0);
47 
48  ipcl_json_skipws(s);
49  s->ctx = s->next;
50 
51  pcl_vector_t *arr = pcl_vector_create(8, sizeof(pcl_json_value_t), value_cleanup);
52 
53  /* empty array */
54  if(*s->next == ']')
55  {
56  s->next++;
57  return arr;
58  }
59 
60  do
61  {
62  pcl_json_value_t valbuf;
63  pcl_json_value_t *val = ipcl_json_parse_value(s, &valbuf);
64 
65  if(!val)
66  {
67  pcl_vector_free(arr);
68  return NULL;
69  }
70 
71  if(!pcl_vector_append(arr, val))
72  {
73  pcl_vector_free(arr);
74  ipcl_json_value_clear(val); // not managed by vector
75  return R_TRCMSG(NULL, "failed to add value to array", 0);
76  }
77 
78  if(!ipcl_json_skipws(s))
79  {
80  pcl_vector_free(arr);
81  return NULL;
82  }
83 
84  if(*s->next == ']')
85  {
86  s->next++;
87  return arr;
88  }
89 
90  if(*s->next++ != ',')
91  {
92  pcl_vector_free(arr);
93  JSON_THROW("missing comma after value", 0);
94  }
95 
96  if(!ipcl_json_skipws(s))
97  {
98  pcl_vector_free(arr);
99  return NULL;
100  }
101  }
102  while(s->next < s->end);
103 
104  pcl_vector_free(arr);
105  JSON_THROW("unexpected end of input", 0);
106 }
