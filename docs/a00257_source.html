36 pcl_cipher_reset(pcl_cipher_t *ciph, const void *iv, const void *aad, uint32_t aad_len)
37 {
38  if(!ciph || !iv)
39  return BADARG();
40 
41  if(!memcmp(EVP_CIPHER_CTX_original_iv(ciph->ctx), iv, EVP_CIPHER_CTX_iv_length(ciph->ctx)))
42  return SETERRMSG(PCL_EINVAL, "attempt to reset %s cipher with same IV", ciph->algo);
43 
44  /* The IV should always be changed between cycles! We enforce this. However, the key should
45  * never be changed for a single pcl_cipher_t instance. The whole goal is to maintain the
46  * key for a series of operations that requires the same key: a set of files, set of network
47  * packets, etc. We do not allow changing the 'enc' value, thus the -1.
48  */
49  if(EVP_CipherInit_ex(ciph->ctx, NULL, NULL, NULL, iv, -1) == 0)
50  return SETERRMSG(PCL_ECRYPT, "Failed to reset %s context", ciph->algo);
51 
52  if(aad && aad_len && pcl_cipher_isaead(ciph))
53  {
54  int n;
55 
56  if(EVP_CipherUpdate(ciph->ctx, NULL, &n, (const unsigned char *) aad, (int) aad_len) == 0)
57  return SETERRMSG(PCL_ECRYPT, "Failed to set AAD: %s", ciph->algo);
58  }
59 
60  return 0;
61 }
