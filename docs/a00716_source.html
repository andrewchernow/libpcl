38 static void
39 walk_path(pcl_json_t *node, const pcl_json_path_t *path, pcl_array_t *results)
40 {
41  switch(path->type)
42  {
43  case PclPathRoot:
44  {
45  if(path->next == NULL)
46  pcl_array_append(results, pcl_json_ref(node, 1));
47  else
48  walk_path(node, path->next, results);
49 
50  break;
51  }
52 
53  case PclPathMember:
54  {
55  if(pcl_json_isobj(node))
56  {
57  pcl_json_t *mbr = pcl_json_objget(node, path->member);
58 
59  if(mbr)
60  {
61  if(path->next == NULL)
62  pcl_array_append(results, pcl_json_ref(mbr, 1));
63  else
64  walk_path(mbr, path->next, results);
65  }
66  }
67 
68  break;
69  }
70 
71  case PclPathRecursiveDescent:
72  {
73  if(pcl_json_isobj(node))
74  {
75  int index = 0;
76  pcl_htable_entry_t *ent;
77 
78  while((ent = pcl_htable_iter(node->object, &index)))
79  {
80  if(path->next->type == PclPathMember && !strcmp(path->next->member, ent->key))
81  walk_path(node, path->next, results);
82  else
83  walk_path(ent->value, path, results);
84  }
85  }
86  else if(pcl_json_isarr(node))
87  {
88  for(int i = 0; i < node->array->count; i++)
89  {
90  pcl_json_t *elem = node->array->elements[i];
91 
92  if(path->next->type == PclPathElement && path->next->index == i)
93  walk_path(node, path->next, results);
94  else
95  walk_path(elem, path, results);
96  }
97  }
98 
99  break;
100  }
101 
102  case PclPathWildcardMember:
103  {
104  if(!pcl_json_isobj(node))
105  {
106  pcl_array_append(results, pcl_json_ref(node, 1));
107  break;
108  }
109 
110  int index = 0;
111  pcl_htable_entry_t *ent;
112 
113  while((ent = pcl_htable_iter(node->object, &index)))
114  {
115  if(path->next)
116  walk_path(ent->value, path->next, results);
117  else
118  pcl_array_append(results, pcl_json_ref(ent->value, 1));
119  }
120 
121  break;
122  }
123 
124  case PclPathElement:
125  {
126  if(pcl_json_isarr(node))
127  {
128  pcl_err_freeze(true);
129  pcl_json_t *elem = pcl_array_get(node->array, path->index);
130  pcl_err_freeze(false);
131 
132  if(elem)
133  {
134  if(path->next == NULL)
135  pcl_array_append(results, pcl_json_ref(elem, 1));
136  else
137  walk_path(elem, path->next, results);
138  }
139  }
140 
141  break;
142  }
143 
144  case PclPathElementList:
145  {
146  if(!pcl_json_isarr(node))
147  break;
148 
149  for(int i = 0; i < path->idx_list->count; i++)
150  {
151  int *index = pcl_vector_get(path->idx_list, i);
152 
153  pcl_err_freeze(true);
154  pcl_json_t *elem = pcl_array_get(node->array, *index);
155  pcl_err_freeze(false);
156 
157  if(elem)
158  {
159  if(path->next == NULL)
160  pcl_array_append(results, pcl_json_ref(elem, 1));
161  else
162  walk_path(elem, path->next, results);
163  }
164  }
165 
166  break;
167  }
168 
169  case PclPathWildcardElement:
170  {
171  if(!pcl_json_isarr(node))
172  {
173  pcl_array_append(results, pcl_json_ref(node, 1));
174  break;
175  }
176 
177  for(int i = 0; i < node->array->count; i++)
178  {
179  pcl_json_t *elem = node->array->elements[i];
180 
181  if(path->next == NULL)
182  pcl_array_append(results, pcl_json_ref(elem, 1));
183  else
184  walk_path(elem, path->next, results);
185  }
186 
187  break;
188  }
189 
190  case PclPathElementSlice:
191  {
192  if(!pcl_json_isarr(node))
193  break;
194 
195  int start = path->idx_slice.start;
196  int end = path->idx_slice.end ? path->idx_slice.end : node->array->count;
197 
198  if(start < 0)
199  start = node->array->count + path->idx_slice.start;
200 
201  if(end < 0)
202  end = node->array->count + path->idx_slice.end;
203 
204  for(int i = start; i < end; i += path->idx_slice.step)
205  {
206  pcl_err_freeze(true);
207  pcl_json_t *elem = pcl_array_get(node->array, i);
208  pcl_err_freeze(false);
209 
210  if(elem)
211  {
212  if(path->next == NULL)
213  pcl_array_append(results, pcl_json_ref(elem, 1));
214  else
215  walk_path(elem, path->next, results);
216  }
217  }
218 
219  break;
220  }
221 
222  default:
223  break;
224  }
225 }
