36 /** @defgroup init Initialize Application
37  * Responisble for intializing the PCL library. ::pcl_init must be called before any other
38  * PCL function. This is typically accomplished by calling it from within the
39  * application's \c main function. This module also includes a \c getopt implementation
40  * since calling ::pcl_init and argument parsing are typically both done in \c main.
41  * #### Basic Usage
42  * @code
43  * #include <pcl/init.h>
44  * #include <pcl/error.h>
45  * #include <stdlib.h>
46  *
47  * int pcl_tmain(int argc, tchar_t **argv)
48  * {
49  * // always call first
50  * pcl_init();
51  *
52  * // a requires argument, b has optional argument and c & d have no arguments
53  * const tchar_t *optstr = _T("a:b::cd");
54  *
55  * // match the 'val' member of pcl_option_t to the 'optstr' short options (not required)
56  * pcl_option_t options[] = {
57  * {_T("apple"), PclReqArg, _T('a')},
58  * {_T("banana"), PclOptArg, _T('b')},
59  * {_T("cherry"), PclNoArg, _T('c')}
60  * // 'd' is not mapped to a long option
61  * };
62  *
63  * // initialize option state
64  * pcl_optstate_t *state = pcl_initopt(argc, argv, optstr, options, countof(options));
65  *
66  * if(!state)
67  * PANIC("failed to initialize option state", 0);
68  *
69  * int opt;
70  * tchar_t *value;
71  * pcl_option_t *longopt;
72  *
73  * // 3rd argument longopt can be NULL, which is common
74  * while((opt = pcl_getopt(state, &value, &longopt)) > 0)
75  * {
76  * tchar_t *longname = longopt ? longopt->name : _T("<short-only>");
77  *
78  * switch(opt)
79  * {
80  * case _T('a'):
81  * printf("a:%ts] %ts\n", longname, value);
82  * break;
83  *
84  * case _T('b'):
85  * printf("b:%ts] %s\n", longname, value ? value : _T("<no-value>"));
86  * break;
87  *
88  * case _T('c'):
89  * printf("c:%ts] set\n", longname);
90  * break;
91  *
92  * case _T('d'):
93  * default:
94  * printf("d:%ts] set\n", longname); // prints "d:<short-only>] set\n"
95  * }
96  * }
97  *
98  * pcl_free(state);
99  *
100  * if(opt == -1)
101  * PANIC("getopt failed", 0);
102  *
103  * return app_start();
104  * }
105  * @endcode
106  * @{
107  */

158 /** Intialize command-line argument parser. This behaves very similar to POSIX \c getopt_long.
159  * Some differences are
160  * * This doesn't permute (rearrange) arguments
161  * * No support for \c '+' or \c POSIXLY_CORRECT
162  * * opaque pcl_optstate_t manages state rather than global variables
163  * * returns 0 when complete, not -1. -1 indicates an error
164  * * ? never returned to indicate an error, -1 is always returned
165  * * this never outputs to stderr, it instead returns -1 and set pcl_errno
166  *
167  * @warning If the short options \a optstr defines different argument policies than the
168  * matching pcl_option_t.val, the behavior is undefined. If time is taken to map short => long,
169  * make sure policies match: ::PclNoArg, ::PclReqArg or ::PclOptArg.
170  *
171  * @param argc argument count, typically from \c main
172  * @param argv pointer to an array of arguments, typically from \c main
173  * @param optstr option characters. If a character is followed by a colon, the option requires
174  * an argument. If followed by two colons, the option takes an optional argument. In either case,
175  * the argument can be specified as: \c "-Ovalue", \c "-O=value" or \c "-O value". When dealing
176  * with an optional argument, the 3rd form excludes the value if it begins with a '-': ie.
177  * \c "-O -value". This is not the case for a required argument. This can be \c NULL as long as
178  * the \a options parameter is not \c NULL.
179  * @param options array of option structures defining legitimate long options. The option
180  * structure defines an option \a name, option argument \a policy and a \a val used as the
181  * return value when a long option is found. Typically, \a val is set to its corresponding short
182  * option character, but this is not a requirement. The argument can be specified as
183  * \c "--long=value" or \c "--long value". As with short options, the 2nd form excludes optional
184  * arguments that begin with a '-': ie. \c "-long -value". This is not the case for a required
185  * argument. This can be \c NULL as long as the \a optstr parameter is not \c NULL.
186  * @param num_options number of option structures in the \a options parameter.
187  * @return pointer to an option state structure, that must be freed by caller, or \c NULL on error.
188  * An error only occurs if there is an invalid argument.
189  */
190 PCL_EXPORT pcl_optstate_t *pcl_initopt(int argc, tchar_t **argv, const tchar_t *optstr,
191  pcl_option_t *options, int num_options);

193 /** Parse command-line arguments.
194  * @param state pointer to a getopt state structure intialized via ::pcl_initopt
195  * @param value pointer to a string to receive the value, if any. This can be \c NULL. If
196  * this is \c NULL, there is no way to get an option's argument value. This value must be
197  * copied to persist after argument parsing, since it is direct pointer to the \c argv array
198  * provided during ::pcl_initopt.
199  * @param longopt pointer to an option to receive the matching option object. The comparison
200  * works for long or short option parsing. When a short option is found with a matching
201  * pcl_option_t.val value, that option object is written here. For long options, the comparison
202  * is based off pcl_option_t.name. If a short option has no matching \a val within long option
203  * objects, or they were not provided to ::pcl_initopt, then this is set to \c NULL.
204  * This is an optional parameter and can be \c NULL.
205  * @return The short option character or long option pcl_option_t.val. When all arguments
206  * have been parsed, this returns 0. On error, -1 is returned.
207  */
208 PCL_EXPORT int pcl_getopt(pcl_optstate_t *state, tchar_t **value, pcl_option_t **longopt);
