49 pcl_array_t *
50 pcl_net_dnstxtrec(const char *host)
51 {
52  pcl_array_t *arr;
53 
54 #ifdef PCL_WINDOWS
55  DNS_RECORD *rlist = NULL;
56  DNS_STATUS status = DnsQuery_A(host, DNS_TYPE_TEXT, 0, NULL, &rlist, NULL);
57 
58  if(status)
59  return R_SETOSERRMSG(NULL, status, "Cannot resolve: '%s'", host);
60 
61  DNS_TXT_DATA *data = &rlist->Data.Txt;
62  arr = pcl_array(data->dwStringCount, pcl_cleanup_ptr);
63 
64  for(DWORD i=0; i < data->dwStringCount; i++)
65  pcl_array_append(arr, pcl_pcs_to_utf8(data->pStringArray[i], 0, NULL));
66 
67  DnsRecordListFree(rlist, DnsFreeRecordList);
68 
69 #else
70  ns_rr rr;
71  ns_msg handle;
72  uint8_t answer[65535];
73  const uint8_t *data;
74  int len = res_query(host, C_IN, T_TXT, answer, (int) sizeof(answer));
75 
76  if(len == -1)
77  return R_SETERRMSG(NULL, pcl_net_dns2pcl(h_errno), "Cannot resolve: '%s'", host);
78 
79  if(ns_initparse(answer, len, &handle) < 0)
80  return R_SETLASTERR(NULL);
81 
82  if(ns_parserr(&handle, ns_s_an, 0, &rr))
83  return R_SETLASTERR(NULL);
84 
85  arr = pcl_array(4, pcl_cleanup_ptr);
86  len = ns_rr_rdlen(rr);
87  data = ns_rr_rdata(rr);
88 
89  for(int n = 0; n < len;)
90  {
91  int q = data[n++]; /* first byte is txt length */
92  pcl_array_append(arr, pcl_strndup((const char *) (data + n), q));
93  n += q;
94  }
95 #endif
96 
97  return arr;
98 }
