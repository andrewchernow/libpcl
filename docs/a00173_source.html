36 pcl_buf_grow(pcl_buf_t *b, int len)
37 {
38  if(!b)
39  return R_SETERR(NULL, PCL_EINVAL);
40 
41  int avail = (int) b->size - b->pos;
42 
43  if(len >= (int) avail)
44  {
45  b->size = (b->size + (len - avail)) * 2;
46  b->data = pcl_realloc(b->data, b->size * b->chrsize);
47  }
48 
49  return b;
50 }
