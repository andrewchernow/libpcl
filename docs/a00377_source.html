37 static char *
38 escape_jsonstr(pcl_buf_t *b, const char *s)
39 {
40  pcl_buf_reset(b);
41 
42  for(; *s; s++)
43  {
44  switch(*s)
45  {
46  case '\\':
47  pcl_buf_putstr(b, "\\\\");
48  break;
49 
50  case '/':
51  pcl_buf_putstr(b, "\\/");
52  break;
53 
54  case '"':
55  pcl_buf_putstr(b, "\\\"");
56  break;
57 
58  case '\b':
59  pcl_buf_putstr(b, "\\b");
60  break;
61 
62  case '\t':
63  pcl_buf_putstr(b, "\\t");
64  break;
65 
66  case '\n':
67  pcl_buf_putstr(b, "\\n");
68  break;
69 
70  case '\f':
71  pcl_buf_putstr(b, "\\f");
72  break;
73 
74  case '\r':
75  pcl_buf_putstr(b, "\\r");
76  break;
77 
78  default:
79  {
80  /* control characters */
81  if(*s < 0x20)
82  pcl_buf_putf(b, "\\u%04x", (int) *s);
83  else
84  pcl_buf_putchar(b, *s);
85  }
86  }
87  }
88 
89  return b->data;
90 }

92 char *
93 pcl_err_vjson(const char *message, va_list ap)
94 {
95  pcl_err_t *err = pcl_err_get();
96  pcl_buf_t *b = err->buffer;
97  pcl_buf_t escbuf;
98 
99  pcl_buf_init(&escbuf, 256, PclBufText);
100 
101  if(!b)
102  b = err->buffer = pcl_buf_init(NULL, 512, PclBufText);
103  else
104  pcl_buf_reset(b);
105 
106  char *msg = NULL;
107 
108  if(!strempty(message))
109  pcl_vasprintf(&msg, message, ap);
110 
111  pcl_buf_putf(b,"{\"err\":%d,\"oserr\":%u,\"msg\":", err->err, err->oserr);
112 
113  if(strempty(msg))
114  pcl_buf_putstr(b, "null");
115  else
116  pcl_buf_putf(b, "\"%s\"", escape_jsonstr(&escbuf, msg));
117 
118  pcl_free_safe(msg);
119 
120  pcl_buf_putstr(b, ",\"strace\":[");
121 
122  for(pcl_err_trace_t *t = err->strace; t; t = t->next)
123  {
124  pcl_buf_putstr(b, "{\"file\":");
125 
126  if(strempty(t->file))
127  pcl_buf_putstr(b, "null");
128  else
129  pcl_buf_putf(b, "\"%s\"", escape_jsonstr(&escbuf, t->file));
130 
131  pcl_buf_putstr(b, ",\"func\":");
132 
133  if(strempty(t->func))
134  pcl_buf_putstr(b, "null");
135  else
136  pcl_buf_putf(b, "\"%s\"", escape_jsonstr(&escbuf, t->func));
137 
138  pcl_buf_putf(b, ",\"line\":%d,\"msg\":", t->line);
139 
140  if(strempty(t->msg))
141  pcl_buf_putstr(b, "null");
142  else
143  pcl_buf_putf(b, "\"%s\"", escape_jsonstr(&escbuf, t->msg));
144 
145  pcl_buf_putchar(b, '}');
146 
147  if(t->next)
148  pcl_buf_putchar(b, ',');
149  }
150 
151  pcl_buf_putstr(b, "]}");
152 
153  pcl_buf_clear(&escbuf);
154 
155  /* steal data pointer */
156  char *json = b->data;
157  b->data = NULL;
158  pcl_buf_clear(b);
159 
160  return json;
161 }
