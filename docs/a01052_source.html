35 int
36 pcl_ssl_setpair(pcl_ssl_t *ssl, const char *cert, const char *key)
37 {
38  if(!ssl || !ssl->ctx || strempty(cert) || strempty(key))
39  return SETERR(PCL_EINVAL);
40 
41  ERR_clear_error();
42 
43  if(!SSL_CTX_use_certificate_file(ssl->ctx, cert, SSL_FILETYPE_PEM))
44  return SETERRMSG(PCL_ESSL,"Failed to load certificate: '%s', %s",
45  cert, sslerror(ssl));
46 
47  if(!SSL_CTX_use_PrivateKey_file(ssl->ctx, key, SSL_FILETYPE_PEM))
48  return SETERRMSG(PCL_ESSL,"Failed to load private key: '%s', %s",
49  key, sslerror(ssl));
50 
51  if(!SSL_CTX_check_private_key(ssl->ctx))
52  return SETERRMSG(PCL_ESSL, "Invalid cert:key pair: '%s:%s', %s",
53  cert, key, sslerror(ssl));
54 
55  /* only do this for clients, servers already send their certs with
56  * SSL_VERIFY_NONE. On clients, this forces cert requests to be sent
57  * by server.
58  */
59  if(!(ssl->flags & (PCL_SSL_PASSIVE | PCL_SSL_SERVER)))
60  SSL_CTX_set_verify(ssl->ctx, SSL_VERIFY_PEER, NULL);
61 
62  return 0;
63 }
