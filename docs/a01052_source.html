51 XFUNC(istr)(const xchar *haystack, const xchar *needle)
52 {
53  xint c = toxlower((xint) *needle);
54 
55  if(c == '\0')
56  return (xchar *) haystack;
57 
58  for(; *haystack; haystack++)
59  {
60  if(toxlower((xint) *haystack) == c)
61  {
62  for(size_t i = 0;;)
63  {
64  if(needle[++i] == '\0')
65  return (xchar *) haystack;
66 
67  if(toxlower((xint) haystack[i]) != toxlower((xint) needle[i]))
68  break;
69  }
70  }
71  }
72 
73  return NULL;
74 }
