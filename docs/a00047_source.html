69 /* Convert a _level bit to a trace bit */
70 #define PCL_LOGLEV_TOTRACE(_level) \
71  (((_level) & (unsigned)PCL_LOGLEV_MASK) << (unsigned)PCL_LOGTRC_LSH)

73 /* Test if a _level bit has it's trace bit set (enabled/on) */
74 #define PCL_LOGLEV_HASTRACE(_level) ((_level) & PCL_LOGLEV_TOTRACE(_level))
