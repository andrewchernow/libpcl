37 int
38 ipcl_htable_rehash(pcl_htable_t *ht, bool grow)
39 {
40  int new_capacity = ipcl_htable_chkcapacity(
41  grow ? (uint64_t) ht->capacity << 1U : (uint64_t) ht->capacity >> 1U);
42 
43  int new_table_mask = new_capacity - 1;
44 
45  if(new_capacity < 0)
46  return TRC();
47 
48  int *new_entry_lookup;
49  pcl_htable_entry_t *new_entries;
50  ipcl_htable_init(new_capacity, &new_entries, &new_entry_lookup);
51 
52  /* recompute all non-deleted entries */
53  for(int count = 0, i = 0; i < ht->count_used; i++)
54  {
55  pcl_htable_entry_t *oldent = &ht->entries[i];
56 
57  /* skip deleted slots */
58  if(!oldent->key)
59  continue;
60 
61  pcl_htable_entry_t *ent = &new_entries[count];
62  ent->key = oldent->key;
63  ent->value = oldent->value;
64  ent->code = oldent->code;
65 
66  int hashidx = (int) (ent->code & new_table_mask);
67  int entidx = new_entry_lookup[hashidx];
68 
69  if(entidx != -1)
70  {
71  pcl_htable_entry_t *head = &new_entries[entidx];
72  ent->next = head->next;
73  head->next = count;
74  }
75  else
76  {
77  ent->next = -1;
78  new_entry_lookup[hashidx] = count;
79  }
80 
81  count++;
82  }
83 
84  /* frees entry_lookup[] as well */
85  pcl_free(ht->entries);
86 
87  ht->capacity = new_capacity;
88  ht->table_mask = new_table_mask;
89  ht->entries = new_entries;
90  ht->entry_lookup = new_entry_lookup;
91  ht->count_used = ht->count; // reset count_used, we defrag'd entries array
92 
93  return 0;
94 }
