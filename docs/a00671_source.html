40 pcl_socket_t *
41 pcl_accept(pcl_socket_t *sock)
42 {
43  int val = 0;
44  pcl_socket_t *cli = ipcl_socket_alloc();
45 
46 #if defined(PCL_LINUX) && defined(SOCK_NONBLOCK)
47  /* Saves a fcntl call to accept the socket and set it to nonblock
48  * in a single call, only available on linux.
49  */
50  val = sock->flags & PCL_SOCK_NONBLOCK ? SOCK_NONBLOCK : 0;
51 
52  cli->fd = accept4(sock->fd, (struct sockaddr *)cli->addr, (socklen_t *)&cli->sa_len, val);
53 
54  if(cli->fd != INVALID_SOCKET && val == SOCK_NONBLOCK)
55  cli->flags |= PCL_SOCK_NONBLOCK;
56 
57 #else
58  cli->fd = accept(sock->fd, (struct sockaddr *) cli->addr, (socklen_t *) &cli->sa_len);
59 #endif
60 
61  if(cli->fd == INVALID_SOCKET)
62  {
63  int err = pcl_err_os2pcl(pcl_sockerrno);
64 
65  pcl_socket_close(cli);
66  switch(err)
67  {
68  case PCL_EAGAIN: /* same as EINPROGRESS, PCL_EWOULDBLOCK */
69  case PCL_EINPROGRESS: /* connect would block, try again */
70  case PCL_EINTR: /* interrupted */
71  err = PCL_EAGAIN;
72  default:
73  break;
74  }
75 
76  return R_SETERR(NULL, err);
77  }
78 
79  /* For some reason, accept doesn't properly populate the AF_UNIX sun_path. So, manually
80  * copy it from server sock.
81  */
82  if(cli->addr->sa_family == AF_UNIX)
83  {
84  memcpy(cli->addr, sock->addr, sock->sa_len);
85  cli->sa_len = sock->sa_len;
86  }
87 
88  cli->sa_protocol = sock->sa_protocol;
89  cli->sa_socktype = sock->sa_socktype;
90  cli->flags |= PCL_SOCK_CONNECTED;
91 
92  /* if accept4 was used, pcl_socket_setnonblocking does nothing since 'cli' is already
93  * flagged as nonblocking.
94  */
95  if(sock->flags & PCL_SOCK_NONBLOCK)
96  pcl_socket_setnonblocking(cli, true);
97 
98  setsockopt(cli->fd, SOL_SOCKET, SO_LINGER, (char *) &val, sizeof(val));
99  return cli;
100 }
