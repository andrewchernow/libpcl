36 void
37 pcl_unix_siginfo(char *buf, size_t bufl, siginfo_t *info)
38 {
39  int l = 0;
40  int n;
41  char *reason = "unknown";
42 
43  if(!info)
44  {
45  if(buf && bufl > 0)
46  *buf = 0;
47  return;
48  }
49 
50  char code[64]={0};
51 
52  if(info->si_code > 0)
53  pcl_sprintf(code, sizeof(code), " code=%d", info->si_code);
54 
55  l = pcl_sprintf(buf, bufl, "%s%s", pcl_strsignal(info->si_signo), code);
56  if(l < 0)
57  return;
58 
59  /* append reason, si_value and si_addr if signal fills it in */
60  switch(info->si_signo)
61  {
62  case SIGILL:
63  {
64  switch(info->si_code)
65  {
66 #ifdef ILL_ILLOPC
67  case ILL_ILLOPC:
68  reason = "Illegal opcode";
69  break;
70 #endif
71 
72 #ifdef ILL_ILLOPN
73  case ILL_ILLOPN:
74  reason = "Illegal operand";
75  break;
76 #endif
77 
78 #ifdef ILL_ILLADR
79  case ILL_ILLADR:
80  reason = "Illegal addressing mode";
81  break;
82 #endif
83 
84 #ifdef ILL_ILLTRP
85  case ILL_ILLTRP:
86  reason = "Illegal trap";
87  break;
88 #endif
89 
90 #ifdef ILL_PRVOPC
91  case ILL_PRVOPC:
92  reason = "Privileged opcode";
93  break;
94 #endif
95 
96 #ifdef ILL_PRVREG
97  case ILL_PRVREG:
98  reason = "Privileged register";
99  break;
100 #endif
101 
102 #ifdef ILL_COPROC
103  case ILL_COPROC:
104  reason = "Coprocessor error";
105  break;
106 #endif
107 
108 #ifdef ILL_BADSTK
109  case ILL_BADSTK:
110  reason = "Internal stack error";
111  break;
112 #endif
113 
114  default:
115  break;
116  }
117 
118  goto PRINTERROR;
119  }
120 
121  case SIGFPE:
122  {
123  switch(info->si_code)
124  {
125  case FPE_INTDIV:
126  reason = "Integer divide-by-zero";
127  break;
128 
129  case FPE_INTOVF:
130  reason = "Integer overflow";
131  break;
132 
133  case FPE_FLTDIV:
134  reason = "Floating point divide-by-zero";
135  break;
136 
137  case FPE_FLTOVF:
138  reason = "Floating point overflow";
139  break;
140 
141  case FPE_FLTUND:
142  reason = "Floating point underflow";
143  break;
144 
145  case FPE_FLTRES:
146  reason = "Floating point inexact result";
147  break;
148 
149  case FPE_FLTINV:
150  reason = "Invalid floating point operation";
151  break;
152 
153  case FPE_FLTSUB:
154  reason = "Subscript out of range";
155  break;
156  }
157 
158  goto PRINTERROR;
159  }
160 
161  case SIGSEGV:
162  {
163 #ifdef SEGV_MAPERR
164  if(info->si_code == SEGV_MAPERR)
165  {
166  reason = "Address not mapped";
167  goto PRINTERROR;
168  }
169 #endif
170 
171 #ifdef SEGV_ACCERR
172  if(info->si_code == SEGV_ACCERR)
173  reason = "Invalid permissions";
174 #endif
175  goto PRINTERROR;
176  }
177 
178  case SIGBUS:
179  {
180 #ifdef BUS_ADRALN
181  if(info->si_code == BUS_ADRALN)
182  {
183  reason = "Invalid address alignment";
184  goto PRINTERROR;
185  }
186 #endif
187 
188 #ifdef BUS_ADRERR
189  if(info->si_code == BUS_ADRERR)
190  {
191  reason = "Non-existent physical address";
192  goto PRINTERROR;
193  }
194 #endif
195 
196 #ifdef BUS_OBJERR
197  if(info->si_code == BUS_OBJERR)
198  reason = "Object-specific hardware error";
199 #endif
200 
201 PRINTERROR:
202  n = pcl_sprintf(buf+l, bufl-l, ": reason=%s, addr=0x%lx",
203  reason, info->si_addr);
204 
205  if(n < 0)
206  return;
207 
208  l += n;
209  break;
210  }
211 
212  case SIGCHLD:
213  {
214  char tmp[32];
215  *tmp = 0;
216  switch(info->si_code)
217  {
218 #ifdef CLD_EXITED
219  case CLD_EXITED:
220  reason = "exited";
221  sprintf(tmp, ", exitcode=%d", info->si_status);
222  break;
223 #endif
224 
225 #ifdef CLD_KILLED
226  case CLD_KILLED:
227  reason = "killed";
228  sprintf(tmp, ", chldsig=%s",
229  pcl_strsignal(info->si_value.sival_int));
230  break;
231 #endif
232 
233 #ifdef CLD_DUMPED
234  case CLD_DUMPED:
235  reason = "dumped";
236  sprintf(tmp, ", chldsig=%s",
237  pcl_strsignal(info->si_value.sival_int));
238  break;
239 #endif
240 
241 #ifdef CLD_TRAPPED
242  case CLD_TRAPPED:
243  reason = "trapped";
244  break;
245 #endif
246 
247 #ifdef CLD_STOPPED
248  case CLD_STOPPED:
249  reason = "stopped";
250  break;
251 #endif
252 
253 #ifdef CLD_CONTINUED
254  case CLD_CONTINUED:
255  reason = "continued";
256  break;
257 #endif
258 
259  default:
260  break;
261  }
262 
263  if((n = pcl_sprintf(buf+l, bufl-l, ": reason=%s%s", reason, tmp)) < 0)
264  return;
265 
266  l += n;
267  break;
268  }
269  }
270 
271  /* only append errors, instead of "0 Success" everywhere */
272  if(info->si_errno)
273  pcl_sprintf(buf+l, bufl-l, ", errno=%d, errmsg=%s", info->si_errno, strerror(info->si_errno));
274 }
