46  pcl_err_ctx_t *ctx = pcl_err_ctx();
47  ctx->frozen = true;
48 
49  if(file->flags & PCL_FF_FLOCK)
50  (void) pcl_file_lock(file, PCL_UNLOCK);
51 
52 #ifdef PCL_WINDOWS
53  bool badfd = file->fd == PCL_BADFD;
54 
55  if(file->flags & PCL_FF_NONBLOCK)
56  {
57  if(file->flags & PCL_FF_READ)
58  {
59  if(file->aio_read.pending && !badfd)
60  (void) CancelIoEx(file->fd, &file->aio_read.ol);
61 
62  if(file->aio_read.ol.hEvent)
63  CloseHandle(file->aio_read.ol.hEvent);
64  }
65 
66  if(file->flags & PCL_FF_WRITE)
67  {
68  if(file->aio_write.pending && !badfd)
69  (void) CancelIoEx(file->fd, &file->aio_write.ol);
70 
71  if(file->aio_write.ol.hEvent)
72  CloseHandle(file->aio_write.ol.hEvent);
73  }
74  }
75 
76  if(!badfd)
77  CloseHandle(file->fd);
78 
79 #else
80  if(file->fd != PCL_BADFD)
81  (void) close(file->fd);
82 #endif
83 
84  pcl_free(file);
