45 pcl_htable_t *
46 ipcl_json_parse_object(ipcl_json_state_t *s)
47 {
48  if(*s->next++ != '{')
49  JSON_THROW("expected opening object '{'", 0);
50 
51  if(!ipcl_json_skipws(s))
52  return NULL;
53 
54  s->ctx = s->next;
55 
56  pcl_htable_t *obj = pcl_htable_create(0);
57  obj->remove_entry = remove_entry;
58 
59  /* empty object */
60  if(*s->next == '}')
61  {
62  s->next++;
63  return obj;
64  }
65 
66  do
67  {
68  char *key = ipcl_json_parse_string(s);
69 
70  if(!key)
71  {
72  pcl_htable_free(obj);
73  return NULL;
74  }
75 
76  if(!ipcl_json_skipws(s))
77  {
78  pcl_htable_free(obj);
79  pcl_free(key);
80  return NULL;
81  }
82 
83  if(*s->next++ != ':')
84  {
85  pcl_htable_free(obj);
86  pcl_free(key);
87  JSON_THROW("expected value ':' separator", 0);
88  }
89 
90  pcl_json_value_t *jv = ipcl_json_parse_value(s, NULL);
91 
92  if(!jv)
93  {
94  pcl_htable_free(obj);
95  pcl_free(key);
96  return NULL;
97  }
98 
99  if(pcl_htable_put(obj, key, jv, true) < 0)
100  {
101  TRCMSG("object '%s' put failed", key);
102 
103  pcl_htable_free(obj);
104 
105  /* 'jv' and 'key' are not managed by 'obj' cuz put failed */
106  ipcl_json_value_clear(jv);
107  pcl_free(jv);
108  pcl_free(key);
109 
110  return NULL;
111  }
112 
113  if(!ipcl_json_skipws(s))
114  {
115  pcl_htable_free(obj);
116  return NULL;
117  }
118 
119  if(*s->next == '}')
120  {
121  s->next++;
122  return obj;
123  }
124 
125  if(*s->next++ != ',')
126  {
127  pcl_htable_free(obj);
128  JSON_THROW("missing comma after value", 0);
129  }
130 
131  if(!ipcl_json_skipws(s))
132  {
133  pcl_htable_free(obj);
134  return NULL;
135  }
136  }
137  while(s->next < s->end);
138 
139  pcl_htable_free(obj);
140  JSON_THROW("unexpected end of input", 0);
141 }
