42 int
43 pcl_log_vevent(pcl_log_event_t *e, const char *message, va_list ap)
44 {
45  if(!e)
46  return 0;
47 
48  char timestr[128];
49  char *msg = NULL;
50  char *location = NULL;
51  pcl_err_ctx_t *ctx = e->ctx ? e->ctx : pcl_err_ctx();
52 
53  ctx->frozen = 1;
54 
55  /* format msg */
56  if(!strempty(message))
57  (void) pcl_vasprintf(&msg, message, ap);
58 
59  /* format when...time */
60  pcl_tm_t when;
61 
62  if(e->level & PCL_LOGOPT_UTC)
63  pcl_gmtime(&e->when, &when);
64  else
65  pcl_localtime(&e->when, &when);
66 
67  (void) pcl_strftime(timestr, countof(timestr), PCL_LOG_TIMEFORMAT, &when);
68 
69  char file[512] = {0};
70  char func[200] = {0};
71  if(!strempty(e->file))
72  {
73  pcl_strcpy(file, sizeof(file), e->file);
74 #ifdef PCL_WINDOWS
75  pcl_strrepchr(file, '\\', '/');
76 #endif
77  }
78 
79  if(!strempty(e->func))
80  {
81  pcl_strcpy(func, sizeof(func), e->func);
82 #ifdef PCL_WINDOWS
83  pcl_strrepchr(func, '\\', '/');
84 #endif
85  }
86 
87  /* format location */
88  if(!strempty(file) || !strempty(func) || e->line > 0)
89  {
90  char linestr[16] = {0};
91 
92  if(e->line > 0)
93  (void) pcl_sprintf(linestr, countof(linestr), "(%d)", e->line);
94 
95  (void) pcl_asprintf(&location, "%s%s%s%s: ",
96  *file ? file : "",
97  *file && *func ? ":" : "",
98  *func ? func : "",
99  linestr);
100  }
101 
102  /* Example output: (see pcl_err_fprintf) - first line is log specific
103  *
104  * [Jun 11 05:27:12 facility.level] file:func(line): There was a problem with something
105  * some.c:func(463): optional trace message
106  * another.c:getuserid(2029)
107  * query.c:select(99): PCL_EBUF - buffer too small
108  * >An optional user message goes here
109  */
110 
111  int count = pcl_fprintf(e->stream, "[%s %s%s%s] %s%s\n",
112  timestr,
113  strempty(e->facility) ? "" : e->facility,
114  strempty(e->facility) ? "" : ".",
115  pcl_log_strlevel(e->level),
116  location ? location : "",
117  msg ? msg : "");
118 
119  if(PCL_LOGLEV_HASTRACE(e->level))
120  count += pcl_err_ctx_fprintf(ctx, e->stream, 2, NULL);
121 
122  fflush(e->stream);
123  pcl_free_safe(location);
124  pcl_free_safe(msg);
125  ctx->frozen = 0;
126 
127  /* Exit Policy: should this level cause an app `exit(3)` */
128  if(e->exit_levels & e->level)
129  {
130  /* callee provided an exit_handler */
131  if(e->exit_handler)
132  {
133  e->ctx = ctx;
134  e->exit_handler(e, e->exit_context);
135  }
136 
137  /* exit_handler was NULL or provided one didn't exit. No need to free ctx. */
138  exit(e->exit_code);
139  }
140 
141  return count;
142 }
