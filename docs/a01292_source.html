37 int
38 pcl_getuid(const char *name, uid_t *uidp)
39 {
40  int n;
41 
42  if(strempty(name))
43  return BADARG();
44 
45 #ifdef PCL_WINDOWS
46  char *s = NULL;
47  DWORD size = SECURITY_MAX_SID_SIZE;
48  char buf[SECURITY_MAX_SID_SIZE];
49  char dom[512];
50  DWORD dom_len = 512;
51  PSID sid = (PSID) buf;
52  SID_NAME_USE snu;
53 
54  if(!LookupAccountNameA(NULL, name, sid, &size, dom, &dom_len, &snu))
55  {
56  n = SETLASTERR();
57  }
58  else if(!ConvertSidToStringSidA(sid, &s))
59  {
60  n = SETLASTERR();
61  }
62  else
63  {
64  if(uidp)
65  pcl_strcpy(*uidp, sizeof(uid_t), s);
66  n = 0;
67  }
68 
69  if(s)
70  LocalFree(s);
71 
72 #else
73  char buf[4096];
74  struct passwd pwbuf, *pw = NULL;
75 
76  n = getpwnam_r(name, &pwbuf, buf, sizeof(buf), &pw);
77 
78  if(n == 0)
79  {
80  if(!pw)
81  n = SETERR(PCL_ENOTFOUND);
82  else if(uidp)
83  *uidp = pw->pw_uid;
84  }
85  else
86  {
87  n = SETOSERR(n);
88  }
89 #endif
90 
91  return n;
92 }
