35 static int
36 break_down_time(const pcl_time_t *_t, pcl_tm_t *tu, bool utc)
37 {
38  pcl_time_t t;
39  struct tm tm;
40 
41  if(!tu)
42  return SETERRMSG(PCL_EINVAL, "output storage cannot be null", 0);
43 
44  /* use now if _t is NULL*/
45  t = !_t ? pcl_time() : *_t;
46 
47 #ifdef PCL_WINDOWS
48  errno_t err = utc ? _gmtime64_s(&tm, &t.sec) : _localtime64_s(&tm, &t.sec);
49  if(err)
50  return SETOSERR(pcl_err_crt2os(err));
51 #else
52  time_t systime = (time_t) t.sec;
53  if(utc)
54  gmtime_r(&systime, &tm);
55  else
56  localtime_r(&systime, &tm);
57 #endif
58 
59  tu->tm_sec = tm.tm_sec;
60  tu->tm_min = tm.tm_min;
61  tu->tm_hour = tm.tm_hour;
62  tu->tm_mday = tm.tm_mday;
63  tu->tm_mon = tm.tm_mon;
64  tu->tm_year = tm.tm_year;
65  tu->tm_wday = tm.tm_wday;
66  tu->tm_yday = tm.tm_yday;
67  tu->tm_isdst = tm.tm_isdst;
68 
69 #ifdef PCL_WINDOWS
70  int daylight = 0;
71 
72  /* if daylight savings time, get offset in minutes */
73  if(tu->tm_isdst)
74  {
75  _get_daylight(&daylight);
76  daylight *= 60;
77  }
78 
79  _get_timezone(&tu->tm_gmtoff);
80  tu->tm_gmtoff -= daylight;
81 
82  /* Windows uses West of UTC rather than EAST ... translate */
83  tu->tm_gmtoff = -tu->tm_gmtoff;
84 
85  size_t n;
86  *tu->tm_zone = 0;
87  _get_tzname(&n, tu->tm_zone, sizeof(tu->tm_zone), tu->tm_isdst);
88 
89 #else
90  tu->tm_gmtoff = tm.tm_gmtoff;
91  tu->tm_zone = (char *) tm.tm_zone;
92 #endif
93 
94  tu->tm_nsec = t.nsec;
95  return 0;
96 }
