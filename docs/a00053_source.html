57 /* Indicates that the pcl_proc_exec command should be passed to the SHELL.
58  * On unix, this uses `$SHELL -c "your_command"` and on windows, this uses
59  *
60  * %COMSPEC% /C "your_command"
61  *
62  * This is only required for internal (builtin) commands. These are commands
63  * that are built into windows CMD.EXE or the Unix shell: like BASH `cd`.
64  * Thus, you must do `cmd.exe /C "command"` or `sh -c "command"` to exec them.
65  */
66 #define PCL_PREX_SHELL 0x0008
67 
68 /* If PCL_PREX_STDIN, PCL_PREX_STDOUT or PCL_PREX_STDERR are set, open them in
69  * non-blocking mode. Otherwise, this has no effect. Setting this flag will
70  * cause all standard streams to be non-blocking.
71  */
72 #define PCL_PREX_NONBLOCK 0x0010
73 
74 /* Populates the pcl_proc_exec_t.handle member with the process handle. This is
75  * only meaningful for Windows, it is the process HANDLE of the process
76  * pcl_proc_exec() just spawned. It is just the PID on Unix. On Windows, this
77  * HANDLE must be closed via pcl_proc_close(). On unix, pcl_proc_close() does
78  * nothing, although calling it in a multi-platform application is best practice.
79  * It keeps the app non-specific to any platform.
80  */
81 #define PCL_PREX_HANDLE 0x0020
82 
83 /* Sets the library search path within the new process. The library path is set
84  * to the value of pcl_proc_exec_t.ldpath; a colon separated list of paths (semi-colon
85  * on windows). If 'ldpath' is NULL, on Unix pcl_proc_exec tries pcl_proc_exec_t.cwd and
86  * then `getcwd`. If none of those work, pcl_proc_exec() ignores this flag and continues.
87  * On windows, if pcl_proc_exec_t.ldpath is NULL, pcl_proc_exec ignores this flag (continues)
88  * and there are no fallback attempts like on Unix. This is because on windows,
89  * the current working directory is already part of the library search path.
90  *
91  * These values must be NULL for pcl_proc_exec to try another one or assume they are
92  * unset. An empty string is completely valid and can be used to clear the library
93  * search path for this new process; but not effect the system-wide search paths.
94  * If no value can be obtained while this flag is set, the pcl_proc_exec call will
95  * give up and continue as if this flag was not set.
96  *
97  * Windows: added to PATH
98  * > https://docs.microsoft.com/en-us/cpp/build/search-path-used-by-windows-to-locate-a-dll
99  * Linux: added to LD_LIBRARY_PATH
100  * Darwin: added to DYLD_LIBRARY_PATH
101  */
102 #define PCL_PREX_LDPATH 0x0040
103 
104 /* same as PCL_O_CLOEXEC or PCL_O_NOINHERIT */
105 #define PCL_PREX_CLOEXEC 0x0080

137 /* Contains input and output parameters for a call to pcl_proc_exec().
138  *
139  * Output params are populated by pcl_proc_exec(). Contains information
140  * about the process pcl_proc_exec() spawned, including optional standard
141  * input, output and error redirected file handles. See PCL_PREX_xxx flags.
142  */
143 typedef struct
144 {
145  /* input parameters */
146  const tchar_t *command; /* command and arguments: "ls -la /home/foobar" */
147  const tchar_t *user; /* run as user name */
148  const tchar_t *group; /* runs as Windows LAN group (domain/workgroup) or Unix group */
149  const tchar_t *pass; /* Windows only...account password */
150  const tchar_t *cwd; /* set current working directory of new process. */
151  const tchar_t *ldpath; /* library search path to use for exec'd process, requires
152  PCL_PREX_LDPATH. For more info, see PCL_PREX_LDPATH. Set this
153  to an empty string "" to clear the search path for new
154  process and only use the system defaults. */
155 
156  /* output parameters */
157  pid_t pid; /* always set */
158  pcl_time_t start; /* always set */
159  pcl_prochandle_t handle; /* PCL_PREX_HANDLE (pcl_proc_close), NULL means not set */
160  pcl_file_t *f_stdout; /* PCL_PREX_STDOUT, NULL means not set */
161  pcl_file_t *f_stdin; /* PCL_PREX_STDERR, NULL means not set */
162  pcl_file_t *f_stderr; /* PCL_PREX_STDIN, NULL means not set */
163 } pcl_proc_exec_t;
