36 pcl_err_ctx_t *
37 pcl_err_ctx_clone(const pcl_err_ctx_t *src)
38 {
39  if(!src)
40  return NULL;
41 
42  pcl_err_ctx_t *dest = pcl_err_ctx_create();
43 
44  /* note: do not clone ctx->buffer. That is just a cache for serialize */
45  dest->err = src->err;
46  dest->oserr = src->oserr;
47 
48  /* clone src stack trace */
49  for(pcl_err_trace_t *tail = NULL, *current = src->strace; current; current = current->next)
50  {
51  pcl_err_trace_t *node = pcl_malloc(current->size);
52 
53  /* first copy entire block of memory */
54  memcpy(node, current, current->size);
55 
56  /* repoint file, func and msg */
57  size_t file_len = current->file ? strlen(current->file) + 1 : 0;
58  size_t func_len = current->func ? strlen(current->func) + 1 : 0;
59  size_t msg_len = current->msg ? strlen(current->msg) + 1 : 0;
60  char *end = (char *) (node + 1);
61 
62  node->file = file_len ? end : NULL;
63  node->func = func_len ? end + file_len : NULL;
64  node->msg = msg_len ? end + file_len + func_len : NULL;
65  node->next = NULL;
66 
67  /* add to list */
68  if(!dest->strace)
69  tail = dest->strace = node;
70  else
71  tail = tail->next = node;
72  }
73 
74  return dest;
75 }
