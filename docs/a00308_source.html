45 bool
46 pcl_readdir(pcl_dir_t *dir, pcl_dirent_t *ent, pcl_stat_t *stbuf)
47 {
48  if(!dir || !ent)
49  return R_SETERR(false, PCL_EINVAL);
50 
51  memset(ent, 0, sizeof(*ent));
52  pcl_err_osclear();
53 
54 #ifdef PCL_WINDOWS
55  /* false for the first pcl_readdir since pcl_opendir does FindFirstFileEx
56  * whose results are handled here
57  */
58  if(!dir->findnextfile)
59  {
60  /* use FindNextFile from this point forward */
61  dir->findnextfile = true;
62 
63  /* always skip "." and ".." */
64  if(pcl_pcscmp(dir->wfd.cFileName, _P(".")) && pcl_pcscmp(dir->wfd.cFileName, _P("..")))
65  goto FOUND_ENTRY;
66  }
67 
68  while(true)
69  {
70  if(FindNextFile(dir->handle, &dir->wfd))
71  {
72  /* always skip "." and ".." */
73  if(!pcl_pcscmp(dir->wfd.cFileName, _P(".")) || !pcl_pcscmp(dir->wfd.cFileName, _P("..")))
74  continue;
75 
76  /* have an entry */
77  break;
78  }
79 
80  /* skip access denied. if FindNextFile is finished, error is ERROR_NO_MORE_FILES
81  * which maps to PCL_ENOMORE.
82  */
83  if(pcl_oserrno != ERROR_ACCESS_DENIED)
84  {
85  SETLASTERR(); // success if GetLastError() == ERROR_NO_MORE_FILES
86  return false;
87  }
88  }
89 
90  FOUND_ENTRY:;
91  ent->type = PclDirentUnknown;
92  ent->namlen = (uint16_t) pcl_pcslen(ent->name);
93  pcl_pcsncpy(ent->name, countof(ent->name), dir->wfd.cFileName, ent->namlen);
94 
95  /* get path to entry */
96  pchar_t *entpath;
97  pcl_aspprintf(&entpath, _P("%Ps%/%Ps"), dir->path, ent->name);
98 
99  /* use GetFileType for fifo and chr types, need a win32 file HANDLE */
100  pcl_file_t *file = pcl_file_open(entpath, PCL_O_RDONLY);
101  pcl_free(entpath);
102 
103  if(file)
104  {
105  DWORD type = GetFileType(file->fd);
106 
107  if(type == FILE_TYPE_CHAR)
108  ent->type = PclDirentChar;
109  else if(type == FILE_TYPE_PIPE)
110  ent->type = PclDirentFifo;
111 
112  pcl_file_close(file);
113  }
114 
115  /* pcl_file_open failed or GetFileType was FILE_TYPE_DISK (disk file) or FILE_TYPE_UNKNOWN */
116  if(ent->type == PclDirentUnknown)
117  {
118  if(dir->wfd.dwFileAttributes & FILE_ATTRIBUTE_REPARSE_POINT)
119  ent->type = PclDirentLink;
120  else if(dir->wfd.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY)
121  ent->type = PclDirentDir;
122  else
123  ent->type = PclDirentFile;
124  }
125 
126 #else
127  while(true)
128  {
129 # ifdef PCL_DARWIN
130  char entbuf[4096] = {0};
131  struct dirent *e = (struct dirent *) entbuf, *result = NULL;
132  int err = readdir_r(dir->handle, e, &result);
133 # else // apparently, readdir_r is now deprecated, but not on darwin.
134  errno = 0;
135  struct dirent *e = readdir(dir->handle);
136  struct dirent *result = e;
137  int err = errno;
138 # endif
139 
140  if(err)
141  return R_SETOSERR(false, err);
142 
143  /* nothing left in directory */
144  if(!result)
145  return R_SETERR(false, PCL_ENOMORE);
146 
147  /* do not report these */
148  if(!pcl_pcscmp(e->d_name, _P(".")) || !pcl_pcscmp(e->d_name, _P("..")))
149  continue;
150 
151  /* got an entry */
152  switch(e->d_type)
153  {
154  case DT_DIR:
155  ent->type = PclDirentDir;
156  break;
157 
158  case DT_REG:
159  ent->type = PclDirentFile;
160  break;
161 
162  case DT_LNK:
163  ent->type = PclDirentLink;
164  break;
165 
166  case DT_SOCK:
167  ent->type = PclDirentSocket;
168  break;
169 
170  case DT_CHR:
171  ent->type = PclDirentChar;
172  break;
173 
174  case DT_BLK:
175  ent->type = PclDirentBlock;
176  break;
177 
178  case DT_FIFO:
179  ent->type = PclDirentFifo;
180  break;
181 
182  default:
183  ent->type = PclDirentUnknown;
184  }
185 
186  ent->namlen = (uint16_t) strlen(e->d_name);
187  pcl_pcsncpy(ent->name, countof(ent->name), e->d_name, ent->namlen);
188 
189  break;
190  }
191 #endif
192 
193  /* caller requested an lstat on entry */
194  if(stbuf)
195  {
196  pchar_t *path;
197  int r = pcl_aspprintf(&path, _P("%Ps%/%Ps"), dir->path, ent->name);
198 
199  if(r < 0)
200  return R_SETERR(false, -r);
201 
202  r = pcl_lstat(path, stbuf);
203  pcl_free(path);
204 
205  if(r)
206  return R_TRC(false);
207  }
208 
209  return true;
210 }
