37 pcl_htable_clear(pcl_htable_t *ht, bool shrink)
38 {
39  if(!ht)
40  return;
41 
42  for(int i = 0; i < ht->count_used; i++)
43  {
44  pcl_htable_entry_t *ent = &ht->entries[i];
45 
46  if(ent->key && ht->remove_entry)
47  ht->remove_entry(ent->key, ent->value);
48  }
49 
50  if(shrink && ht->capacity != MINTBLSIZE)
51  {
52  pcl_free(ht->entries);
53  ht->capacity = MINTBLSIZE;
54  ht->table_mask = ht->capacity - 1;
55  ipcl_htable_init(ht->capacity, &ht->entries, &ht->entry_lookup);
56  }
57  else
58  {
59  memset(ht->entries, 0, ht->capacity * sizeof(pcl_htable_entry_t));
60  for(int i = 0; i < ht->capacity; i++)
61  ht->entry_lookup[i] = -1;
62  }
63 
64  ht->count = ht->count_used = 0;
65 }
