37 int
38 pcl_json_objput(pcl_json_t *obj, char *key, pcl_json_t *value, uint32_t flags)
39 {
40  uint32_t freeval = flags & PCL_JSON_FREEVALONERR;
41 
42  if(!obj || !key || !value)
43  {
44  if(value && freeval)
45  pcl_json_free(value);
46 
47  return BADARG();
48  }
49 
50  if(!pcl_json_isobj(obj))
51  {
52  if(freeval)
53  pcl_json_free(value);
54 
55  return SETERRMSG(PCL_ETYPE, "expected type 'o', got '%c'", obj->type);
56  }
57 
58  if(!(flags & PCL_JSON_SKIPUTF8CHK) && pcl_utf8_check(key, 0) < 0)
59  {
60  if(freeval)
61  pcl_json_free(value);
62 
63  return TRC();
64  }
65 
66  char *k = key;
67 
68  if(!(flags & PCL_JSON_SHALLOW))
69  k = pcl_strdup(key);
70 
71  if(pcl_htable_put(obj->object, k, value, true) < 0)
72  {
73  if(k != key)
74  pcl_free(k);
75 
76  if(freeval)
77  pcl_json_free(value);
78 
79  return TRCMSG("failed to put '%s' key", key);
80  }
81 
82  return 0;
83 }
