215 /* Indicates if a given error code is a network error. Excludes EAGAIN */
216 PCL_INLINE bool
217 pcl_err_isnet(int e)
218 {
219  return (e >= PCL_ERR_NETSTART && e < PCL_ERR_NETSTOP) || e == PCL_EPIPE;
220 }
