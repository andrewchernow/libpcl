36 pcl_vector_t *
37 ipcl_net_resolve(const char *host)
38 {
39  int r;
40  pcl_vector_t *addrs;
41  struct addrinfo *info = NULL, *ai, hints = {0};
42 
43  hints.ai_socktype = SOCK_STREAM;
44  if((r = getaddrinfo(host, NULL, &hints, &info)))
45  return R_SETERR(NULL, pcl_net_ai2pcl(r));
46 
47  addrs = pcl_vector_create(3, sizeof(char **), pcl_vector_cleanup_dblptr);
48 
49  for(ai=info; ai; ai=ai->ai_next)
50  {
51  if(ai->ai_family == AF_INET || ai->ai_family == AF_INET6)
52  {
53  char ip[100];
54 
55  if(getnameinfo(ai->ai_addr, (socklen_t)ai->ai_addrlen, ip, (int)sizeof(ip),
56  NULL, 0, NI_NUMERICHOST) == 0)
57  {
58  char *a = pcl_strdup(ip);
59  pcl_vector_append(addrs, &a);
60  }
61  }
62  }
63 
64  freeaddrinfo(info);
65 
66  return addrs;
67 }
