35 pcl_json_t *
36 pcl_json_objget(const pcl_json_t *obj, const char *key)
37 {
38  if(!obj || !key)
39  return R_SETERR(NULL, PCL_EINVAL);
40 
41  if(!pcl_json_isobj(obj))
42  return R_SETERRMSG(NULL, PCL_ETYPE, "expected type 'o', got '%c'", obj->type);
43 
44  pcl_json_t *val = pcl_htable_get(obj->object, key);
45 
46  if(!val)
47  return pcl_errno ? R_TRC(NULL) : R_SETERR(NULL, PCL_ENOTFOUND);
48 
49  return val;
50 }
