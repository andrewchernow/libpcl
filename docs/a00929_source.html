57 int
58 pcl_thread_create(pthread_t *t, pcl_thread_start_t routine, void *arg)
59 {
60  pthread_t tbuf;
61 
62  if(!routine)
63  return SETERR(EINVAL);
64 
65  if(!t)
66  t = &tbuf;
67 
68  pthread_attr_t attr;
69  pthread_attr_init(&attr);
70  pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_DETACHED);
71  pthread_attr_setscope(&attr, PTHREAD_SCOPE_SYSTEM);
72 
73  /* create thread arg wrapper and start thread */
74  thread_start_t *start = pcl_malloc(sizeof(thread_start_t));
75  start->arg = arg;
76  start->routine = routine;
77  int err = pthread_create(t, &attr, thread_start_wrapper, start);
78 
79  pthread_attr_destroy(&attr);
80 
81  if(err)
82  {
83  pcl_free(start);
84  return SETOSERR(err);
85  }
86 
87  return 0;
