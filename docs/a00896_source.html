38 /* Ofcourse, Windows requires a mini program with two completely different implementations:
39  * one for the current process and another for the remote process. Uh vey. Like getting a
40  * process's image path is some kind of top secret operation.
41  */

43 static pchar_t *
44 remote_proc_path(pid_t pid)
45 {
46  pchar_t *path = NULL;
47 
48  /* ignore error. Setting this allows access to local system processes. However, if we
49  * can't enable it, we can still open many other remote processes.
50  */
51  bool enabled_priv = pcl_proc_setpriv(PRIV_PROCESS, true) == 0;
52 
53  /* Open the remote process */
54  HANDLE hproc = OpenProcess(PROCESS_QUERY_INFORMATION | PROCESS_VM_READ, false, pid);
55 
56  if(!hproc)
57  {
58  SETLASTERR();
59  goto CLEANUP;
60  }
61 
62  for(DWORD size = 64; size <= PCL_MAXLONGPATH + 1; size *= 2)
63  {
64  path = pcl_prealloc(path, size);
65 
66  DWORD len = size;
67  BOOL ok = QueryFullProcessImageName(hproc, 0, path, &len);
68 
69  if(ok)
70  break;
71 
72  if(pcl_oserrno != ERROR_INSUFFICIENT_BUFFER)
73  {
74  SETLASTERR();
75  path = pcl_free(path);
76  break;
77  }
78  }
79 
80 CLEANUP:
81  if(hproc)
82  CloseHandle(hproc);
83 
84  if(enabled_priv)
85  FREEZE_ERR(pcl_proc_setpriv(PRIV_PROCESS, false));
86 
87  return path;
88 }

90 static pchar_t *
91 current_proc_path()
92 {
93  pchar_t *path = NULL;
94 
95  for(DWORD size = 64; size <= PCL_MAXLONGPATH + 1; size *= 2)
96  {
97  path = pcl_prealloc(path, size);
98 
99  DWORD len = GetModuleFileName(NULL, path, size);
100 
101  /* Docs indicate a return of zero is a bust */
102  if(len == 0)
103  break;
104 
105  /* Docs state if function succeeds, it returns char len excluding NUL. If buf is too small,
106  * it returns size and sets ERROR_INSUFFICIENT_BUFFER. Thus, if return value is greater than
107  * zero but less than size, it worked.
108  */
109  if(len < size)
110  return path;
111 
112  /* 'len' MUST be size here, only remaining possibility. In this case, last os error
113  * should be ERROR_INSUFFICIENT_BUFFER. If not, GetModuleFileName docs are wrong.
114  * This is a sanity check and should never happen.
115  */
116  if(pcl_oserrno != ERROR_INSUFFICIENT_BUFFER)
117  break;
118  }
119 
120  pcl_free_safe(path);
121  return R_SETLASTERR(NULL);
122 }

124 pchar_t *
125 pcl_proc_path(pid_t pid)
126 {
127  pchar_t *path = pid < 0 ? current_proc_path() : remote_proc_path(pid);
128 
129  if(!path)
130  TRC();
131 
132  return path;
133 }
