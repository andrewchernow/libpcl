50 /* Extract version from registry, preferred method. */
51 static bool
52 windows_version(char *buf, int len)
53 {
54  HKEY key;
55  int n=0;
56  DWORD dwSize;
57  char product[80];
58  char version[24];
59  char build[24];
60  char service_pack[80];
61 
62  if(RegOpenKeyExA(HKEY_LOCAL_MACHINE,
63  "SOFTWARE\\Microsoft\\Windows NT\\CurrentVersion",
64  0, KEY_READ, &key) != ERROR_SUCCESS)
65  {
66  return false;
67  }
68 
69  /* $product v$version (Build $build: $service_pack)
70  * Example:
71  * Windows Server 2008 (R) Standard v6.0 (Build 6002: Service Pack 2)
72  */
73 
74  dwSize = (DWORD)sizeof(product);
75  memset(product, 0, dwSize);
76  (void)RegQueryValueExA(key, "ProductName", NULL, NULL,
77  (LPBYTE)product, &dwSize);
78 
79  dwSize = (DWORD)sizeof(version);
80  memset(version, 0, dwSize);
81  (void)RegQueryValueExA(key, "CurrentVersion", NULL, NULL,
82  (LPBYTE)version, &dwSize);
83 
84  dwSize = (DWORD)sizeof(build);
85  memset(build, 0, dwSize);
86  (void)RegQueryValueExA(key, "CurrentBuild", NULL, NULL,
87  (LPBYTE)build, &dwSize);
88 
89  /* service pack */
90  dwSize = (DWORD)sizeof(service_pack);
91  memset(service_pack, 0, dwSize);
92  (void)RegQueryValueExA(key, "CSDVersion", NULL, NULL,
93  (LPBYTE)service_pack, &dwSize);
94 
95  RegCloseKey(key);
96 
97  if(*product)
98  {
99  pcl_strcpy(buf, len, product);
100  n = (int)strlen(buf);
101  }
102 
103  if(*version)
104  {
105  if(n)
106  buf[n++] = ' ';
107 
108  buf[n++] = 'v';
109  pcl_strcpy(buf + n, len - n, version);
110  n = (int)strlen(buf);
111  }
112 
113  if(*build || *service_pack)
114  {
115  if(n)
116  buf[n++] = ' ';
117 
118  buf[n++] = '(';
119 
120  if(*build)
121  n += pcl_sprintf(buf + n, len - n, "Build %s%s",
122  build, *service_pack ? ": " : "");
123 
124  if(*service_pack)
125  n += pcl_sprintf(buf + n, len - n, "%s", service_pack);
126 
127  buf[n++] = ')';
128  buf[n] = 0;
129  }
130 
131  return true;
132 }

134 static int
135 windows_langrp_type(const pchar_t *computer_name)
136 {
137  LSA_HANDLE hPolicy = NULL;
138  LSA_OBJECT_ATTRIBUTES attrs;
139  POLICY_PRIMARY_DOMAIN_INFO *info = NULL;
140  int langrp_type = PclLanGrpStandalone;
141  POLICY_INFORMATION_CLASS infocls = PolicyPrimaryDomainInformation;
142  ACCESS_MASK mask = GENERIC_READ | POLICY_VIEW_LOCAL_INFORMATION;
143  LSA_UNICODE_STRING str;
144 
145  zero(attrs);
146 
147  str.Buffer = (PWSTR)computer_name;
148  str.Length = (USHORT)(pcl_pcslen(computer_name) * sizeof(pchar_t));
149  str.MaximumLength = str.Length + (1 * sizeof(pchar_t));
150 
151  if(!LsaOpenPolicy(&str, &attrs, mask, &hPolicy) &&
152  !LsaQueryInformationPolicy(hPolicy, infocls, &info))
153  {
154  /* If Sid is present, this computer is on a Domain */
155  if(info->Sid)
156  {
157  langrp_type = PclLanGrpDomain;
158  }
159  else
160  {
161  pchar_t domain[4096];
162 
163  pcl_pmemcpy(domain, info->Name.Buffer, info->Name.Length);
164  domain[info->Name.Length] = 0;
165 
166  /* If domain doesn't match computer name, this is a workgroup. */
167  if(pcl_pcsnicmp(domain, computer_name, pcl_pcslen(computer_name)))
168  langrp_type = PclLanGrpWorkgroup;
169  }
170  }
171 
172  if(info)
173  LsaFreeMemory((LPVOID)info);
174 
175  if(hPolicy)
176  LsaClose(hPolicy);
177 
178  return langrp_type;
179 }

181 /* Gets the langrp (workgroup/domain) name and the NetBIOS name. */
182 static int
183 windows_computer_name(pchar_t *langrp, size_t langrp_size,
184  pchar_t *name, size_t name_size, pcl_langrp_t *langrp_type)
185 {
186  LPWKSTA_INFO_100 info100 = NULL;
187  pchar_t namebuf[512];
188 
189  if(!name)
190  {
191  name = namebuf;
192  name_size = countof(namebuf);
193  *name = 0;
194  }
195 
196  if(langrp)
197  *langrp = 0;
198 
199  /* Get the Lan Group & Computer name */
200  if(NetWkstaGetInfo(NULL, 100, (LPBYTE *)&info100) == NERR_Success)
201  {
202  if(langrp)
203  pcl_pcscpy(langrp, langrp_size, info100->wki100_langroup);
204 
205  pcl_pcscpy(name, name_size, info100->wki100_computername);
206  NetApiBufferFree(info100);
207  }
208  /* Try a different method, langrp not provided if workgroup. */
209  else
210  {
211  DWORD len = (DWORD)name_size;
212 
213  if(!GetComputerNameEx(ComputerNameNetBIOS, name, &len))
214  {
215  if(GetLastError() == ERROR_MORE_DATA)
216  return SETOSERR(ERROR_MORE_DATA);
217  }
218 
219  if(langrp)
220  {
221  len = (DWORD)langrp_size;
222 
223  if(!GetComputerNameEx(ComputerNameDnsDomain, langrp, &len))
224  {
225  if(GetLastError() == ERROR_MORE_DATA)
226  return SETOSERR(ERROR_MORE_DATA);
227  }
228  }
229  }
230 
231  if(langrp_type && !strempty(name))
232  *langrp_type = windows_langrp_type(name);
233 
234  return 0;
235 }

240 static void
241 windows_cpuinfo(pcl_sysinfo_t *info)
242 {
243  HKEY key;
244  DWORD dw;
245  SYSTEM_LOGICAL_PROCESSOR_INFORMATION *lpilist = NULL;
246  DWORD size = (DWORD)sizeof(info->cpu_model);
247  LONG l = RegOpenKeyExA(HKEY_LOCAL_MACHINE, PROCNAMESTR, 0, KEY_READ, &key);
248 
249  if(l == ERROR_SUCCESS)
250  {
251  l = RegQueryValueExA(key, "ProcessorNameString", NULL, &dw,
252  (LPBYTE)info->cpu_model, &size);
253 
254  if(l == ERROR_SUCCESS && size > 1)
255  info->cpu_model[size] = 0;
256 
257  RegCloseKey(key);
258  }
259 
260  /* fallback method */
261  if(!*info->cpu_model && RegOpenKeyExA(HKEY_LOCAL_MACHINE, PROCENV,
262  0, KEY_READ, &key) == ERROR_SUCCESS)
263  {
264  size = (DWORD)sizeof(info->cpu_model);
265 
266  l = RegQueryValueExA(key, "PROCESSOR_IDENTIFIER", NULL, NULL,
267  (LPBYTE)info->cpu_model, &size);
268 
269  RegCloseKey(key);
270 
271  if(l == ERROR_SUCCESS)
272  info->cpu_model[size] = 0;
273  }
274 
275  size = 0;
276  if(!GetLogicalProcessorInformation(lpilist, &size) &&
277  GetLastError() == ERROR_INSUFFICIENT_BUFFER)
278  {
279  int cores = 0, sockets = 0;
280 
281  lpilist = pcl_malloc(size);
282 
283  if(GetLogicalProcessorInformation(lpilist, &size))
284  {
285  int i;
286  int count = (int)(size / sizeof(SYSTEM_LOGICAL_PROCESSOR_INFORMATION));
287 
288  for(i=0; i < count; i++)
289  {
290  if(lpilist[i].Relationship == RelationProcessorCore)
291  cores++;
292  else if(lpilist[i].Relationship == RelationProcessorPackage)
293  sockets++;
294  }
295  }
296 
297  pcl_free(lpilist);
298 
299  if(sockets > 0)
300  info->cpu_sockets = sockets;
301  if(cores > 0)
302  info->cpu_cores = cores;
303  }
304 }

307 ipcl_sysinfo(pcl_sysinfo_t *info, void *_uts)
308 {
309  MEMORYSTATUSEX msex;
310 
311  UNUSED(_uts);
312  pcl_strcpy(info->vendor, sizeof(info->vendor), "Microsoft");
313 
314  /* Get the LAN Group & Computer name */
315  windows_computer_name(info->langrp, countof(info->langrp),
316  info->name, countof(info->name), &info->langrp_type);
317 
318  windows_version(info->version, countof(info->version));
319  windows_cpuinfo(info);
320 
321  msex.dwLength = (DWORD)sizeof(msex);
322  if(GlobalMemoryStatusEx(&msex))
323  info->memory = (int64_t)msex.ullTotalPhys;
324 }
