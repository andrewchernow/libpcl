81 /** A PCL location is the file, func and line where a call occurred.
82  * Whenever an PCL function/macro requires these 3 values, you can do:
83  * @code
84  * debug_func(PCL_LOCATION_ARGS)
85  * debug_func(__FILE__, __func__, __LINE__) // instead of this
86  * @endcode
87  */
88 #define PCL_LOCATION_ARGS __FILE__, __func__, __LINE__

90 /** PCL_LOCATION_PARAMS expands to a function/macro declaration's parameter
91  * list: ie. const char *file, etc.... Thus, declaring:
92  * @code
93  * void debug_func(const char *file, const char *func, int line)
94  * void debug_func(PCL_LOCATION_PARAMS) // shortened to this
95  * @endcode
96  */
97 #define PCL_LOCATION_PARAMS const char *file, const char *func, int line
