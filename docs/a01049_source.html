69 /** Get the SECURITY_DESCRIPTOR for the given path.
70  *
71  * @param[in] path
72  * @param[out] sd a pointer to a security descriptor pointer to receive an allocated self-relative
73  * descriptor for the given path. Caller must free this with pcl_free.
74  * @param sdlen
75  * @param world_sid optional pointer to an allocated SID that must be passed to FreeSid.
76  * @param owner_sid optional pointer to the owner sid. This memory is owned by the security
77  * descriptor and must not passed to a free function.
78  * @param group_sid optional pointer to the group sid. This memory is owned by the security
79  * descriptor and must not passed to a free function.
80  * @param have_dacl optional pointer to a BOOL that indicates if the security descriptor
81  * has a DACL.
82  * @param dacl optional pointer to a DACL. This memory is owned by the security
83  * descriptor and must not passed to a free function.
84  * @param secinfo bit mask that indicates what security information is desired. See the WinAPI
85  * GetKernelObjectSecurity function.
86  * @return 0 for success and -1 on error. If this fails, no output arguments are valid.
87  */
88 int ipcl_win32_sd_get(const pchar_t *path, SECURITY_DESCRIPTOR **sd, DWORD *sdlen,
89  PSID *world_sid, PSID *owner_sid, PSID *group_sid, BOOL *have_dacl, PACL *dacl,
90  SECURITY_INFORMATION secinfo);

95  * @param sd pointer to a SECURITY_DESCRIPTOR. Note, this function doesn't free the descriptor
