37 void
38 pcl_json_free(pcl_json_t *j)
39 {
40  if(!j)
41  return;
42 
43  if(j == pcl_json_null() || j == pcl_json_true() || j == pcl_json_false())
44  return;
45 
46  if(pcl_json_ref(j, -1)->nrefs > 0)
47  return;
48 
49  switch(j->type)
50  {
51  case 's':
52  pcl_free_safe(j->string);
53  break;
54 
55  case 'a':
56  pcl_array_free(j->array);
57  break;
58 
59  case 'o':
60  pcl_htable_free(j->object);
61  break;
62  }
63 
64  pcl_free(j);
65 }
