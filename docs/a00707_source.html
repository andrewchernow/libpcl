45 pcl_socket_t *
46 pcl_socket(const char *host, int port)
47 {
48  pcl_socket_t *sock;
49 
50  if(strempty(host) || port < 0 || port > 65535)
51  return R_SETERR(NULL, PCL_EINVAL);
52 
53  sock = ipcl_socket_alloc();
54 
55  /* Unix domain socket, Windows added support in late 2017 (about time):
56  * See: https://devblogs.microsoft.com/commandline/af_unix-comes-to-windows/
57  */
58  if(*host == '/')
59  {
60  struct sockaddr_un *u = (struct sockaddr_un *) sock->addr;
61 
62  u->sun_family = AF_UNIX;
63 
64  if(!pcl_strcpy(u->sun_path, sizeof(u->sun_path), host))
65  {
66  TRC();
67  pcl_socket_close(sock);
68  return NULL;
69  }
70 
71  sock->sa_len = (int) (strlen(u->sun_path) + sizeof(u->sun_family));
72  sock->sa_protocol = 0;
73  sock->sa_socktype = SOCK_STREAM;
74  }
75  else
76  {
77  struct addrinfo *info = NULL, *ai, hints = {0};
78 
79  hints.ai_socktype = SOCK_STREAM;
80 
81  if(!strcmp(host, "::"))
82  hints.ai_flags = AI_PASSIVE;
83 
84  int r = getaddrinfo(host, NULL, &hints, &info);
85 
86  if(r)
87  {
88  SETERR(pcl_net_ai2pcl(r));
89  pcl_socket_close(sock);
90  return NULL;
91  }
92 
93  /* We only support AF_INET/INET6 SOCK_STREAM addresses */
94  for(ai = info; ai; ai = ai->ai_next)
95  if(ai->ai_family == AF_INET || ai->ai_family == AF_INET6)
96  break;
97 
98  if(!ai)
99  {
100  freeaddrinfo(info);
101  SETERRMSG(PCL_EPFNOSUPPORT, "Cannot find suitable IPv4 or IPv6 socket stream address", 0);
102  pcl_socket_close(sock);
103  return NULL;
104  }
105 
106  sock->sa_len = (int) ai->ai_addrlen;
107  sock->sa_protocol = ai->ai_protocol;
108  sock->sa_socktype = SOCK_STREAM;
109  memcpy(sock->addr, ai->ai_addr, ai->ai_addrlen);
110 
111  freeaddrinfo(info);
112 
113  /* set the port */
114  if(sock->addr->sa_family == AF_INET)
115  {
116  ((struct sockaddr_in *) sock->addr)->sin_port = htons((uint16_t) port);
117  }
118  else if(sock->addr->sa_family == AF_INET6)
119  {
120  ((struct sockaddr_in6 *) sock->addr)->sin6_port = htons((uint16_t) port);
121  }
122  else
123  {
124  SETERR(PCL_EAFNOSUPPORT);
125  pcl_socket_close(sock);
126  return NULL;
127  }
128  }
129 
130  sock->fd = socket(sock->addr->sa_family, sock->sa_socktype, sock->sa_protocol);
131 
132  if(sock->fd == INVALID_SOCKET)
133  {
134  SETLASTERR();
135  pcl_socket_close(sock);
136  return NULL;
137  }
138 
139  return sock;
140 }
