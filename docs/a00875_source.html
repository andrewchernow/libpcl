49 static xchar *
50 XFUNC(replace_impl)(const xchar *s, const xchar *from, const xchar *to, int casefold)
51 {
52  int to_len = (int) xstrlen(to), from_len = (int) xstrlen(from);
53 
54  /* use stack buffer so we can just steal it's data member */
55  pcl_buf_t buf;
56  pcl_buf_init(&buf, 64, XBufText);
57 
58  while(*s)
59  {
60  int n = casefold ? xstrnicmp(s, from, from_len) : xstrncmp(s, from, from_len);
61 
62  if(n == 0)
63  {
64  pcl_buf_put(&buf, to, to_len);
65  s += from_len;
66  }
67  else
68  {
69  pcl_buf_putchar(&buf, *s++);
70  }
71  }
72 
73  return (xchar *) buf.data;
74 }
