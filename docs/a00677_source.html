35 /* Table found here:
36  * https://lemire.me/blog/2018/05/09/how-quickly-can-you-check-that-a-string-is-valid-unicode-utf-8/
37  *
38  * Code Points 1st 2s 3s 4s
39  * U+0000..U+007F 00..7F
40  * U+0080..U+07FF C2..DF 80..BF
41  * U+0800..U+0FFF E0 A0..BF 80..BF
42  * U+1000..U+CFFF E1..EC 80..BF 80..BF
43  * U+D000..U+D7FF ED 80..9F 80..BF
44  * U+E000..U+FFFF EE..EF 80..BF 80..BF
45  * U+10000..U+3FFFF F0 90..BF 80..BF 80..BF
46  * U+40000..U+FFFFF F1..F3 80..BF 80..BF 80..BF
47  * U+100000..U+10FFFF F4 80..8F 80..BF 80..BF
48  */

50 int
51 ipcl_json_utf8check(const char *s, size_t len)
52 {
53  if(len == 0)
54  len = strlen(s);
55 
56  const char *end = s + len;
57 
58  while(s < end)
59  {
60  unsigned char c = (unsigned char) *s;
61 
62  /* ascii */
63  if(c <= 0x7F)
64  {
65  s++;
66  }
67  else if(c >= 0xC2 && c <= 0xDF) /* 2-byte sequence */
68  {
69  if(s + 1 >= end)
70  return SETERRMSG(PCL_EILSEQ, "2-byte sequence missing byte: first=%02hhx", c);
71 
72  unsigned char c2 = (unsigned char) *++s;
73 
74  if(c2 < 0x80 || c2 > 0xBF)
75  return SETERRMSG(PCL_EILSEQ, "2-byte sequence has invalid 2nd byte: "
76  "first=%02hhx, second=%02hhx", c, c2);
77 
78  s++;
79  }
80  else if(c >= 0xE0 && c <= 0xEF) /* 3-byte sequence */
81  {
82  if(s + 2 >= end)
83  return SETERRMSG(PCL_EILSEQ, "3-byte sequence missing bytes: first=%02hhx", c);
84 
85  unsigned char c2 = (unsigned char) *++s;
86 
87  if(c == 0xE0)
88  {
89  if(c2 < 0xA0 || c2 > 0xBF)
90  return SETERRMSG(PCL_EILSEQ, "3-byte sequence has invalid 2nd byte: "
91  "first=%02hhx, second=%02hhx", c, c2);
92  }
93  else if(c == 0xED)
94  {
95  if(c2 < 0x80 || c2 > 0x9F)
96  return SETERRMSG(PCL_EILSEQ, "3-byte sequence has invalid 2nd byte: "
97  "first=%02hhx, second=%02hhx", c, c2);
98  }
99  else if(c2 < 0x80 || c2 > 0xBF)
100  {
101  return SETERRMSG(PCL_EILSEQ, "3-byte sequence has invalid 2nd byte: "
102  "first=%02hhx, second=%02hhx", c, c2);
103  }
104 
105  unsigned char c3 = (unsigned char) *++s;
106 
107  if(c3 < 0x80 || c3 > 0xBF)
108  return SETERRMSG(PCL_EILSEQ, "3-byte sequence has invalid 3rd byte: "
109  "first=%02hhx, second=%02hhx, third=%02hhx", c, c2, c3);
110 
111  s++;
112  }
113  else if(c >= 0xF0 && c <= 0xF4) /* 4-byte sequence */
114  {
115  if(s + 3 >= end)
116  return SETERRMSG(PCL_EILSEQ, "4-byte sequence missing bytes: first=%02hhx", c);
117 
118  unsigned char c2 = (unsigned char) *++s;
119 
120  if(c == 0xF0)
121  {
122  if(c2 < 0x90 || c2 > 0xBF)
123  return SETERRMSG(PCL_EILSEQ, "4-byte sequence has invalid 2nd byte: "
124  "first=%02hhx, second=%02hhx", c, c2);
125  }
126  else if(c2 < 0x80 || c > 0xBF)
127  {
128  return SETERRMSG(PCL_EILSEQ, "4-byte sequence has invalid 2nd byte: "
129  "first=%02hhx, second=%02hhx", c, c2);
130  }
131 
132  unsigned char c3 = (unsigned char) *++s;
133 
134  if(c3 < 0x80 || c3 > 0xBF)
135  return SETERRMSG(PCL_EILSEQ, "4-byte sequence has invalid 3rd byte: "
136  "first=%02hhx, second=%02hhx, third=%02hhx", c, c2, c3);
137 
138  unsigned char c4 = (unsigned char) *++s;
139 
140  if(c4 < 0x80 || c4 > 0xBF)
141  return SETERRMSG(PCL_EILSEQ, "4-byte sequence has invalid 4th byte: first=%02hhx, "
142  "second=%02hhx, third=%02hhx, fourth=%02hhx", c, c2, c3, c4);
143  s++;
144  }
145  else
146  {
147  return SETERRMSG(PCL_EILSEQ, "invalid first byte: %02hxx", c);
148  }
149  }
150 
151  return 0;
152 }
