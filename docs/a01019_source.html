36 ipcl_ssl_configure_socket(pcl_ssl_t *ssl, const char *host, int port)
37 {
38  int n = 65535;
39 
40  if(host)
41  {
42  ssl->sock = pcl_socket(host, port);
43  if(!ssl->sock)
44  return TRCMSG("Failed to create socket: host=%s port=%d", host, port);
45  }
46 
47  if(ssl->flags & PCL_SOCK_NONBLOCK)
48  pcl_socket_setnonblocking(ssl->sock, true);
49 
50  pcl_socketfd_t fd = pcl_socket_fd(ssl->sock);
51 
52  (void) setsockopt(fd, SOL_SOCKET, SO_RCVBUF, (const char *) &n, sizeof(int));
53  (void) setsockopt(fd, SOL_SOCKET, SO_SNDBUF, (const char *) &n, sizeof(int));
54 
55  if(ssl->flags & PCL_SSL_PASSIVE)
56  {
57  n = 1;
58  (void) setsockopt(fd, SOL_SOCKET, SO_REUSEADDR, (const char *) &n, sizeof(int));
59  }
60 
61  return 0;
62 }
