71 int
72 pcl_proc_exec(pcl_proc_exec_t *exec, int flags)
73 {
74  int err;
75  int child_stdin_fd = -1, child_stdout_fd = -1, child_stderr_fd = -1;
76 
77  exec->pid = 0;
78  exec->handle = 0;
79  exec->start.sec = 0;
80  exec->start.nsec = 0;
81  exec->f_stderr = exec->f_stdin = exec->f_stdout = NULL;
82 
83  if(strempty(exec->command))
84  return BADARG();
85 
86  fflush(stdout);
87  fflush(stderr);
88 
89  pchar_t **argv = NULL;
90  int argc = pcl_proc_parsecmd(pcl_pcsskipws(exec->command), &argv);
91 
92  if(argc == 0)
93  {
94  pcl_proc_freeargv(argc, argv);
95  return SETERRMSG(PCL_EBADCMD, "command is empty", 0);
96  }
97 
98  if(argc < 0)
99  return TRCMSG("error parsing command: '%Ps'", exec->command);
100 
101  /* use SHELL to execute command */
102  if(flags & PCL_EXEC_SHELL)
103  {
104  pchar_t *shell = getenv("SHELL");
105 
106  if(!shell)
107  shell = "/bin/sh";
108 
109  int tmp_argc = 1;
110  pchar_t *tmp_argv[5] = {shell, NULL, NULL, NULL, NULL};
111 
112  /* set -m for tcsh/csh: loads ~/.tcshrc */
113  if(pcl_stristr(shell, "/tcsh") || pcl_stristr(shell, "/csh"))
114  tmp_argv[tmp_argc++] = "-m";
115  /* note: fish, dash, sh and bash will all set -l, login shell */
116  else if(!pcl_stristr(shell, "/zsh"))
117  tmp_argv[tmp_argc++] = "-l";
118 
119  /* Man Page: "commands are read from this argument" ... (must be a single
120  * arg, at shell this means quoted). Ex: `bash -l -c "printf '%s\n' 'Hello'"`,
121  * is how pcl_proc_exec() would send this to the shell for builtin `printf`.
122  */
123  tmp_argv[tmp_argc++] = "-c";
124  tmp_argv[tmp_argc++] = (pchar_t *) exec->command;
125 
126  /* free old argv */
127  pcl_proc_freeargv(argc, argv);
128 
129  /* allocate a new argv */
130  argc = tmp_argc;
131  argv = (pchar_t **) pcl_malloc((argc + 1) * sizeof(pchar_t *));
132 
133  /* allocate each argv element */
134  for(int i = 0; i < argc; i++)
135  argv[i] = pcl_pcsdup(tmp_argv[i]);
136 
137  /* must NULL terminate for execvp */
138  argv[argc] = NULL;
139  }
140 
141  pcl_file_t *pipes[2];
142 
143  if(flags & (PCL_EXEC_STDIN | PCL_EXEC_STDIN_NB))
144  {
145  int nonblock = flags & PCL_EXEC_STDIN_NB ? PCL_O_NONBLOCK : 0;
146 
147  if(pcl_proc_pipe(pipes, 0, nonblock | PCL_O_CLOEXEC))
148  {
149  CLOSE_ALL;
150  pcl_proc_freeargv(argc, argv);
151  return SETLASTERR();
152  }
153 
154  exec->f_stdin = pipes[1];
155  child_stdin_fd = pipes[0]->fd;
156  pipes[0]->fd = PCL_BADFD;
157  pcl_file_close(pipes[0]);
158  }
159 
160  if(flags & (PCL_EXEC_STDOUT | PCL_EXEC_STDOUT_NB))
161  {
162  int nonblock = flags & PCL_EXEC_STDOUT_NB ? PCL_O_NONBLOCK : 0;
163 
164  if(pcl_proc_pipe(pipes, nonblock | PCL_O_CLOEXEC, 0))
165  {
166  CLOSE_ALL;
167  pcl_proc_freeargv(argc, argv);
168  return SETLASTERR();
169  }
170 
171  exec->f_stdout = pipes[0];
172  child_stdout_fd = pipes[1]->fd;
173  pipes[1]->fd = PCL_BADFD;
174  pcl_file_close(pipes[1]);
175  }
176 
177  if(flags & (PCL_EXEC_STDERR | PCL_EXEC_STDERR_NB))
178  {
179  int nonblock = flags & PCL_EXEC_STDERR_NB ? PCL_O_NONBLOCK : 0;
180 
181  if(pcl_proc_pipe(pipes, nonblock | PCL_O_CLOEXEC, 0))
182  {
183  CLOSE_ALL;
184  pcl_proc_freeargv(argc, argv);
185  return SETLASTERR();
186  }
187 
188  exec->f_stderr = pipes[0];
189  child_stderr_fd = pipes[1]->fd;
190  pipes[1]->fd = PCL_BADFD;
191  pcl_file_close(pipes[1]);
192  }
193 
194  /* additional set of pipes so child process can write an err to parent if execvp fails */
195  int err_fds[2];
196 
197  if(pipe(err_fds))
198  {
199  CLOSE_ALL;
200  pcl_proc_freeargv(argc, argv);
201  return SETLASTERR();
202  }
203 
204  exec->pid = fork(); /* Ahhh, finally we can fork */
205 
206  /* failed to fork process */
207  if(exec->pid < 0)
208  {
209  FREEZE_ERR
210  (
211  close(err_fds[0]);
212  close(err_fds[1]);
213  CLOSE_ALL;
214  pcl_proc_freeargv(argc, argv);
215  );
216 
217  return SETLASTERR();
218  }
219 
220  /* parent */
221  if(exec->pid > 0)
222  {
223  err = 0;
224 
225  /* close all child_xxx FDs, only for child */
226  CLOSE_FDS;
227 
228  /* close write end of err FDs, only child uses this */
229  close(err_fds[1]);
230 
231  /* parent doesn't need this at all, it is for child execvp. Avoid parent leak. */
232  pcl_proc_freeargv(argc, argv);
233 
234  /* read err code written when child execvp fails. Otherwise, this will return zero
235  * (EOF/broken-pipe) since child sets FD_CLOEXEC.
236  */
237  if(read(err_fds[0], &err, sizeof(err)) == (ssize_t) sizeof(err))
238  CLOSE_FILES; // child failed since this read an err code, cleanup
239  else
240  exec->start = pcl_time(); /* real start time, not when forked */
241 
242  close(err_fds[0]);
243 
244  if(err)
245  return SETERRMSG(err, "Failed to execute '%Ps'", exec->command);
246 
247  if(flags & PCL_EXEC_HANDLE)
248  exec->handle = exec->pid;
249 
250  return 0;
251  }
252 
253  /* CHILD .............. */
254 
255  /* parent using this, child doesn't need it */
256  close(err_fds[0]);
257 
258  /* If execvp is successful, err_fds[1] should be automatically closed, causing parent's read
259  * to return 0 (broken pipe)
260  */
261  (void) fcntl(err_fds[1], F_SETFD, fcntl(err_fds[1], F_GETFD) | FD_CLOEXEC);
262 
263  if(child_stdin_fd != -1)
264  {
265  pcl_file_close(exec->f_stdin); /* for parent only */
266  dup2(child_stdin_fd, STDIN_FILENO);
267  close(child_stdin_fd);
268  }
269 
270  if(child_stdout_fd != -1)
271  {
272  pcl_file_close(exec->f_stdout); /* for parent only */
273  dup2(child_stdout_fd, STDOUT_FILENO);
274  close(child_stdout_fd);
275  }
276 
277  if(child_stderr_fd != -1)
278  {
279  pcl_file_close(exec->f_stderr); /* for parent only */
280  dup2(child_stderr_fd, STDERR_FILENO);
281  close(child_stderr_fd);
282  }
283 
284  /* The below code impersonates the user and group requested by callee.
285  * Not the effective user/group, the real ones.
286  */
287  uid_t uid = -1;
288 
289  /* If requested username, ensure the group list is correct */
290  if(exec->user)
291  {
292  if(pcl_getuid(exec->user, &uid))
293  {
294  TRC();
295  pcl_err_fprintf(stderr, 0, "Invalid username for command: "
296  "'%s', user=%s", argv[0], exec->user);
297  _exit(1);
298  }
299 
300  if(initgroups(exec->user, (gid_t) uid))
301  {
302  TRC();
303  pcl_err_fprintf(stderr, 0, "Cannot init group access list for command: "
304  "'%s', user=%s", argv[0], exec->user);
305  _exit(1);
306  }
307  }
308 
309  if(exec->group)
310  {
311  gid_t gid;
312 
313  if(pcl_getgid(exec->group, &gid))
314  {
315  TRC();
316  pcl_err_fprintf(stderr, 0, "Invalid group for command: "
317  "'%s', group=%s", argv[0], exec->group);
318  _exit(1);
319  }
320 
321  if(setgid(gid))
322  {
323  TRC();
324  pcl_err_fprintf(stderr, 0, "Invalid group for command: "
325  "'%s', group=%s", argv[0], exec->group);
326  _exit(1);
327  }
328  }
329 
330  if(uid != (uid_t) -1 && setuid(uid))
331  {
332  TRC();
333  pcl_err_fprintf(stderr, 0, "Invalid username for command: "
334  "'%s', user=%s", argv[0], exec->user);
335  _exit(1);
336  }
337 
338  err = PCL_EOKAY;
339 
340  /* change working directory if asked to */
341  if(!strempty(exec->cwd))
342  {
343  if(pcl_chdir(exec->cwd) < 0)
344  err = pcl_errno;
345  }
346 
347  /* Set the LDPATH env variable: like LD_LIBRARY_PATH for example. */
348  if(err == PCL_EOKAY && (flags & PCL_EXEC_LDPATH))
349  {
350  const char *ldpath = exec->ldpath;
351  char cwd[PCL_MAXPATH];
352 
353  /* try fallbacks, the exec->cwd fallback just helps us avoid a call
354  * to getcwd when we already have the cwd.
355  */
356  if(!ldpath)
357  {
358  ldpath = exec->cwd;
359  if(!ldpath)
360  ldpath = getcwd(cwd, countof(cwd));
361  }
362 
363  /* If we don't get an ldpath, ignore the PCL_EXEC_LDPATH request */
364  if(ldpath && setenv(LIBPATH, ldpath, 1))
365  err = pcl_err_os2pcl(errno);
366  }
367 
368  /* NOTE: execvp will search system PATH if command doesn't contain a '/'.
369  * Thus, callee should avoid paths with '/' for a shell PATH search.
370  */
371  if(err == PCL_EOKAY && execvp(argv[0], (char *const *) argv))
372  err = pcl_err_os2pcl(errno);
373 
374  /* NEVER GETS HERE IF `execvp` SUCCEEDED !!!
375  *
376  * tell parent there is an error, its waiting on read end of pipe. If execvp succeeded,
377  * err_fds[1] has already been closed via FD_CLOEXEC
378  */
379  (void) write(err_fds[1], &err, sizeof(err));
380 
381  close(err_fds[1]);
382  _exit(1);
383 }
