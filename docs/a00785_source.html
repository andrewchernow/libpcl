35 ipcl_json_parse_array(ipcl_json_state_t *s)
36 {
37  if(*s->next++ != '[')
38  JSON_THROW("expected opening array '['", 0);
39 
40  ipcl_json_skipws(s);
41  s->ctx = s->next;
42 
43  pcl_json_t *arr = pcl_json_arr();
44 
45  /* empty array */
46  if(*s->next == ']')
47  {
48  s->next++;
49  return arr;
50  }
51 
52  do
53  {
54  pcl_json_t *val = ipcl_json_parse_value(s);
55 
56  if(!val)
57  {
58  pcl_json_free(arr);
59  return NULL;
60  }
61 
62  if(pcl_json_arradd(arr, val, PCL_JSON_FREEVALONERR) < 0)
63  {
64  pcl_json_free(arr);
65  return R_TRCMSG(NULL, "failed to add value to array", 0);
66  }
67 
68  if(!ipcl_json_skipws(s))
69  {
70  pcl_json_free(arr);
71  return NULL;
72  }
73 
74  if(*s->next == ']')
75  {
76  s->next++;
77  return arr;
78  }
79 
80  if(*s->next++ != ',')
81  {
82  pcl_json_free(arr);
83  JSON_THROW("missing comma after value", 0);
84  }
85 
86  if(!ipcl_json_skipws(s))
87  {
88  pcl_json_free(arr);
89  return NULL;
90  }
91  }
92  while(s->next < s->end);
93 
94  pcl_json_free(arr);
95  JSON_THROW("unexpected end of input", 0);
96 }
