39 pcl_err_vjson(const char *message, va_list ap)
40 {
41  pcl_err_t *err = pcl_err_get();
42  pcl_json_t *root = pcl_json_object();
43  uint32_t flags = PCL_JSON_SKIPUTF8CHK | PCL_JSON_FREEVALONERR;
44  uint32_t strflags = PCL_JSON_EMPTYASNULL | PCL_JSON_ALLOWNULL;
45 
46  if(pcl_json_object_put(root, "err", pcl_json_integer(err->err), flags) < 0)
47  {
48  pcl_json_free(root);
49  return R_TRC(NULL);
50  }
51 
52  char *name = (char *) (err->err ? pcl_err_name(err->err) : "");
53 
54  if(pcl_json_object_put(root, "name", pcl_json_string(name, strflags), flags) < 0)
55  {
56  pcl_json_free(root);
57  return R_TRC(NULL);
58  }
59 
60  if(pcl_json_object_put(root, "oserr", pcl_json_integer(err->oserr), flags) < 0)
61  {
62  pcl_json_free(root);
63  return R_TRC(NULL);
64  }
65 
66  name = (char *) (err->oserr ? pcl_err_osname(err->oserr) : "");
67 
68  if(pcl_json_object_put(root, "osname", pcl_json_string(name, strflags), flags) < 0)
69  {
70  pcl_json_free(root);
71  return R_TRC(NULL);
72  }
73 
74  char *msg = NULL;
75 
76  if(!strempty(message))
77  pcl_vasprintf(&msg, message, ap);
78 
79  if(pcl_json_object_put(root, "msg", pcl_json_string(msg, strflags | PCL_JSON_SHALLOW), flags) < 0)
80  {
81  pcl_json_free(root); // 'msg' managed by string & freed due to PCL_JSON_FREEVALONERR
82  return R_TRC(NULL);
83  }
84 
85  pcl_json_t *strace = pcl_json_array();
86 
87  if(pcl_json_object_put(root, "strace", strace, flags) < 0)
88  {
89  pcl_json_free(root);
90  return R_TRC(NULL);
91  }
92 
93  for(pcl_err_trace_t *t = err->strace; t; t = t->next)
94  {
95  pcl_json_t *jtrace = pcl_json_object();
96 
97  if(pcl_json_object_put(jtrace, "file", pcl_json_string(t->file, strflags), flags) < 0)
98  {
99  pcl_json_free(root);
100  return R_TRC(NULL);
101  }
102 
103  if(pcl_json_object_put(jtrace, "func", pcl_json_string(t->func, strflags), flags) < 0)
104  {
105  pcl_json_free(root);
106  return R_TRC(NULL);
107  }
108 
109  if(pcl_json_object_put(jtrace, "line", pcl_json_integer(t->line), flags) < 0)
110  {
111  pcl_json_free(root);
112  return R_TRC(NULL);
113  }
114 
115  if(pcl_json_object_put(jtrace, "msg", pcl_json_string(t->msg, strflags), flags) < 0)
116  {
117  pcl_json_free(root);
118  return R_TRC(NULL);
119  }
120 
121  if(pcl_json_array_add(strace, jtrace, flags) < 0)
122  {
123  pcl_json_free(root);
124  return R_TRC(NULL);
125  }
126  }
127 
128  return root;
129 }
