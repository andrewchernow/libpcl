37 ipcl_file_open(pcl_file_t *file, const pchar_t *path, int pcl_oflags, mode_t mode)
38 {
39  if(pcl_oflags & PCL_O_DIRECTORY)
40  {
41 #ifndef O_DIRECTORY
42  sys_stat_t st;
43 
44  if(stat(path, &st))
45  return SETLASTERR();
46 
47  if(!S_ISDIR(st.st_mode))
48  return SETERR(PCL_ENOTDIR);
49 #endif
50  }
51 
52  if(pcl_oflags & PCL_O_NOFOLLOW)
53  {
54 #ifndef O_NOFOLLOW
55  sys_stat_t st;
56 
57  if(!sys_lstat(path, &st) && S_ISLNK(st.st_mode))
58  return SETERRMSG(PCL_ETYPE, "%Ps is a symbolic link", path);
59 #endif
60  }
61 
62  if((file->fd = open(path, ipcl_sysoflags(pcl_oflags), mode)) == -1)
63  return SETLASTERR();
64 
65  /* this should overwrite SHLOCK */
66  if(pcl_oflags & PCL_O_EXLOCK)
67  {
68 #ifdef O_EXLOCK
69  file->flags |= PCL_FF_OLOCK; /* obtained lock through open() ... "O"LOCK */
70 #else
71  /* need to manually lock the file */
72  if(pcl_file_lock(file, PCL_WRLOCK))
73  return TRCMSG("file open cannot aquire lock on '%Ps'", path);
74 
75  file->flags |= PCL_FF_FLOCK; /* obtained lock through flock */
76 #endif
77  }
78  else if(pcl_oflags & PCL_O_SHLOCK)
79  {
80 #ifdef O_SHLOCK
81  file->flags |= PCL_FF_OLOCK; /* obtained lock through open() ... "O"LOCK */
82 #else
83  /* need to manually lock the file */
84  if(pcl_file_lock(file, PCL_RDLOCK))
85  return TRCMSG("file open cannot aquire lock on '%Ps'", path);
86 
87  file->flags |= PCL_FF_FLOCK; /* obtained lock through flock */
88 #endif
89  }
90 
91  if(pcl_oflags & PCL_O_TEMP)
92  (void) pcl_unlink(path);
93 
94  return 0;
95 }
