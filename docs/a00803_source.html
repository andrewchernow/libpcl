35 int
36 pcl_ssl_setpeerca(pcl_ssl_t *ssl, const char *path, int peer_depth)
37 {
38  if(!ssl || !ssl->ctx || strempty(path))
39  return SETERR(PCL_EINVAL);
40 
41  ERR_clear_error();
42 
43  /* To verify the client, server must have a list of CA "names", not
44  * complete certificates, to send to client as a certificate request.
45  *
46  * Look at NOTES section here:
47  *
48  * http://www.openssl.org/docs/ssl/SSL_CTX_set_client_CA_list.html
49  *
50  * Also peek at SSL_CTX_load_verify_locations (NOTES section):
51  *
52  * http://www.openssl.org/docs/ssl/SSL_CTX_load_verify_locations.html
53  *
54  * which states "In server mode, when requesting a client certificate,
55  * the server must send the list of CAs of which it will accept client
56  * certificates. This list is not influenced by the contents of CAfile
57  * or CApath and must explicitly be set using the
58  * SSL_CTX_set_client_CA_list(3) family of functions."
59  */
60  if(ssl->flags & (PCL_SSL_PASSIVE | PCL_SSL_SERVER))
61  {
62  STACK_OF(X509_NAME) *cert_names = SSL_load_client_CA_file(path);
63 
64  if(!cert_names && sk_X509_NAME_num(cert_names) == 0)
65  return SETERRMSG(PCL_ENOTFOUND, "No CA certs found at '%s', %s",
66  path, sslerror(ssl));
67 
68  SSL_CTX_set_client_CA_list(ssl->ctx, cert_names);
69  }
70 
71  /* Used to verify peer certificate. Without this on the server, the
72  * client will send its certificate but the server won't verify it.
73  * Above CA_list() stuff only requests the certificate from a client.
74  * Client also needs this call to verify server certificate, however
75  * client doesn't need to load a CA list of names like server.
76  */
77  if(SSL_CTX_load_verify_locations(ssl->ctx, path, NULL) == 0)
78  return SETERRMSG(PCL_ESSL, "Failed to load peer CA certificate: %s, %s",
79  path, sslerror(ssl));
80 
81  /* Must enable verify_peer on server-side */
82  int server_flags = SSL_VERIFY_PEER | SSL_VERIFY_FAIL_IF_NO_PEER_CERT;
83 
84  if(ssl->flags & (PCL_SSL_PASSIVE | PCL_SSL_SERVER))
85  SSL_CTX_set_verify(ssl->ctx, server_flags, NULL);
86  /* clients use verify_none, this flag ensures we catch verify errors. */
87  else
88  ssl->flags |= PCL_SSL_VERIFY_PEER;
89 
90  SSL_CTX_set_verify_depth(ssl->ctx, max(0,peer_depth));
91 
92  return 0;
93 }
