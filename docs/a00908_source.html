58  if(!info)

63  /* timezone data */
64  if(pcl_localtime(NULL, &tm) == 0)
65  {

67  pcl_strcpy(info->tzabbr, sizeof(info->tzabbr), tm.tm_zone);

69  else

72  }
73 
74  /* Windows will overwrite the computer name 'info->name' with the

76  */
77  if(!gethostname(info->hostname, sizeof(info->hostname)))
78  {
79  if((p = strchr(info->hostname, '.')))
80  *p = 0;
81 
82  tchar_t *name = pcl_utf8_to_tcs(info->hostname, 0, NULL);
83  pcl_tcscpy(info->name, countof(info->name), name);
84  pcl_free(name);
85 
86  if(p)
87  *p = '.';
88  }
89 
90  ipcl_sysinfo(info, &uts);
91 
92  /* remove any double spaces, this is common (straight from DMI) */
93  while((p = strstr(info->cpu_model, " ")))
