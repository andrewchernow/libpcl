36 pcl_vector_remove(pcl_vector_t *v, int pos)
37 {
38  int success = (v && pos >= 0 && pos < v->count) ? 0 : -1;
39 
40  if(success == 0)
41  {
42  if(v->cleanup)
43  {
44  void *e = pcl_vector_get(v, pos);
45 
46  if(e)
47  v->cleanup(e);
48  }
49 
50  if(pos < v->count - 1) /* memmove only when NOT last elem */
51  (void) memmove(pcl_vector_get(v, pos), pcl_vector_get(v, pos + 1),
52  (v->count - pos - 1) * v->size);
53 
54  v->count--;
55  }
56 
57  return success;
58 }
