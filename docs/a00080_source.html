83 /** Indicates that the command to exec should be passed to the SHELL. On unix, this uses
84  * `$SHELL -c "your_command"` and on windows, this uses `%COMSPEC% /C "your_command`.
85  *
86  * This is only required for internal (builtin) commands. These are commands
87  * that are built into windows CMD.EXE or the Unix shell: like BASH `cd`.
88  * Thus, you must do `cmd.exe /C "command"` or `sh -c "command"` to exec them. This
89  * will work for non-built-in commands but incurs an unnecessary shell.
90  */
91 #define PCL_EXEC_SHELL 0x0040

101 /** Sets the library search path within the new process.
102  * The library path is set to the value of pcl_proc_exec_t.ldpath; a colon separated list of
103  * paths. On windows colons are replaced for ';' so either can be used.
104  *
105  * On Unix, if 'ldpath' is \c NULL, exec tries pcl_proc_exec_t.cwd and then \c getcwd. If none of
106  * these work, the flag is ignored.
107  *
108  * On windows, if pcl_proc_exec_t.ldpath is \c NULL, pcl_proc_exec ignores this flag (continues)
109  * and there are no fallback attempts like on Unix. This is because on windows, the current
110  * working directory is already part of the library search path.
111  *
112  * pcl_proc_exec_t.ldpath must be \c NULL for exec to try another one. An empty string is
113  * completely valid and can be used to clear the library search path for this new process; but
114  * not effect the system-wide search paths.
115  *
116  * * Windows: added to PATH => https://docs.microsoft.com/en-us/cpp/build/search-path-used-by-windows-to-locate-a-dll
117  * * Linux: added to LD_LIBRARY_PATH
118  * * Darwin: added to DYLD_LIBRARY_PATH
119  */
120 #define PCL_EXEC_LDPATH 0x0100

223 /** Creates a pipe pair of pcl_file_t objects using the given read and write
224  * open flags (PCL_O_xxx). Note that unlike the unix `pipe2` function, this
225  * version allows you to set different flags for both the read and write
226  * ends of the pipe. pipes[0]=read and pipes[1]=write. Again, the rd_oflags and
227  * wr_oflags are PCL_O_xxx open flags, not POSIX O_xxx flags. PCL_O_RDONLY is
228  * automatically added to the rd_oflags and PCL_O_WRONLY to wr_oflags. PCL_O_RDWR is
229  * removed if present.
230  * @param pipes a two element array of ::pcl_file_t pointers. Element 0 is set to the read
231  * end of the pipe and element 1 to the write end.
232  * @param rd_oflags a bitmask of PCL_O_xxx flags for the read end of the pipe
233  * @param wr_oflags a bitmask of PCL_O_xxx flags for the write end of the pipe
234  * @return 0 on success and -1 on error
235  */
236 PCL_EXPORT int pcl_proc_pipe(pcl_file_t *pipes[2], int rd_oflags, int wr_oflags);
