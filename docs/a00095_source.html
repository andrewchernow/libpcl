45 /** Convert a character string to an integer.
46 * The integer conversion functions do range checks in addition to the initial conversion. If
47 * the converted value is out of range, -1 is returned and \c PCL_ERANGE is set.
48 *
49 * @param nptr pointer to a character string
50 * @param end a pointer that if provided, will point to the first invalid character
51 * @param base integer base between 2 and 36 inclusive or special value of 0. When 0,
52 * if \a nptr begins with '0x', base 16 is used. If \a nptr begins with '0', octal is used.
53 * Otherwise, base 10 is used.
54 * @param r pointer to the result of the conversion
55 * @return 0 on success and -1 on error.
56 */
57 PCL_EXPORT int pcl_strtoi(const char *nptr, char **end, int base, int *r);
