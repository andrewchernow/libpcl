35 /** @defgroup buf Dynamic Buffer
36  * The buffer API provides a dynamically sized buffer for putting text and/or binary data.
37  * When a buffer is created, a buffer mode must be provided: one of ::PclBufBinary, ::PclBufText,
38  * ::PclBufTextW or ::PclBufTextP. Regardless of a buffer's mode, a trailing NUL is added on
39  * every put. The space is reserved internally and it not represented in the buffer's
40  * current position or length. The buffer thinks in "characters", not bytes. When in binary
41  * mode, it still performs character calculations but based on a one byte character.
42  *
43  * ### Binary Mode
44  * A binary buffer is an array of data types: integers, strings and blobs. A binary string is
45  * written to the buffer with a trailing NUL, avoiding the need for a length prefix.
46  * Integers are written in network-byte order and blobs are written as is. For blobs, if they
47  * are variable length, do ::pcl_buf_putint32 followed by ::pcl_buf_put. There is also
48  * int8, int16 and int64.
49  *
50  * All buffer functions are available to ::PclBufBinary. Again, this mode deals with bytes but
51  * treats them as one byte characters.
52  *
53  * @code
54  * pcl_buf_t *buf = pcl_buf_init(NULL, 64, PclBufBinary);
55  *
56  * pcl_buf_putstr(buf, "Hello"); // buffer contents: "Hello\0\0"
57  *
58  * // no trailing NUL for single characters
59  * pcl_buf_putchar(buf, ' '); // buffer contents: "Hello\0 \0"
60  *
61  * // it may appear that the space is joined to World. However, when getting data back
62  * // out of the buffer, the idea is to call the same set of functions to deserialize.
63  * // Thus, pcl_buf_getchar gets the space and pcl_buf_getstr gets "World"
64  * pcl_buf_putstr(buf, "World"); // buffer contents: "Hello\0 World\0\0"
65  *
66  * // be careful with the raw put function, it won't add a NUL for strings
67  * // unless you include the NUL in the length argument.
68  * pcl_buf_put(buf, "abc", 3); // No trailing NUL put. Pass 4 or use pcl_buf_putstr
69  *
70  * // binary mode only, "Hello\0 World\0abc[32 bits in network-byte order]\0"
71  * // note the "abc" string without a NUL. pcl_buf_getstr cannot be used, one
72  * // must use pcl_buf_getchar 3 times or pcl_buf_get with a length of 3.
73  * pcl_buf_putint32(buf, 3897463623);
74  *
75  * pcl_buf_free(buf);
76  * @endcode
77  *
78  * ### Text Mode
79  * In text mode, the buffer is an array of characters with the always present NUL at the end.
80  * Text mode buffers are limited to put, putstr, putchar, putf and putvf. Get functions
81  * aren't needed in text mode since one can simply access pcl_buf_t.data, which will be a
82  * NUL-terminated string. Directly accessing pcl_buf_t.data, means you must properly cast it if
83  * using ::PclBufTextW or ::PclBufTextP.
84  *
85  * @code
86  * pcl_buf_t buf; // use stack memory
87  *
88  * pcl_buf_init(&buf, 64, PclBufText);
89  * pcl_buf_putstr(&buf, "Hello"); // buffer contents: "Hello\0"
90  * pcl_buf_putstr(&buf, "World"); // buffer contains: "HelloWorld\0"
91  * printf("%d] %s\n", buf.len, buf.data); // 10] HelloWorld
92  * pcl_buf_clear(&buf); // not pcl_buf_free!
93  *
94  * // For PclBufTextW (PclBufTextP is identical but use _P("") and pchar_t)
95  * pcl_buf_init(&buf, 64, PclBufTextW);
96  * pcl_buf_putstr(&buf, L"Hello");
97  * printf("%ls\n", (wchar_t *)buf.data);
98  * @endcode
99  * @{
100  */

108 enum pcl_buf_mode
109 {
110  /** Binary buffer mode. Single byte characters and all put and get functions are available. */
111  PclBufBinary,
112 
113  /** Text buffer mode. This is a \c char buffer with limited access to functions. */
114  PclBufText,
115 
116  /** Wide-text buffer mode. This is a \c wchar_t buffer with limited acccess to functions. */
117  PclBufTextW,
118 
119 #ifdef PCL_WINDOWS
120  PclBufTextP=PclBufTextW
121 #else
122  /** portable-text buffer mode. This is a \c pchar_t buffer with limited acccess to functions. */
123  PclBufTextP=PclBufText
124 #endif
125 };
