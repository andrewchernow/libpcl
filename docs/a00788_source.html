40  if(!want_io)
41  want_io = &x;

43 
44  if(!(ssl->flags & PCL_SSL_SERVER))
45  return SETERR(PCL_EINVAL);
46 
47  ERR_clear_error();
48  switch(SSL_get_error(ssl->ssl, SSL_accept(ssl->ssl)))
49  {
50  case SSL_ERROR_NONE:
51  break;
52 
53  case SSL_ERROR_ZERO_RETURN:
54  SETERRMSG(PCL_ECONNRESET, "%s", sslerror(ssl));
55  pcl_ssl_close(ssl);
56  return -1;
57 
58  case SSL_ERROR_WANT_READ:
59  *want_io = PCL_SSL_WANTREAD;
60  return SETERR(PCL_EAGAIN);
61 
62  case SSL_ERROR_WANT_WRITE:
63  *want_io = PCL_SSL_WANTWRITE;
64  return SETERR(PCL_EAGAIN);
65 
66  case SSL_ERROR_SYSCALL:
67  {
68  const char *e = sslerror(ssl);
69 
70  if(e)
71  SETLASTERRMSG("%s", e);
72  else
73  SETLASTERR();
74 
75  pcl_ssl_close(ssl);
76  return -1;
77  }
78 
79  default:

82

84  SETERRMSG(PCL_ESSL, "%s", e);
85  else
86  SETERR(PCL_ESSL);
87 
88  pcl_ssl_close(ssl);
89  return -1;

91  }
92
