36 pcl_connect(pcl_socket_t *sock)
37 {
38  int r = 0, err;
39 
40  /* already connected */
41  if(sock->flags & PCL_SOCK_CONNECTED)
42  return 0;
43 
44  if(sock->fd == INVALID_SOCKET)
45  return SETERR(PCL_EINVAL);
46 
47  /* connect succeeded */
48  if(!connect(sock->fd, sock->addr, sock->sa_len))
49  {
50  sock->flags |= PCL_SOCK_CONNECTED;
51  return 0;
52  }
53 
54  err = pcl_err_os2pcl(pcl_sockerrno);
55 
56  switch(err)
57  {
58  case PCL_EISCONN: /* already connected */
59  sock->flags |= PCL_SOCK_CONNECTED;
60  break;
61 
62  case PCL_EALREADY: /* previous connect not yet finished */
63  case PCL_EAGAIN: /* same as EINPROGRESS, PCL_EWOULDBLOCK */
64  case PCL_EINPROGRESS: /* connect would block, try again */
65  case PCL_EINTR: /* interrupted */
66  r = SETERR(PCL_EAGAIN);
67  break;
68 
69  default:
70  r = SETERR(err);
71  }
72 
73  return r;
74 }
