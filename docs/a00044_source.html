47 /** @defgroup errmac Error Macros
48  * Error macros allow for easily settting errors and adding traces to the per-thread stack trace.
49  * The error functions to set an error or add a trace, are cumbersome to interact with directly.
50  * They have many parameters, all of which can be easily abstracted away within macros. Note that
51  * all SET* and TRC* macros return -1 in all cases unless the error being set or traced is
52  * \c PCL_EOKAY(0).
53  * @code
54  * // when a native function fails
55  * FILE *fp = fopen(path, "r");
56  *
57  * // set last err means last OS error: typically errno or GetLastError. In addition,
58  * // we elect to set a message (MSG) as well.
59  * if(fp == NULL)
60  * SETLASTERRMSG("cannot open '%s'", path);
61  *
62  * // when a PCL function fails, or any function known to utilize PCL error
63  * FILE *fp = pcl_fopen(path, _P("r")); // pcl_fopen uses pchar_t
64  *
65  * // instead of setting an error, we add a trace and include a message.
66  * if(fp == NULL)
67  * TRCMSG("cannot open '%Ps'", path);
68  *
69  * // or, add a trace and return. In the PCL library, as well as
70  * // most unix libraries, -1 indicates an error while 0 is for success.
71  * // This is why all SET* and TRC* macros return -1.
72  * if(fp == NULL)
73  * return TRC(); // no message set here
74  *
75  * // If the function has a specific return value, use the R_SET* or R_TRC* macros
76  * if(fp == NULL)
77  * return R_TRC(NULL); // return NULL instead of -1
78  * return R_SETERR(NULL); // return NULL instead of -1
79  *
80  * // example of a PCL stack trace
81  * pcl_err_fprintf(stdout, 0, NULL); // indent=0, format_msg=NULL
82  *
83  * some.c:func(463): optional trace message <- last trace added w/msg
84  * another.c:getuserid(2029) <- trace no msg
85  * query.c:select(99): PCL_EBUF - buffer too small <- error site w/name and default msg
86  * >An optional user message goes here <- context msg set at call site
87  * @endcode
88  */
