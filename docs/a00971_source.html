40 win32_nanosleep(uint64_t ns)
41 {
42  HANDLE timer;
43  LARGE_INTEGER li;
44 
45  if(!(timer = CreateWaitableTimer(NULL, TRUE, NULL)))
46  return SETLASTERR();
47 
48  /* windows uses 100ns intervals, value must be negative -- see docs */
49  li.QuadPart = - (LONGLONG) (ns / 100);
50

56  }

65 static bool
66 sleep_cond(uint64_t nsecs, bool (*cond)(void), uint32_t cond_interval)
67 {
68  if(cond())
69  return true;
70 
71  if(cond_interval == 0)
72  cond_interval = COND_INTERVAL;
73 
74  do
75  {
76  uint64_t s = min(nsecs, cond_interval);
77 
78  /* recurse but with NULL cond callback */
79  if(pcl_sleep(s, NULL, 0))
80  return R_TRC(false);
81 
82  nsecs -= s;
83 
84  if(cond())
85  break;
86  }
87  while(nsecs > 0);
88 
89  return true;
90 }

92 bool
93 pcl_sleep(uint64_t nsecs, bool (*cond)(void), uint32_t cond_interval)
94 {
95  if(cond)
96  return sleep_cond(nsecs, cond, cond_interval);
97 
98 #ifdef PCL_WINDOWS
99  if(win32_nanosleep(nsecs))
100  return R_TRC(false);
101 
102 #else
103  struct timespec amt, remaining_amt;
104 
105  amt.tv_sec = (time_t) (nsecs / PCL_NSECS);
106  amt.tv_nsec = (long) (nsecs % PCL_NSECS);
107 
108  while(nanosleep(&amt, &remaining_amt))
109  {
110  /* failed (very unlikely) */
111  if(errno != EINTR)
112  return R_SETLASTERR(false);
113 
114  /* EINTR, continue sleeping for the remaining amount */
115  memcpy(&amt, &remaining_amt, sizeof(amt));
116  }
117 #endif
118 
119  return true;
120 }
