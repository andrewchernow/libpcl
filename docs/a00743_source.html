38 {
39  pcl_ssl_t *client;
40  pcl_socket_t *sock;
41 
42  if(!server || !(server->flags & PCL_SSL_PASSIVE))
43  return R_SETERRMSG(NULL, PCL_EINVAL, "Not a passive SSL socket", 0);
44 
45  if(!(sock = pcl_accept(server->sock)))
46  return R_TRC(NULL);
47 
48  client = pcl_zalloc(sizeof(pcl_ssl_t));
49  client->flags = PCL_SSL_SERVER | (server->flags & PCL_SOCK_NONBLOCK);
50  client->sock = sock;
51  ipcl_ssl_configure_socket(client, NULL, 0);
52 
53  client->ssl = SSL_new(server->ctx);
54  SSL_set_fd(client->ssl, (int) pcl_socket_fd(client->sock));
55 
56  return client;
57 }
