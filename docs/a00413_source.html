44 pcl_err_vjson(const char *message, va_list ap)
45 {
46  pcl_err_t *err = pcl_err_get();
47  pcl_json_t *root = pcl_json_obj();
48  uint32_t keyflags = PCL_JSON_SKIPUTF8CHK;
49  uint32_t strflags = PCL_JSON_EMPTYASNULL | PCL_JSON_ALLOWNULL;
50 
51  if(pcl_json_objputint(root, "err", err->err, keyflags) < 0)
52  FAILED;
53 
54  char *name = (char *) (err->err ? pcl_err_name(err->err) : NULL);
55 
56  if(pcl_json_objputstr(root, "name", name, keyflags | strflags) < 0)
57  FAILED;
58 
59  if(pcl_json_objputint(root, "oserr", err->oserr, keyflags) < 0)
60  FAILED;
61 
62  name = (char *) (err->oserr ? pcl_err_osname(err->oserr) : NULL);
63 
64  if(pcl_json_objputstr(root, "osname", name, keyflags | strflags) < 0)
65  FAILED;
66 
67  char *msg = NULL;
68 
69  if(!strempty(message))
70  pcl_vasprintf(&msg, message, ap);
71 
72  pcl_json_t *jmsg = pcl_json_str(msg, strflags | PCL_JSON_SHALLOW);
73 
74  if(pcl_json_objput(root, "msg", jmsg, keyflags | PCL_JSON_FREEVALONERR) < 0)
75  FAILED;
76 
77  pcl_json_t *strace = pcl_json_arr();
78 
79  if(pcl_json_objput(root, "strace", strace, keyflags | PCL_JSON_FREEVALONERR) < 0)
80  FAILED;
81 
82  for(pcl_err_trace_t *t = err->strace; t; t = t->next)
83  {
84  pcl_json_t *jtrace = pcl_json_obj();
85 
86  if(pcl_json_objputstr(jtrace, "file", t->file, keyflags | strflags) < 0)
87  {
88  pcl_json_free(jtrace);
89  FAILED;
90  }
91 
92  if(pcl_json_objputstr(jtrace, "func", t->func, keyflags | strflags) < 0)
93  {
94  pcl_json_free(jtrace);
95  FAILED;
96  }
97 
98  if(pcl_json_objputint(jtrace, "line", t->line, keyflags) < 0)
99  {
100  pcl_json_free(jtrace);
101  FAILED;
102  }
103 
104  if(pcl_json_objputstr(jtrace, "msg", t->msg, keyflags | strflags) < 0)
105  {
106  pcl_json_free(jtrace);
107  FAILED;
108  }
109 
110  if(pcl_json_arradd(strace, jtrace, keyflags | PCL_JSON_FREEVALONERR) < 0)
111  FAILED;
112  }
113 
114  return root;
115 }
