32 /* this function is hard to look at, sorry. A primary goal was to avoid
33  * allocation and make this as fast as possible. In addition, the function
34  * must deal with converting a uniform format string (PCL's input) into
35  * one of at least 3 different platform format strings...mostly due to windows
36  * changing format type specs based on printf or wprintf...OMG!
37  *
38  * Anyway, this creates a platform-specific format string given an PCL format
39  * string. The returned string might be a pointer to 'buf' (which is most likely
40  * stack) or this function might allocate a new buffer if more room is needed.
41  * The callee must check if the provided 'buf' argument address matches the
42  * returned address when this function returns. If it doesn't match, the callee
43  * must free() the returned pointer.
44  */

61 xchar *
62 XFUNC(fmtconv)(xchar *buf, size_t bufsz, const xchar *pcl_format)
63 {
64  xchar *os_format = buf;
65  xchar *out = os_format;
66  size_t len = xstrlen(pcl_format) + 1;
67 
68  /* allocate when needed */
69  if(len > bufsz)
70  {
71  os_format = (xchar *) pcl_malloc(len * sizeof(xchar));
72  out = os_format;
73  }
74 
75  while(*pcl_format)
76  {
77  xchar *p = xstrchr(pcl_format, X('%'));
78  size_t siz = len - (out - os_format);
79 
80  /* done */
81  if(!p)
82  {
83  len = xstrlen(pcl_format);
84  xstrncpy(out, siz, pcl_format, len);
85  out += len;
86  break;
87  }
88 
89  /* move past % so its included in copy */
90  p++;
91  xstrncpy(out, siz, pcl_format, p - pcl_format);
92 
93  /* adjust ptr positions */
94  out += (p - pcl_format);
95  pcl_format = p;
96 
97  /* special case, "%%" to encode a '%'. The below while loop won't handle
98  * this properly so we handle it here. If the below loop encounters a
99  * percent, it knows to exit the below loop and wrap around to the outer
100  * loop'out strchr('%').
101  */
102  if(*pcl_format == X('%'))
103  {
104  pcl_format++; /* no copy, already have 1st '%', just move past 2nd one */
105  continue;
106  }
107 
108  /* scan for end of format spec */
109  int finished = 0;
110  while(!finished && *pcl_format)
111  {
112  switch(*pcl_format)
113  {
114  case X('%'):
115  /* don't copy, no format advancement and let strchr in outer loop
116  * handle this. This is either a new format or a user-end typo.
117  */
118  finished = 1;
119  break;
120 
121  /* "%/" path sep: PCL extension */
122  case X('/'):
123  {
124  /* '/' must appear directly after '%' */
125  p = (xchar *) pcl_format - 1;
126 
127  if(*p == X('%'))
128  {
129  out--; /* overwrite previous '%' */
130  *out++ = XPATHSEPCHAR;
131  pcl_format++; /* skip the '/' */
132  }
133  else
134  {
135  /* '/' is an invalid spec char, copy it and finish */
136  finished = 1;
137  *out++ = *pcl_format++;
138  }
139 
140  break;
141  }
142 
143  /* "%Ps | %Pc" pchar_t: Portable Character String or Portable Character */
144  case X('P'):
145  {
146  xchar c = *++pcl_format;
147 
148  /* %Ps always wide on Windows, always single byte on unixes */
149  if(c == X('s') || c == X('c'))
150  {
151 #ifdef PCL_WINDOWS
152 # ifdef XWIDE
153  *out++ = c; /* lower s,c is wide for wprintf funcs */
154 # else
155  *out++ = (xchar) toupper(c); /* upper S,C is wide for printf funcs */
156 # endif
157 #else
158  *out++ = c; /* lower s,c always single byte */
159 #endif //PCL_WINDOWS
160 
161  pcl_format++; /* skip s or c */
162  }
163  else
164  {
165  *out++ = X('P'); /* consume P, invalid spec char */
166  }
167 
168  /* either had %Ps, %Pc or invalid spec char */
169  finished = 1;
170  break;
171  }
172 
173  /* convert %ls and %lc to windows format specifiers. Handle on windows
174  * only. Notice 'l' and 'h' field width fall-thru to the catch all
175  * when not on windows.
176  */
177  case X('h'):
178  case X('l'):
179 #ifdef PCL_WINDOWS
180  {
181  xchar c = pcl_format[1];
182 
183  if(c == X('s') || c == X('c'))
184  {
185 #ifdef XWIDE
186  *out++ = c; /* lower out,c is wide for wprintf funcs */
187 #else
188  *out++ = (xchar) toxupper(c); /* upper S,C is wide for printf funcs */
189 #endif
190 
191  pcl_format += 2;
192 
193  }
194  else
195  {
196  *out++ = *pcl_format++;
197  }
198 
199  break;
200  }
201 
202  case X('s'):
203  case X('c'):
204  {
205  xchar c = *pcl_format++;
206 
207 #ifdef XWIDE
208  c = (xchar)toupper(c);
209 #endif
210 
211  *out++ = c;
212  }
213 #endif //PCL_WINDOWS
214 
215  /* flags, field width or precision */
216  case X('0'):
217  case X('1'):
218  case X('2'):
219  case X('4'):
220  case X('5'):
221  case X('6'):
222  case X('7'):
223  case X('8'):
224  case X('9'):
225  case X('-'):
226  case X(' '):
227  case X('+'):
228  case X('\''):
229  case X('*'):
230  case X('.'):
231  case X('L'):
232  case X('q'):
233  case X('j'):
234  case X('z'):
235  *out++ = *pcl_format++;
236  break;
237 
238  /* we either have a conversion specifier (d, i, s, etc...) or an
239  * invalid spec char. Either way, copy char to the buffer
240  * and mark as finished.
241  */
242  default:
243  *out++ = *pcl_format++;
244  finished = 1;
245  break;
246  }
247  }
248  }
249 
250  *out = 0;
251  return os_format;
252 }
