36 int
37 ipcl_file_open(pcl_file_t *file, const pchar_t *path, int pcl_oflags, mode_t mode)
38 {
39  DWORD a = INVALID_FILE_ATTRIBUTES;
40 
41  /* -------------------------------------------------------------------------
42  * Portability Implementations for O_DIRECTORY and O_NOFOLLOW
43  */
44 
45  if(pcl_oflags & PCL_O_NOFOLLOW)
46  {
47  WIN32_FIND_DATA wfd;
48  HANDLE hfind = FindFirstFile(path, &wfd);
49 
50  if(hfind != INVALID_HANDLE_VALUE)
51  {
52  FindClose(hfind);
53  a = wfd.dwFileAttributes;
54 
55  if((a & FILE_ATTRIBUTE_REPARSE_POINT) && (wfd.dwReserved0 & IO_REPARSE_TAG_SYMLINK))
56  return SETERRMSG(PCL_ETYPE, "%Ps is a symbolic link", path);
57  }
58  }
59 
60  /* implement O_DIRECTORY: fail if file doesn't exist, fail if not a dir */
61  if(pcl_oflags & PCL_O_DIRECTORY)
62  {
63  /* don't call getfileattrs if we just called findfirstfile. if file
64  * doesn't exist, fail since O_DIRECTORY is set
65  */
66  if(a == INVALID_FILE_ATTRIBUTES && (a = GetFileAttributes(path)) == INVALID_FILE_ATTRIBUTES)
67  return SETLASTERR();
68 
69  /* file is not a dir, punt */
70  if(!(a & FILE_ATTRIBUTE_DIRECTORY))
71  return SETERR(PCL_ENOTDIR);
72  }
73 
74 
75  /* -------------------------------------------------------------------------
76  * Share Mode
77  */
78 
79  DWORD share_mode;
80 
81  if(pcl_oflags & PCL_O_SHLOCK)
82  {
83  share_mode = FILE_SHARE_READ;
84  file->flags |= PCL_FF_OLOCK;
85  }
86  else if(pcl_oflags & PCL_O_EXLOCK)
87  {
88  share_mode = 0;
89  file->flags |= PCL_FF_OLOCK;
90  }
91  else
92  {
93  share_mode = FILE_SHARE_READ | FILE_SHARE_WRITE;
94  }
95 
96 
97  /* -------------------------------------------------------------------------
98  * File Access
99  */
100 
101  DWORD access = READ_CONTROL | FILE_READ_ATTRIBUTES | GENERIC_READ;
102 
103  if(pcl_oflags & PCL_O_RDWR)
104  access |= (GENERIC_READ | GENERIC_WRITE);
105  else if(pcl_oflags & PCL_O_WRONLY)
106  access |= GENERIC_WRITE;
107 
108 
109  /* -------------------------------------------------------------------------
110  * Create Disposition
111  */
112 
113  DWORD create_mode = 0;
114 
115  switch(pcl_oflags & (PCL_O_CREAT | PCL_O_EXCL | PCL_O_TRUNC))
116  {
117  case 0:
118  case PCL_O_EXCL: /* ignore EXCL w/o CREAT */
119  create_mode = OPEN_EXISTING;
120  break;
121 
122  case PCL_O_CREAT:
123  create_mode = OPEN_ALWAYS;
124  break;
125 
126  case PCL_O_CREAT | PCL_O_EXCL:
127  case PCL_O_CREAT | PCL_O_TRUNC | PCL_O_EXCL:
128  create_mode = CREATE_NEW;
129  break;
130 
131  case PCL_O_TRUNC:
132  case PCL_O_TRUNC | PCL_O_EXCL: /* ignore EXCL w/o CREAT */
133  create_mode = TRUNCATE_EXISTING;
134  break;
135 
136  case PCL_O_CREAT | PCL_O_TRUNC:
137  create_mode = CREATE_ALWAYS;
138  break;
139 
140  default:
141  return SETERRMSG(PCL_EINVAL, "Invalid open pcl_oflags: %08x", pcl_oflags);
142  }
143 
144 
145  /* -------------------------------------------------------------------------
146  * Flags and Attributes
147  */
148 
149  DWORD attrs = 0;
150 
151  if(pcl_oflags & PCL_O_NONBLOCK)
152  attrs |= FILE_FLAG_OVERLAPPED;
153 
154  if(pcl_oflags & PCL_O_BACKUP)
155  attrs |= FILE_FLAG_BACKUP_SEMANTICS;
156 
157  if((pcl_oflags & PCL_O_CREAT) && !(mode & 0222))
158  attrs |= FILE_ATTRIBUTE_READONLY;
159  else
160  attrs |= FILE_ATTRIBUTE_NORMAL;
161 
162  if(pcl_oflags & PCL_O_TEMP)
163  {
164  attrs |= (FILE_ATTRIBUTE_TEMPORARY | FILE_FLAG_DELETE_ON_CLOSE);
165  attrs &= ~FILE_ATTRIBUTE_READONLY;
166  }
167 
168  if(pcl_oflags & PCL_O_SYNC)
169  attrs |= FILE_FLAG_WRITE_THROUGH;
170 
171 
172  /* -------------------------------------------------------------------------
173  * Security Attributes
174  */
175 
176  SECURITY_DESCRIPTOR sd;
177  SECURITY_ATTRIBUTES sa = {0};
178  sa.bInheritHandle = (pcl_oflags & PCL_O_NOINHERIT) ? false : true; /* O_CLOEXEC */
179  sa.nLength = sizeof(sa);
180 
181  /* create secdescr w/mode (use calling process owner as file owner) */
182  if(pcl_oflags & PCL_O_CREAT)
183  {
184  memset(&sd, 0, sizeof(sd));
185 
186  if(ipcl_win32_sd_create(&sd, NULL, NULL, mode))
187  return TRC();
188 
189  sa.lpSecurityDescriptor = &sd;
190  }
191 
192  /* finally, create the file */
193  file->fd = CreateFile(path, access, share_mode, &sa, create_mode, attrs, NULL);
194 
195  if(file->fd == PCL_BADFD)
196  {
197  if(pcl_oserrno == ERROR_ACCESS_DENIED)
198  {
199  /* remove readonly attribute and try to open/create file again */
200  if((a = GetFileAttributes(path)) != INVALID_FILE_ATTRIBUTES &&
201  (a & FILE_ATTRIBUTE_READONLY) && SetFileAttributes(path, a & ~FILE_ATTRIBUTE_READONLY))
202  {
203  file->fd = CreateFile(path, access, share_mode, &sa, create_mode, attrs, NULL);
204  }
205  }
206 
207  if(file->fd == PCL_BADFD)
208  {
209  if(sa.lpSecurityDescriptor)
210  ipcl_win32_sd_clear(sa.lpSecurityDescriptor);
211 
212  return SETLASTERRMSG("%Ps", path);
213  }
214  }
215 
216  if(sa.lpSecurityDescriptor)
217  ipcl_win32_sd_clear(sa.lpSecurityDescriptor);
218 
219  return 0;
220 }
