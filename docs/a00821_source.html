51 int
52 pcl_log_event_default(PCL_LOCATION_PARAMS, uint32_t level, const char *message, ...)
53 {
54  /* enable tracing except for INFO and WARN events */
55  if(level != PCL_LOGLEV_INFO && level != PCL_LOGLEV_WARN)
56  level |= PCL_LOGLEV_TOTRACE(level);
57 
58  /* The NULL value members can be omitted from declaration, since C99 states
59  * if omitted they are initialized to zero. However, this function serves
60  * as an example implementation for PCL_LOG_EVENT macro and its best to
61  * have all pcl_log_event_t members accounted for.
62  */
63  pcl_log_event_t e = {
64  /* streams */
65  .stream = stderr,
66 
67  /* location */
68  .file = file,
69  .func = func,
70  .line = line,
71 
72  /* event details */
73  .when = pcl_time(),
74  .level = level,
75  .facility = "libpcl",
76 
77  /* exit config: like panic or even fatal */
78  .exit_code = EXIT_FAILURE,
79  .exit_levels = PCL_LOGLEV_PANIC,
80  .exit_handler = default_exit_handler,
81  .exit_context = NULL
82  };
83 
84  /* proxy to pcl_log_vevent */
85  va_list ap;
86  va_start(ap, message);
87  int w = pcl_log_vevent(&e, message, ap);
88  va_end(ap);
89 
90  return w;
