38 pcl_array_insert(pcl_array_t *arr, void *elem, int index)
39 {
40  if(!arr)
41  return SETERR(PCL_EINVAL);
42 
43  if(arr->count == arr->capacity)
44  {
45  arr->capacity = arr->capacity > 0 ? arr->capacity * 2 : 8;
46  arr->elements = pcl_realloc(arr->elements, arr->capacity * sizeof(void*));
47  }
48 
49  /* memmove only required when NOT appending: ie. index '<' count */
50  if(index < arr->count)
51  memmove(arr->elements + index + 1, arr->elements + index,
52  (arr->count - index) * sizeof(void *));
53 
54  arr->elements[index] = elem;
55 
56  return ++arr->count;
57 }
