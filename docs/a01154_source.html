39 static pcl_array_t *
40 XFUNC(strsplit_impl)(const xchar *s, const xchar *delim, int casefold)
41 {
42  size_t delim_len = xstrlen(delim);
43  pcl_array_t *arr = pcl_array(4, pcl_cleanup_ptr);
44  xchar *(*fn_strstr)(const xchar *, const xchar *) = casefold ? xstristr : xstrstr;
45 
46  while(s && *s)
47  {
48  xchar *elem, *p = fn_strstr(s, delim);
49 
50  if(!p)
51  {
52  elem = xstrdup(s);
53  s = NULL;
54  }
55  else
56  {
57  elem = xstrndup(s, p - s);
58  s = p + delim_len;
59  }
60 
61  pcl_array_push(arr, elem);
62  }
63 
64  return arr;
65 }
