39 {
40  pcl_json_value_t val;
41 
42  s->ctx = s->next;
43 
44  if(!ipcl_json_skipws(s))
45  return NULL;
46 
47  s->ctx = s->next;
48 
49  switch(*s->next)
50  {
51  case '{':
52  {
53  val.type = 'o';
54 
55  if(!(val.object = ipcl_json_parse_object(s)))
56  return NULL;
57 
58  break;
59  }
60 
61  case '[':
62  {
63  val.type = 'a';
64 
65  if(!(val.array = ipcl_json_parse_array(s)))
66  return NULL;
67 
68  break;
69  }
70 
71  case '"':
72  {
73  val.type = 's';
74 
75  if(!(val.string = ipcl_json_parse_string(s)))
76  return NULL;
77 
78  break;
79  }
80 
81  case 't':
82  {
83  if(s->end - s->next < 4 || strncmp(s->next + 1, "rue", 3) != 0)
84  JSON_THROW("invalid json value", 0);
85 
86  s->next += 4;
87  val.type = 'b';
88  val.boolean = true;
89  break;
90  }
91 
92  case 'f':
93  {
94  if(s->end - s->next < 5 || strncmp(s->next + 1, "alse", 4) != 0)
95  JSON_THROW("invalid json value", 0);
96 
97  s->next += 5;
98  val.type = 'b';
99  val.boolean = false;
100  break;
101  }
102 
103  case 'n':
104  {
105  if(s->end - s->next < 4 || strncmp(s->next + 1, "ull", 3) != 0)
106  JSON_THROW("invalid json value", 0);
107 
108  s->next += 4;
109  val.type = 0;
110  break;
111  }
112 
113  case '0':
114  /* 00, 01, 02, etc. is invalid: 0.0 or just 0 is fine */
115  if(isdigit(s->next[1]))
116  JSON_THROW("invalid number value: begins with 0", 0);
117  /* fall-through */
118 
119  case '-':
120  case '.':
121  case '1':
122  case '2':
123  case '3':
124  case '4':
125  case '5':
126  case '6':
127  case '7':
128  case '8':
129  case '9':
130  {
131  char *end;
132 
133  errno = 0;
134  val.number = strtod(s->next, &end);
135 
136  /* ERANGE issues */
137  if(errno && val.number)
138  {
139  pcl_err_set(PCL_LOCATION_ARGS, pcl_err_crt2pcl(errno), 0, "invalid number value");
140  return NULL;
141  }
142

144  if(s->next == end)
145  JSON_THROW("invalid number value: no digits found", 0);
146 
147  s->next = end;
148  val.type = 'n';
149  break;
150  }
151 
152  default:
153  JSON_THROW("expected json value", 0);
154  }
155 
156  /* valbuf is only passed in by ipcl_json_parse_array since vector copies elements */
157  pcl_json_value_t *v = valbuf ? valbuf : pcl_malloc(sizeof(pcl_json_value_t));
158 
159  return memcpy(v, &val, sizeof(val));
