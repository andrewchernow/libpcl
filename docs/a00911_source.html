41 static void
42 convert_times(pcl_stat_t *buf, sys_stat_t *os_buf)
43 {
44  /* Linux stat uses a macro to provide backward compatability with the old st_[a|m|c]time naming.
45  * If that macro is defined, we are being compiled to use st_atim vs. st_atimensec.
46  *
47  * Note: Darwin has both but the way we compile it uses the xtimespec struct member.
48  */
49 
50  /* convert to nanoseconds */
51  buf->atime.sec = os_buf->st_atime;
52 #ifdef PCL_DARWIN
53  buf->atime.nsec = (int) os_buf->st_atimespec.tv_nsec;
54 #elif defined(st_atime)
55  buf->atime.nsec = (int) os_buf->st_atim.tv_nsec;
56 #else
57  buf->atime.nsec = (int) os_buf->st_atimensec;
58 #endif
59 
60  /* convert to nanoseconds */
61  buf->mtime.sec = os_buf->st_mtime;
62 #ifdef PCL_DARWIN
63  buf->mtime.nsec = (int) os_buf->st_mtimespec.tv_nsec;
64 #elif defined(st_mtime)
65  buf->mtime.nsec = (int) os_buf->st_mtim.tv_nsec;
66 #else
67  buf->mtime.nsec = (int) os_buf->st_mtimensec;
68 #endif
69 
70  /* convert to nanoseconds */
71  buf->ctime.sec = os_buf->st_ctime;
72 #ifdef PCL_DARWIN
73  buf->ctime.nsec = (int) os_buf->st_ctimespec.tv_nsec;
74 #elif defined(st_ctime)
75  buf->ctime.nsec = (int) os_buf->st_ctim.tv_nsec;
76 #else
77  buf->ctime.nsec = (int) os_buf->st_ctimensec;
78 #endif
79 
80 #ifdef PCL_DARWIN
81  buf->crtime.sec = os_buf->st_birthtimespec.tv_sec;
82  buf->crtime.nsec = (int) os_buf->st_birthtimespec.tv_nsec;
83 #endif
84 }

86 int
87 ipcl_statent(const pchar_t *path, int fd, pcl_stat_t *buf, int flags)
88 {
89  int n;
90  sys_stat_t os_buf;
91 
92  if(strempty(path) && fd < 0)
93  return BADARG();
94 
95  if(fd > -1)
96  n = sys_fstat(fd, &os_buf);
97  else if(flags & PCL_STAT_LINK)
98  n = sys_lstat(path, &os_buf);
99  else
100  n = sys_stat(path, &os_buf);
101 
102  if(n)
103  return SETLASTERR();
104 
105  if(!buf)
106  return 0; // exists check only
107 
108  buf->size = (int64_t) os_buf.st_size;
109  buf->rdev = os_buf.st_rdev;
110  buf->dev = os_buf.st_dev;
111  buf->ino = os_buf.st_ino;
112  buf->uid = os_buf.st_uid;
113  buf->gid = os_buf.st_gid;
114  buf->nlink = os_buf.st_nlink;
115  buf->mode = os_buf.st_mode;
116  buf->blksize = os_buf.st_blksize;
117  buf->blocks = os_buf.st_blocks;
118 
119  if(path)
120  {
121  const pchar_t *name = pcl_pcsrchr(path, PCL_PPATHSEPCHAR);
122 
123  if(name)
124  name++;
125  else
126  name = path;
127 
128  if(*name == '.')
129  buf->flags |= UF_HIDDEN;
130  }
131 
132  convert_times(buf, &os_buf);
133 
134  /* grab backup time (btime) and finder info: The below uses fgetattrlist if 'fd' is
135  * valid, otherwise getattrlist using 'path'
136  */
137 #ifdef PCL_DARWIN
138  buf->flags |= os_buf.st_flags;
139 
140  struct
141  {
142  u_int32_t length;
143  struct timespec bcktime;
144  char finder_info[32];
145  } __attribute__ ((__packed__)) info;
146 
147  struct attrlist attr;
148  unsigned long opts = (flags & PCL_STAT_LINK) ? FSOPT_NOFOLLOW : 0;
149 
150  memset(&attr, 0, sizeof(attr));
151  attr.bitmapcount = ATTR_BIT_MAP_COUNT;
152  attr.commonattr = ATTR_CMN_BKUPTIME | ATTR_CMN_FNDRINFO;
153 
154  if((fd >= 0 && !fgetattrlist(fd, &attr, &info, sizeof(info), opts)) ||
155  (path && !getattrlist(path, &attr, &info, sizeof(info), opts)))
156  {
157  buf->btime.sec = info.bcktime.tv_sec;
158  buf->btime.nsec = (int) info.bcktime.tv_nsec;
159  memcpy(buf->finder_info, info.finder_info, 32);
160  }
161 #endif
162 
163  return 0;
164 }
