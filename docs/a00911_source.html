43 void
44 ipcl_sysinfo(pcl_sysinfo_t *info, void *_uts)
45 {
46  size_t n;
47  FILE *fp;
48  char buf[4096];
49  const char *codename = "";
50  struct utsname *uts = _uts;
51  char prodname[80], prodver[40], prodbuild[40];
52 
53  pcl_strcpy(info->vendor, sizeof(info->vendor), "Apple Inc.");
54 
55  /* This is the only way to get Marketing Version (10.8.3). It also outputs product name and
56  * product build so we might as well capture them, eventhough there are other ways. 'sw_vers'
57  * has been available since 10.4.
58  */
59  if((fp = popen("sw_vers", "r")))
60  {
61  while(fgets(buf, sizeof(buf), fp))
62  {
63  /* Product Name: Mac OS X, macOS, macOS Server, etc... */
64  if(!pcl_strnicmp(buf, PRODNAME, sizeof(PRODNAME) - 1))
65  strcpy(prodname, pcl_strtrim(buf + (sizeof(PRODNAME) - 1)));
66  /* Product (marketing) version: 10.8.3 */
67  else if(!pcl_strnicmp(buf, PRODVER, sizeof(PRODVER) - 1))
68  strcpy(prodver, pcl_strtrim(buf + (sizeof(PRODVER) - 1)));
69  /* OS Build: 12D78 */
70  else if(!pcl_strnicmp(buf, BUILDVER, sizeof(BUILDVER) - 1))
71  strcpy(prodbuild, pcl_strtrim(buf + (sizeof(BUILDVER) - 1)));
72  }
73 
74  pclose(fp);
75  }
76 
77  char verbuf[20], *p;
78 
79  strcpy(verbuf, prodver);
80 
81  p = strrchr(verbuf, '.');
82  memmove(p, p + 1, strlen(p + 1) + 1);
83 
84  double ver = strtod(verbuf, NULL);
85 
86  /* Leopard was Oct 2007, no need to go back any further */
87  if(ver >= 10.5 && ver < 10.6)
88  codename = "Leopard ";
89  else if(ver >= 10.6 && ver < 10.7)
90  codename = "Snow Leopard ";
91  else if(ver >= 10.7 && ver < 10.8)
92  codename = "Lion ";
93  else if(ver >= 10.8 && ver < 10.9)
94  codename = "Mountain Lion ";
95  else if(ver >= 10.9 && ver < 10.10)
96  codename = "Mavericks ";
97  else if(ver >= 10.10 && ver < 10.11)
98  codename = "Yosemite ";
99  else if(ver >= 10.11 && ver < 10.12)
100  codename = "El Capitan ";
101  else if(ver >= 10.12 && ver < 10.13)
102  codename = "Sierra ";
103  else if(ver >= 10.13 && ver < 10.14)
104  codename = "High Sierra ";
105  else if(ver >= 10.14 && ver < 10.15)
106  codename = "Mojave ";
107  else if(ver >= 10.15 && ver < 10.16)
108  codename = "Catalina ";
109 
110  /* Example: Mac OS X Mountain Lion 10.8.3/12.3.0 (Build 12D78) */
111  sprintf(info->version, "%s %s%s/%s (Build %s)", prodname, codename,
112  prodver, uts->release, prodbuild);
113 
114  /* total physical memory on system */
115  n = sizeof(info->memory);
116  (void) sysctlbyname("hw.memsize", &info->memory, &n, NULL, 0);
117 
118  /* number of processor sockets (physical CPUs) */
119  n = 4;
120  (void) sysctlbyname("hw.packages", &info->cpu_sockets, &n, NULL, 0);
121 
122  /* number of processor cores/threads */
123  n = 4;
124  (void) sysctlbyname("hw.logicalcpu", &info->cpu_cores, &n, NULL, 0);
125 
126  /* Processor Brand: Intel(R) Core(TM) i7-3820QM CPU @ 2.70GHz */
127  *buf = 0;
128  n = sizeof(info->cpu_model);
129  (void) sysctlbyname("machdep.cpu.brand_string", info->cpu_model, &n, NULL, 0);
130 }
