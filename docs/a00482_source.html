36 pcl_file_write(pcl_file_t *file, const void *data, size_t count)
37 {
38  if(!file || !data)
39  return BADARG();
40 
41  if(!(file->flags & PCL_FF_WRITE))
42  return SETERRMSG(PCL_ENOTSUP, "file not opened for write operations", 0);
43 
44  if(file->flags & PCL_FF_NONBLOCK)
45  {
46  int r = pcl_file_trywrite(file, data, count, 0);
47  return r == -1 ? TRC() : r;
48  }
49 
50  DWORD written = 0;
51 
52  SetLastError(0);
53 
54  if(WriteFile(file->fd, data, (DWORD) count, &written, NULL))
55  return (int) written;
56 
57  return SETLASTERR();
58 }
