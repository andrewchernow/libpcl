53 
54 #ifdef PCL_WINDOWS
55  /* query for size: len will include NUL byte in this case */
56  DWORD len = GetFullPathName(path, 0, NULL, NULL);
57 
58  if(len == 0)
59  return R_SETLASTERR(NULL);
60 
61  fullpath = pcl_malloc(len * sizeof(pchar_t));
62 
63  /* call again with allocated buffer */
64  len = GetFullPathName(path, len, fullpath, NULL);
65 
66  if(len == 0)
67  return R_SETLASTERR(NULL);
68 
69 #else
70  char pbuf[PATH_MAX];
71 
72  if(!realpath((char *) path, pbuf))
73  return R_SETLASTERR(NULL);
74 
75  size_t len = strlen(pbuf);
76  fullpath = pcl_strndup(pbuf, len);
77 #endif
78 
79  if(lenp)
80  *lenp = (int) len;
81 
82  return fullpath;
